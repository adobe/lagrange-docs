44template <typename Scalar, typename Index, int NVPF = 3, int DIM = 3, int UV_DIM = 2>
45lagrange::TangentBitangentResult compute_tangent_bitangent_mikktspace(
46 lagrange::SurfaceMesh<Scalar, Index>& mesh,
47 lagrange::TangentBitangentOptions options = {})
48{
49 la_runtime_assert(mesh.get_dimension() == DIM, "Mesh must be 3D");
50 la_runtime_assert(mesh.is_triangle_mesh(), "Input must be a triangle mesh");
51 static_assert(NVPF == 3);
52 static_assert(DIM == 3);
53 static_assert(UV_DIM == 2);
54
55 lagrange::TangentBitangentResult result;
56
57 auto uv_id = internal::find_matching_attribute<Scalar>(
58 mesh,
59 options.uv_attribute_name,
60 AttributeElement::Indexed,
61 AttributeUsage::UV,
62 UV_DIM);
63 la_runtime_assert(uv_id != invalid_attribute_id(), "Mesh must have indexed UVs");
64 auto normal_id = internal::find_matching_attribute<Scalar>(
65 mesh,
66 options.normal_attribute_name,
67 AttributeElement::Indexed,
68 AttributeUsage::Normal,
69 DIM);
70 la_runtime_assert(normal_id != invalid_attribute_id(), "Mesh must have indexed normals");
71
72 const size_t num_channels = (options.pad_with_sign ? 4 : 3);
73
74 result.tangent_id = internal::find_or_create_attribute<Scalar>(
75 mesh,
76 options.tangent_attribute_name,
77 AttributeElement::Corner,
78 AttributeUsage::Tangent,
79 num_channels,
80 internal::ResetToDefault::No);
81
82 result.bitangent_id = internal::find_or_create_attribute<Scalar>(
83 mesh,
84 options.bitangent_attribute_name,
85 AttributeElement::Corner,
86 AttributeUsage::Bitangent,
87 num_channels,
88 internal::ResetToDefault::No);
89
90 struct LocalData
91 {
92 int num_facets;
93 int num_channels;
94
95 span<const Scalar> position_values;
96 span<const Index> position_indices;
97
98 span<const Scalar> normal_values;
99 span<const Index> normal_indices;
100
101 span<const Scalar> uv_values;
102 span<const Index> uv_indices;
103
104 span<Scalar> tangents;
105 span<Scalar> bitangents;
106 };
107
108 LocalData data;
109 data.num_facets = lagrange::safe_cast<int>(mesh.get_num_facets());
110 data.num_channels = static_cast<int>(num_channels);
111 data.position_values = mesh.get_vertex_to_position().get_all();
112 data.position_indices = mesh.get_corner_to_vertex().get_all();
113 data.normal_values = mesh.template get_indexed_attribute<Scalar>(normal_id).values().get_all();
114 data.normal_indices =
115 mesh.template get_indexed_attribute<Scalar>(normal_id).indices().get_all();
116 data.uv_values = mesh.template get_indexed_attribute<Scalar>(uv_id).values().get_all();
117 data.uv_indices = mesh.template get_indexed_attribute<Scalar>(uv_id).indices().get_all();
118 data.tangents = mesh.template ref_attribute<Scalar>(result.tangent_id).ref_all();
119 data.bitangents = mesh.template ref_attribute<Scalar>(result.bitangent_id).ref_all();
120
121 SMikkTSpaceInterface mk_interface;
122 mk_interface.m_getNumFaces = [](const SMikkTSpaceContext* pContext) {
123 auto _data = reinterpret_cast<const LocalData*>(pContext->m_pUserData);
124 return _data->num_facets;
125 };
126 mk_interface.m_getNumVerticesOfFace = [](const SMikkTSpaceContext* pContext, const int iFace) {
127 (void)pContext;
128 (void)iFace;
129 return NVPF;
130 };
131 mk_interface.m_getPosition =
132 [](const SMikkTSpaceContext* pContext, float fvPosOut[], const int iFace, const int iVert) {
133 auto _data = reinterpret_cast<const LocalData*>(pContext->m_pUserData);
134 const Index v = _data->position_indices[iFace * NVPF + iVert];
135 auto pos = _data->position_values.subspan(v * DIM, DIM);
136 fvPosOut[0] = static_cast<float>(pos[0]);
137 fvPosOut[1] = static_cast<float>(pos[1]);
138 fvPosOut[2] = static_cast<float>(pos[2]);
139 };
140 mk_interface.m_getNormal = [](const SMikkTSpaceContext* pContext,
141 float fvNormOut[],
142 const int iFace,
143 const int iVert) {
144 auto _data = reinterpret_cast<const LocalData*>(pContext->m_pUserData);
145 const Index v = _data->normal_indices[iFace * NVPF + iVert];
146 auto nrm = _data->normal_values.subspan(v * DIM, DIM);
147 fvNormOut[0] = static_cast<float>(nrm[0]);
148 fvNormOut[1] = static_cast<float>(nrm[1]);
149 fvNormOut[2] = static_cast<float>(nrm[2]);
150 };
151 mk_interface.m_getTexCoord = [](const SMikkTSpaceContext* pContext,
152 float fvTexcOut[],
153 const int iFace,
154 const int iVert) {
155 auto _data = reinterpret_cast<const LocalData*>(pContext->m_pUserData);
156 const Index v = _data->uv_indices[iFace * NVPF + iVert];
157 auto uv = _data->uv_values.subspan(v * UV_DIM, UV_DIM);
158 fvTexcOut[0] = static_cast<float>(uv[0]);
159 fvTexcOut[1] = static_cast<float>(uv[1]);
160 };
161 mk_interface.m_setTSpaceBasic = [](const SMikkTSpaceContext* pContext,
162 const float fvTangent[],
163 const float fSign,
164 const int iFace,
165 const int iVert) {
166 auto _data = reinterpret_cast<LocalData*>(pContext->m_pUserData);
167 auto tangent =
168 _data->tangents.subspan((iFace * NVPF + iVert) * _data->num_channels, _data->num_channels);
169 tangent[0] = static_cast<Scalar>(fvTangent[0]);
170 tangent[1] = static_cast<Scalar>(fvTangent[1]);
171 tangent[2] = static_cast<Scalar>(fvTangent[2]);
172 if (tangent.size() == 4) {
173 tangent[3] = static_cast<Scalar>(fSign);
174 }
175 };
176 mk_interface.m_setTSpace = [](const SMikkTSpaceContext* pContext,
177 const float fvTangent[],
178 const float fvBiTangent[],
179 const float fMagS,
180 const float fMagT,
181 const tbool bIsOrientationPreserving,
182 const int iFace,
183 const int iVert) {
184 (void)fMagS;
185 (void)fMagT;
186 auto _data = reinterpret_cast<LocalData*>(pContext->m_pUserData);
187 const float fSign = bIsOrientationPreserving ? 1.0f : (-1.0f);
188
189 auto tangent =
190 _data->tangents.subspan((iFace * NVPF + iVert) * _data->num_channels, _data->num_channels);
191 tangent[0] = static_cast<Scalar>(fvTangent[0]);
192 tangent[1] = static_cast<Scalar>(fvTangent[1]);
193 tangent[2] = static_cast<Scalar>(fvTangent[2]);
194 if (tangent.size() == 4) {
195 tangent[3] = static_cast<Scalar>(fSign);
196 }
197
198 auto bitangent = _data->bitangents.subspan(
199 (iFace * NVPF + iVert) * _data->num_channels,
200 _data->num_channels);
201 bitangent[0] = static_cast<Scalar>(fvBiTangent[0]);
202 bitangent[1] = static_cast<Scalar>(fvBiTangent[1]);
203 bitangent[2] = static_cast<Scalar>(fvBiTangent[2]);
204 if (bitangent.size() == 4) {
205 bitangent[3] = static_cast<Scalar>(fSign);
206 }
207 };
208
209 SMikkTSpaceContext context;
210 context.m_pInterface = &mk_interface;
211 context.m_pUserData = reinterpret_cast<void*>(&data);
212
213 lagrange::logger().debug("run mikktspace");
214 genTangSpaceDefault(&context);
215 lagrange::logger().debug("done");
216
217 return result;
218}
