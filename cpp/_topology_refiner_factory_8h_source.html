14// Specify the number of vertices, faces, face-vertices, etc.
15template <>
16bool TopologyRefinerFactory<ConverterType>::resizeComponentTopology(
17 TopologyRefiner& refiner,
18 const ConverterType& conv)
19{
20 const auto& mesh = conv.mesh;
21
22 // Number of vertices
23 int num_vertices = static_cast<int>(mesh.get_num_vertices());
24 setNumBaseVertices(refiner, num_vertices);
25
26 // Number of faces and face-vertices (corners)
27 int num_facets = static_cast<int>(mesh.get_num_facets());
28 setNumBaseFaces(refiner, num_facets);
29 for (int facet = 0; facet < num_facets; ++facet) {
30 int nv = static_cast<int>(mesh.get_facet_size(facet));
31 setNumBaseFaceVertices(refiner, facet, nv);
32 }
33
34 return true;
35}

37// Specify the relationships between vertices, faces, etc. ie the face-vertices, vertex-faces,
38// edge-vertices, etc.
39template <>
40bool TopologyRefinerFactory<ConverterType>::assignComponentTopology(
41 TopologyRefiner& refiner,
42 const ConverterType& conv)
43{
44 const auto& mesh = conv.mesh;
45 using Far::IndexArray;
46
47 // Face relations
48 int num_facets = static_cast<int>(mesh.get_num_facets());
49 for (int facet = 0; facet < num_facets; ++facet) {
50 IndexArray dst_facet_vertices = getBaseFaceVertices(refiner, facet);
51
52 auto fv = mesh.get_facet_vertices(facet);
53 for (size_t lv = 0; lv < fv.size(); ++lv) {
54 dst_facet_vertices[lv] = static_cast<int>(fv[lv]);
55 }
56 }
57
58 // If we have any non-manifold vertices/edges, we need to set their tags before calling this function
59 populateBaseLocalIndices(refiner);
60
61 return true;
62};

64// (Optional) Specify edge or vertex sharpness or face holes.
65template <>
66bool TopologyRefinerFactory<ConverterType>::assignComponentTags(
67 TopologyRefiner& refiner,
68 const ConverterType& conv)
69{
70 const auto& mesh = conv.mesh;
71 const auto& options = conv.options;
72
73 if (options.edge_sharpness_attr.has_value()) {
74 lagrange::subdivision::visit_attribute(
75 mesh,
76 options.edge_sharpness_attr.value(),
77 [&](auto&& attr) {
78 using AttributeType = std::decay_t<decltype(attr)>;
79 using ValueType = typename AttributeType::ValueType;
80 la_runtime_assert(attr.get_num_channels() == 1);
81 la_runtime_assert(
82 std::is_floating_point_v<ValueType>,
83 fmt::format(
84 "Edge sharpness attribute must use a floating point type. Received: {}",
85 lagrange::internal::value_type_name<ValueType>()));
86 la_runtime_assert(attr.get_element_type() == lagrange::AttributeElement::Edge);
87 if constexpr (AttributeType::IsIndexed) {
88 la_runtime_assert("Edge sharpness cannot be an indexed attribute");
89 } else {
90 auto values = attr.get_all();
91 for (int e = 0; e < static_cast<int>(values.size()); ++e) {
92 auto v = mesh.get_edge_vertices(e);
93 Index idx = findBaseEdge(refiner, v[0], v[1]);
94
95 if (idx != INDEX_INVALID) {
96 float s = static_cast<float>(ValueType(10.0) * values[e]);
97 setBaseEdgeSharpness(refiner, idx, s);
98 } else {
99 char msg[1024];
100 snprintf(
101 msg,
102 1024,
103 "Edge %d specified to be sharp does not exist (%d, %d)",
104 e,
105 static_cast<int>(v[0]),
106 static_cast<int>(v[1]));
107 reportInvalidTopology(
108 Vtr::internal::Level::TOPOLOGY_INVALID_CREASE_EDGE,
109 msg,
110 conv);
111 }
112 }
113 }
114 });
115 }
116
117 if (options.vertex_sharpness_attr.has_value()) {
118 lagrange::subdivision::visit_attribute(
119 mesh,
120 options.vertex_sharpness_attr.value(),
121 [&](auto&& attr) {
122 using AttributeType = std::decay_t<decltype(attr)>;
123 using ValueType = typename AttributeType::ValueType;
124 la_runtime_assert(attr.get_num_channels() == 1);
125 la_runtime_assert(attr.get_element_type() == lagrange::AttributeElement::Vertex);
126 la_runtime_assert(
127 std::is_floating_point_v<ValueType>,
128 fmt::format(
129 "Vertex sharpness attribute must use a floating point type. Received: {}",
130 lagrange::internal::value_type_name<ValueType>()));
131 if constexpr (AttributeType::IsIndexed) {
132 la_runtime_assert("Vertex sharpness cannot be an indexed attribute");
133 } else {
134 auto values = attr.get_all();
135 for (int v = 0; v < static_cast<int>(values.size()); ++v) {
136 float s = static_cast<float>(ValueType(10.0) * values[v]);
137 setBaseVertexSharpness(refiner, v, s);
138 }
139 }
140 });
141 }
142
143 if (options.face_hole_attr.has_value()) {
144 lagrange::subdivision::visit_attribute(
145 mesh,
146 options.face_hole_attr.value(),
147 [&](auto&& attr) {
148 using AttributeType = std::decay_t<decltype(attr)>;
149 using ValueType = typename AttributeType::ValueType;
150 la_runtime_assert(attr.get_num_channels() == 1);
151 la_runtime_assert(
152 std::is_integral_v<ValueType>,
153 fmt::format(
154 "Face holes attribute must use an integral type. Received: {}",
155 lagrange::internal::value_type_name<ValueType>()));
156 if constexpr (AttributeType::IsIndexed) {
157 la_runtime_assert("Face holes cannot be an indexed attribute");
158 } else {
159 auto values = attr.get_all();
160 for (int f = 0; f < static_cast<int>(values.size()); ++f) {
161 if (values[f] != ValueType(0)) {
162 lagrange::logger().warn("Setting facet f{} as a hole", f);
163 setBaseFaceHole(refiner, f, true);
164 }
165 }
166 }
167 });
168 }
169
170 return true;
171}

174// (Optional) Specify face-varying data per face.
175template <>
176bool TopologyRefinerFactory<ConverterType>::assignFaceVaryingTopology(
177 TopologyRefiner& refiner,
178 const ConverterType& conv)
179{
180 const auto& mesh = conv.mesh;
181
182 // TODO: Only define one fvar channel for each different set of indices (factorize shared set of
183 // indices)?
184 for (lagrange::AttributeId attr_id : conv.face_varying_attributes) {
185 lagrange::subdivision::visit_attribute(mesh, attr_id, [&](auto&& attr) {
186 using AttributeType = std::decay_t<decltype(attr)>;
187 if constexpr (!AttributeType::IsIndexed) {
188 la_runtime_assert(
189 false,
190 fmt::format(
191 "Face varying attributes must indexed attributes. Received: {}",
192 lagrange::internal::to_string(attr.get_element_type())));
193 } else {
194 int num_values = static_cast<int>(attr.values().get_num_elements());
195 auto src_indices = attr.indices().get_all();
196
197 int channel = createBaseFVarChannel(refiner, num_values);
198
199 for (int f = 0, src_next = 0; f < static_cast<int>(mesh.get_num_facets()); ++f) {
200 IndexArray dst_indices = getBaseFaceFVarValues(refiner, f, channel);
201
202 for (int lv = 0; lv < dst_indices.size(); ++lv) {
203 dst_indices[lv] = src_indices[src_next++];
204 }
205 }
206 }
207 });
208 }
209 return true;
210}
