39namespace detail {
40std::array<std::array<float, 4>, 4> affine3d_to_array(const Eigen::Affine3f& t)
41{
42 std::array<std::array<float, 4>, 4> data;
43 for (int i = 0; i < 4; ++i) {
44 for (int j = 0; j < 4; ++j) {
45 data[i][j] = t(i, j);
46 }
47 }
48 return data;
49};
50Eigen::Affine3f array_to_affine3d(const std::array<std::array<float, 4>, 4> data)
51{
52 Eigen::Affine3f t;
53 for (int i = 0; i < 4; ++i) {
54 for (int j = 0; j < 4; ++j) {
55 t(i, j) = data[i][j];
56 }
57 }
58 return t;
59};
60} // namespace detail
