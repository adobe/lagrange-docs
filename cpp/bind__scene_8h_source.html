36namespace detail {
37std::array<std::array<float, 4>, 4> affine3d_to_array(const Eigen::Affine3f& t)
38{
39 std::array<std::array<float, 4>, 4> data;
40 for (int i = 0; i < 4; ++i) {
41 for (int j = 0; j < 4; ++j) {
42 data[i][j] = t(i, j);
43 }
44 }
45 return data;
46};
47Eigen::Affine3f array_to_affine3d(const std::array<std::array<float, 4>, 4> data)
48{
49 Eigen::Affine3f t;
50 for (int i = 0; i < 4; ++i) {
51 for (int j = 0; j < 4; ++j) {
52 t(i, j) = data[i][j];
53 }
54 }
55 return t;
56};
57} // namespace detail
