186template <typename T>
187bool ImageView<T>::resize(size_t width, size_t height, size_t alignment)
188{
189 if (0 == width || 0 == height) {
190 reset();
191 return false;
192 } else {
193 m_storage = std::make_shared<ImageStorage>(width * sizeof(T), height, alignment);
194 m_view_size = {width, height};
195 m_view_stride_in_byte(0) = sizeof(T);
196 m_view_stride_in_byte(1) = m_storage->get_full_stride();
197 m_view_offset_in_byte.setZero();
198 return true;
199 }
200}

202template <typename T>
203bool ImageView<T>::view(
204 std::shared_ptr<ImageStorage> _storage,
205 size_t width,
206 size_t height,
207 size_t stride_0_in_byte,
208 size_t stride_1_in_row,
209 size_t offset_0_in_byte,
210 size_t offset_1_in_row)
211{
212 if (_storage && 0 < width && 0 < height) {
213 auto full_size = _storage->get_full_size();
214 auto full_stride = _storage->get_full_stride();
215 if (sizeof(T) <= stride_0_in_byte &&
216 offset_0_in_byte + stride_0_in_byte * width <= full_size(0) && 0 < stride_1_in_row &&
217 offset_1_in_row + stride_1_in_row * height <= full_size(1)) {
218 m_storage = _storage;
219 m_view_size = {width, height};
220 m_view_stride_in_byte = {stride_0_in_byte, stride_1_in_row * full_stride};
221 m_view_offset_in_byte = {offset_0_in_byte, offset_1_in_row * full_stride};
222 return true;
223 }
224 }
225 reset();
226 return false;
227}

229template <typename T>
230ImageStorage::aligned_vector<unsigned char> ImageView<T>::pack() const
231{
232 ImageStorage::aligned_vector<unsigned char> buf(sizeof(T) * m_view_size(0) * m_view_size(1));
233 if (is_compact()) {
234 std::copy_n(
235 reinterpret_cast<const unsigned char*>(&(operator()(0, 0))),
236 buf.size(),
237 buf.data());
238 } else if (sizeof(T) == m_view_stride_in_byte(0)) {
239 auto src = reinterpret_cast<const unsigned char*>(&(operator()(0, 0)));
240 auto dst = buf.data();
241 auto dst_stride_in_byte_row = sizeof(T) * m_view_size(0);
242 for (size_t row = 0; row < m_view_size(1); ++row) {
243 std::copy_n(src, dst_stride_in_byte_row, dst);
244 src += m_view_stride_in_byte(1);
245 dst += dst_stride_in_byte_row;
246 }
247 } else {
248 auto src_row = reinterpret_cast<const unsigned char*>(&(operator()(0, 0)));
249 auto dst_row = buf.data();
250 auto dst_stride_in_byte_col = sizeof(T);
251 auto dst_stride_in_byte_row = sizeof(T) * m_view_size(0);
252 for (size_t row = 0; row < m_view_size(1); ++row) {
253 auto src_pix = src_row;
254 auto dst_pix = dst_row;
255 for (size_t col = 0; col < m_view_size(0); ++col) {
256 *reinterpret_cast<T*>(dst_pix) = *reinterpret_cast<const T*>(src_pix);
257 src_pix += m_view_stride_in_byte(0);
258 dst_pix += dst_stride_in_byte_col;
259 }
260 src_row += m_view_stride_in_byte(1);
261 dst_row += dst_stride_in_byte_row;
262 }
263 }
264 return buf;
265}
266template <typename T>
267bool ImageView<T>::unpack(const ImageStorage::aligned_vector<unsigned char>& buf)
268{
269 if (sizeof(T) * m_view_size(0) * m_view_size(1) != buf.size()) {
270 return false;
271 }
272
273 if (is_compact()) {
274 std::copy_n(buf.data(), buf.size(), reinterpret_cast<unsigned char*>(&(operator()(0, 0))));
275 } else if (sizeof(T) == m_view_stride_in_byte(0)) {
276 auto dst = reinterpret_cast<unsigned char*>(&(operator()(0, 0)));
277 auto src = buf.data();
278 auto src_stride_in_byte_row = sizeof(T) * m_view_size(0);
279 for (size_t row = 0; row < m_view_size(1); ++row) {
280 std::copy_n(src, src_stride_in_byte_row, dst);
281 dst += m_view_stride_in_byte(1);
282 src += src_stride_in_byte_row;
283 }
284 } else {
285 auto dst_row = reinterpret_cast<unsigned char*>(&(operator()(0, 0)));
286 auto src_row = buf.data();
287 auto src_stride_in_byte_col = sizeof(T);
288 auto src_stride_in_byte_row = sizeof(T) * m_view_size(0);
289 for (size_t row = 0; row < m_view_size(1); ++row) {
290 auto dst_pix = dst_row;
291 auto src_pix = src_row;
292 for (size_t col = 0; col < m_view_size(0); ++col) {
293 *reinterpret_cast<T*>(dst_pix) = *reinterpret_cast<const T*>(src_pix);
294 dst_pix += m_view_stride_in_byte(0);
295 src_pix += src_stride_in_byte_col;
296 }
297 dst_row += m_view_stride_in_byte(1);
298 src_row += src_stride_in_byte_row;
299 }
300 }
301 return true;

304template <typename T>
305template <typename S, typename CONVERTOR>
306bool ImageView<T>::convert_from(
307 const ImageView<S>& other,
308 size_t alignment,
309 const CONVERTOR& convertor)
310{
311 const auto other_view_size = other.get_view_size();
312 if (!resize(other_view_size(0), other_view_size(1), alignment)) {
313 return false;
314 }
315
316 const char* src = reinterpret_cast<const char*>(&(other(0, 0)));
317 char* dst = reinterpret_cast<char*>(&(operator()(0, 0)));
318 const auto other_view_stride_in_byte = other.get_view_stride_in_byte();
319 tbb::parallel_for(static_cast<size_t>(0), other_view_size(1), [&](size_t y) {
320 const char* src_row = src + other_view_stride_in_byte(1) * y;
321 char* dst_row = dst + m_view_stride_in_byte(1) * y;
322 for (size_t x = 0; x < other_view_size(0); ++x) {
323 const S* src_pix =
324 reinterpret_cast<const S*>(src_row + other_view_stride_in_byte(0) * x);
325 T* dst_pix = reinterpret_cast<T*>(dst_row + m_view_stride_in_byte(0) * x);
326 convertor(*src_pix, *dst_pix);
327 }
328 });
329 return true;
330}

339template <typename T>
340void ImageView<T>::clear(const T val)
341{
342 tbb::parallel_for(static_cast<size_t>(0), m_view_size(1), [&](size_t h) {
343 auto* ptr = reinterpret_cast<unsigned char*>(&((*this)(0, h)));
344 for (size_t w = 0; w < m_view_size(0); ++w) {
345 auto* p = reinterpret_cast<T*>(ptr + w * m_view_stride_in_byte(0));
346 *p = val;
347 }
348 });
349}

351template <typename T>
352T& ImageView<T>::operator()(size_t x, size_t y)
353{
354 auto total_offset = m_view_offset_in_byte(0) + m_view_offset_in_byte(1) +
355 m_view_stride_in_byte(0) * x + m_view_stride_in_byte(1) * y;
356 return *reinterpret_cast<T*>(m_storage->data() + total_offset);
357}
