102 la_debug_assert(dynamic_cast<Derived*>(m_data.get()));
103#endif
104 auto ret = std::make_shared<Derived>(std::move(*ptr));
105 m_data.reset();
106 return ret;
107 }

144#if LAGRANGE_TARGET_FEATURE(RTTI)
