145 template <typename VALUE_SRC, typename VALUE_DST>
146 inline VALUE_DST convert_channel_value(VALUE_SRC val)
147{
148 // convert between same type
149 if constexpr (std::is_same<VALUE_SRC, VALUE_DST>::value) {
150 return val;
151 }
152 // convert from unsigned char and float/double
153 else if constexpr (
154 std::is_same<VALUE_SRC, unsigned char>::value && std::is_floating_point<VALUE_DST>::value) {
155 return static_cast<VALUE_DST>(val) /
156 static_cast<VALUE_DST>(std::numeric_limits<unsigned char>::max());
157 }
158 // convert from float/double to unsigned char
159 else if constexpr (
160 std::is_floating_point<VALUE_SRC>::value && std::is_same<VALUE_DST, unsigned char>::value) {
161 return static_cast<unsigned char>(
162 std::clamp(val, static_cast<VALUE_SRC>(0), static_cast<VALUE_SRC>(1)) *
163 static_cast<VALUE_SRC>(std::numeric_limits<unsigned char>::max()));
164 } else {
165 // clamping, prepare to convert from signed to unsigned
166 if constexpr (std::is_signed<VALUE_SRC>::value && !std::is_signed<VALUE_DST>::value) {
167 val = std::max(val, static_cast<VALUE_SRC>(0));
168 }
169 // convert between integral types, and it may overflow
170 if constexpr (
171 (std::is_integral<VALUE_SRC>::value && std::is_integral<VALUE_DST>::value) &&
172 (sizeof(VALUE_SRC) > sizeof(VALUE_DST) ||
173 (sizeof(VALUE_SRC) == sizeof(VALUE_DST) && std::is_signed<VALUE_DST>::value))) {
174 return static_cast<VALUE_DST>(
175 std::min(val, static_cast<VALUE_SRC>(std::numeric_limits<VALUE_DST>::max())));
176 }
177 // the rest
178 else {
179 return static_cast<VALUE_DST>(val);
180 }
181 }

184template <typename PIX_SRC, typename PIX_DST>
185inline void convert_image_pixel<PIX_SRC, PIX_DST>::operator()(const PIX_SRC& src, PIX_DST& dst)
186 const
187{
188 using V_SRC = typename ImageTraits<PIX_SRC>::TValue;
189 using V_DST = typename ImageTraits<PIX_DST>::TValue;
190 constexpr size_t L_SRC = static_cast<size_t>(ImageTraits<PIX_SRC>::channel);
191 constexpr size_t L_DST = static_cast<size_t>(ImageTraits<PIX_DST>::channel);
192 // convert [1] to [1]
193 if constexpr (1 == L_SRC && 1 == L_DST) {
194 dst = convert_channel_value<V_SRC, V_DST>(src);
195 }
196 // convert [1] to [n]
197 else if constexpr (1 == L_SRC) {
198 dst(0) = convert_channel_value<V_SRC, V_DST>(src);
199 for (size_t i = 1; i < L_DST; ++i) {
200 dst(i) = static_cast<V_DST>(0);
201 }
202 if constexpr (3 <= L_DST) {
203 dst(1) = dst(0);
204 dst(2) = dst(0);
205 }
206 if constexpr (4 == L_DST) {
207 if constexpr (std::is_floating_point<V_DST>::value) {
208 dst(3) = static_cast<V_DST>(1);
209 } else if constexpr (!std::is_signed<V_DST>::value && 1 == sizeof(V_DST)) {
210 dst(3) = std::numeric_limits<V_DST>::max();
211 } else {
212 dst(3) = static_cast<V_DST>(0);
213 }
214 }
215 }
216 // convert [n] to [1]
217 else if constexpr (1 == L_DST) {
218 dst = convert_channel_value<V_SRC, V_DST>(src(0));
219 }
220 // convert [m] to [n]
221 else {
222 constexpr size_t L_MIN = std::min(L_SRC, L_DST);
223 for (size_t i = 0; i < L_MIN; ++i) {
224 dst(i) = convert_channel_value<V_SRC, V_DST>(src(i));
225 }
226 if constexpr (3 == L_MIN && 4 == L_DST) {
227 if constexpr (std::is_floating_point<V_DST>::value) {
228 dst(3) = static_cast<V_DST>(1);
229 } else if constexpr (!std::is_signed<V_DST>::value && 1 == sizeof(V_DST)) {
230 dst(3) = std::numeric_limits<V_DST>::max();
231 } else {
232 dst(3) = static_cast<V_DST>(0);
233 }
234 } else {
235 for (size_t i = L_MIN; i < L_DST; ++i) {
236 dst(i) = static_cast<V_DST>(0);
237 }
238 }
239 }
