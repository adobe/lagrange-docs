43// Load image as the provided type/view. Converts if needed/possible. Returns true on success.
44template <typename T>
45bool load_image_as(const fs::path& path, image::ImageView<T>& img)
46{
47 LoadImageResult result = load_image(path);
48 if (!result.valid) return false;
49
50 if (image::ImageTraits<T>::precision == result.precision &&
51 image::ImageTraits<T>::channel == result.channel) {
52 return img.view(
53 result.storage,
54 result.storage->get_full_size()(0) / sizeof(T),
55 result.storage->get_full_size()(1),
56 sizeof(T),
57 1,
58 0,
59 0);
60 }
61
62 bool valid_conversion = false;
63#define LAGRANGE_TMP(PRECISION, CHANNEL, TYPE) \
64 if (image::ImagePrecision::PRECISION == result.precision && \
65 image::ImageChannel::CHANNEL == result.channel) { \

69 result.storage->get_full_size()(1), \
70 sizeof(TYPE), \
71 1, \
72 0, \
73 0); \
74 valid_conversion = img.convert_from(temp_image_view, 1); \
75 }
76#define LAGRANGE_TMP_COMMA ,
77 LAGRANGE_TMP(uint8, one, unsigned char)
78 else LAGRANGE_TMP(uint8, three, Eigen::Matrix<unsigned char LAGRANGE_TMP_COMMA 3 LAGRANGE_TMP_COMMA 1>) else LAGRANGE_TMP(
