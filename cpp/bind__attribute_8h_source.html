119 self.process([=](auto& attr) { attr.insert_elements(num_elements); });
120 },
121 "num_elements"_a,
122 R"(Insert new elements with default value to the attribute.
123
124:param num_elements: Number of elements to insert.)");
125 attr_class.def(
126 "insert_elements",
127 [](PyAttribute& self, nb::object value) {
128 auto insert_elements = [&](auto& attr) {
129 using ValueType = typename std::decay_t<decltype(attr)>::ValueType;
130 GenericTensor tensor;
131 std::vector<ValueType> buffer;
132 if (nb::try_cast(value, tensor)) {
133 if (tensor.dtype() != nb::dtype<ValueType>()) {
134 throw nb::type_error(fmt::format(
135 "Tensor has a unexpected dtype. Expecting {}.",
136 internal::string_from_scalar<ValueType>())
137 .c_str());
138 }
139 Tensor<ValueType> local_tensor(tensor.handle());
140 auto [data, shape, stride] = tensor_to_span(local_tensor);
141 la_runtime_assert(is_dense(shape, stride));
142 attr.insert_elements(data);
143 } else if (nb::try_cast(value, buffer)) {
144 attr.insert_elements({buffer.data(), buffer.size()});
145 } else {
146 throw nb::type_error("Argument `value` must be either list or np.ndarray.");
147 }
148 };
149 self.process(insert_elements);
150 },
151 "tensor"_a,
152 R"(Insert new elements to the attribute.
153
154:param tensor: A tensor with shape (num_elements, num_channels) or (num_elements,).)");
155 attr_class.def(
156 "empty",
157 [](PyAttribute& self) { return self.process([](auto& attr) { return attr.empty(); }); },
158 "Return true if the attribute is empty.");
159 attr_class.def_prop_ro(
160 "num_elements",
161 [](PyAttribute& self) {
162 return self.process([](auto& attr) { return attr.get_num_elements(); });
163 },
164 "Number of elements in the attribute.");
165 attr_class.def_prop_ro(
166 "external",
167 [](PyAttribute& self) {
168 return self.process([](auto& attr) { return attr.is_external(); });
169 },
170 "Return true if the attribute wraps external buffer.");
171 attr_class.def_prop_ro(
172 "readonly",
173 [](PyAttribute& self) {
174 return self.process([](auto& attr) { return attr.is_read_only(); });
175 },
176 "Return true if the attribute is read-only.");
177 attr_class.def_prop_rw(
178 "data",
179 [](PyAttribute& self) {
180 return self.process([&](auto& attr) {
181 auto tensor = attribute_to_tensor(attr, nb::find(&self));
182 return nb::cast(tensor, nb::rv_policy::reference_internal);
183 });
184 },
185 [](PyAttribute& self, nb::object value) {
186 auto wrap_tensor = [&](auto& attr) {
187 using ValueType = typename std::decay_t<decltype(attr)>::ValueType;
188 GenericTensor tensor;
189 std::vector<ValueType> buffer;
190 if (nb::try_cast(value, tensor)) {
191 if (tensor.dtype() != nb::dtype<ValueType>()) {
192 throw nb::type_error(fmt::format(
193 "Tensor has a unexpected dtype. Expecting {}.",
194 internal::string_from_scalar<ValueType>())
195 .c_str());
196 }
197 Tensor<ValueType> local_tensor(tensor.handle());
198 auto [data, shape, stride] = tensor_to_span(local_tensor);
199 la_runtime_assert(is_dense(shape, stride));
200 la_runtime_assert(shape.size() == 1 ? 1 == attr.get_num_channels() : true);
201 la_runtime_assert(
202 shape.size() == 2 ? shape[1] == attr.get_num_channels() : true);
203 const size_t num_elements = shape[0];
204
205 auto owner = std::make_shared<nb::object>(nb::find(tensor));
206 attr.wrap(make_shared_span(owner, data.data(), data.size()), num_elements);
207 } else if (nb::try_cast(value, buffer)) {
208 attr.clear();
209 attr.insert_elements({buffer.data(), buffer.size()});
210 } else {
211 throw nb::type_error("Attribute.data must be either list or np.ndarray.");
212 }
213 };
214 self.process(wrap_tensor);
215 },
216 "Raw data buffer of the attribute.");
217}
