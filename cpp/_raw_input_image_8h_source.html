224 static_assert(
225 std::is_same<unsigned char, Scalar>::value || std::is_same<Eigen::half, Scalar>::value ||

285template <typename T>
286T wrap_uv(const T u, const RawInputImage::wrap_mode m)
287{
288 if (RawInputImage::wrap_mode::clamp == m) {
289 return std::clamp(u, static_cast<T>(0), static_cast<T>(1));
290 } else if (RawInputImage::wrap_mode::repeat == m) {
291 return u - std::floor(u);
292 } else if (RawInputImage::wrap_mode::mirror == m) {
293 const auto f = static_cast<int>(std::floor(u));
294 auto _u = u - static_cast<T>(f);
295 if (f & 1) {
296 _u = static_cast<T>(1) - _u;
297 }
298 return _u;
299 }
300 assert(0);
301 return u;
302}
