38template <typename Scalar, typename Index, typename DerivedV, typename DerivedF>
39SurfaceMesh<Scalar, Index> eigen_to_surface_mesh(
40 const Eigen::MatrixBase<DerivedV>& V,
41 const Eigen::MatrixBase<DerivedF>& F)
42{
43 SurfaceMesh<Scalar, Index> mesh(static_cast<Index>(V.cols()));
44 mesh.add_vertices(static_cast<Index>(V.rows()));
45 mesh.add_polygons(static_cast<Index>(F.rows()), static_cast<Index>(F.cols()));
46 vertex_ref(mesh) = V.template cast<Scalar>();
47 facet_ref(mesh) = F.template cast<Index>();
48 return mesh;
49}
