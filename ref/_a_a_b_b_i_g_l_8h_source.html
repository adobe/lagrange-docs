33template <typename _VertexArray, typename _ElementArray = lagrange::Triangles>
34class AABBIGL : public BVH<_VertexArray, _ElementArray>

42 using PointType = typename Parent::PointType;
43 using ClosestPoint = typename Parent::ClosestPoint;
44 using AABB = igl::AABB<VertexArray, 3>;

84 ClosestPoint query_closest_point(const PointType& p) const override
85 {
86 ClosestPoint r;
87 int idx = -1;
88 r.squared_distance =
89 m_aabb.squared_distance(m_vertices, m_elements, p, idx, r.closest_point);
90 r.embedding_element_idx = safe_cast<Index>(idx);

114
115 std::vector<ClosestPoint> batch_query_closest_point(const VertexArray& query_pts) const override
116 {
117 const auto num_queries = query_pts.rows();
118 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> sq_dists(num_queries);
119 Eigen::Matrix<Index, Eigen::Dynamic, 1> embedding_elements(num_queries);
120 VertexArray closest_points(num_queries, m_vertices.cols());
121
122 m_aabb.squared_distance(
123 m_vertices,
124 m_elements,
125 query_pts,
126 sq_dists,
127 embedding_elements,
128 closest_points);
129
130 std::vector<ClosestPoint> r(num_queries);
131 for (auto i : range(num_queries)) {
132 r[i].embedding_element_idx = embedding_elements[i];
133 r[i].squared_distance = sq_dists[i];
134 r[i].closest_point = closest_points.row(i);
135 compute_closest_vertex_within_element(r[i], query_pts.row(i));
136 }
137 return r;
138 }
139
140private:
145 void compute_closest_vertex_within_element(ClosestPoint& entry, const PointType& p) const
146 {
147 assert(entry.embedding_element_idx >= 0);
148 assert(entry.embedding_element_idx < safe_cast<Index>(m_elements.rows()));
149 const auto vertex_per_element = m_elements.cols();
150 Scalar min_v_dist = std::numeric_limits<Scalar>::max();
151 for (auto i : range(vertex_per_element)) {
152 const auto vid = m_elements(entry.embedding_element_idx, i);
153 const auto sq_dist = (m_vertices.row(vid) - p).squaredNorm();
154 if (sq_dist < min_v_dist) {
155 min_v_dist = sq_dist;
156 entry.closest_vertex_idx = vid;
157 }
158 }
159 }
160
161
162private:
163 VertexArray m_vertices;
164 ElementArray m_elements;
165 AABB m_aabb;
166};
