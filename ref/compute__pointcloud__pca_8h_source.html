23// compute_pointcloud_pca()
24//
25// Finds the principle components for a pointcloud
26// Assumes that the points are supplied in a matrix where each
27// ``row'' is a point.
28//
29// This is closely related to the inertia tensor, principal directions
30// and principal moments. But it is not exactly the same.
31//
32// COVARIANCE_MATRIX (tensor) = (P-eC)^T (P-eC)
33// Where C is the centroid, and e is column vector of ones.
34// eigenvalues and eigenvectors of this matrix would be
35// the principal weights and components.
36//
37// MOMENT OF INERTIA (tensor) = trace(P^T P)I - P^T P
38// eigenvalues and eigenvectors of this matrix would be
39// the principal moments and directions.
40//
41// This refactors segmentation/Pca.h
42
43template <typename Scalar>
44struct ComputePointcloudPCAOutput
45{
46 // The point around which the covariance matrix is evaluated.
47 // Col vector to be consistent with `components` and weights.
48 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> center;
49 // Each column is a component, sorted by weight magnitudes.
50 // n_rows == n_cols == space dimension
51 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> components;
52 // Each entry is a weight for the corresponding principal component
53 // n_rows == space dimension
54 // Col vector to be consistent with `components`.
55 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> weights;
56};
57
58// Input:
59// points, Each row should be a point in the point cloud.
60// To be consistent with the rest of Lagrange.
61// should_shift_centeroid,
62// when true: covariance = (P-centroid)^T (P-centroid)
63// when false: covariance = (P)^T (P)
64// should_normalize, should we divide the result by number of points?
65template <typename Derived1>
66ComputePointcloudPCAOutput<typename Derived1::Scalar> compute_pointcloud_pca(
67 const Eigen::MatrixBase<Derived1>& points,
68 const bool should_shift_centeroid,
69 const bool should_normalize)
70{
71 //
72 using Scalar = typename Derived1::Scalar;
73 using MatrixXS =
74 Eigen::Matrix<Scalar, Derived1::ColsAtCompileTime, Derived1::ColsAtCompileTime>;
75 using RowVectorXS = Eigen::Matrix<Scalar, 1, Derived1::ColsAtCompileTime>;
76
77 // Prechecks
78 la_runtime_assert(points.rows() >= 2, "There must be at least two points");
79
80 // Compute covariance
81 MatrixXS covariance;
82 RowVectorXS center;
83
84 if (should_shift_centeroid) {
85 center = points.colwise().mean();
86 } else {
87 center = RowVectorXS::Zero(points.cols());
88 } // end of should_shift_centroid
89
90 if (should_normalize) {
91 // We may instead divide by points.rows()-1 which applies the Bessel's correction.
92 // https://en.wikipedia.org/wiki/Bessel%27s_correction
93 covariance =
94 ((points.rowwise() - center).transpose() * (points.rowwise() - center)) / points.rows();
95 } else {
96 covariance = ((points.rowwise() - center).transpose() * (points.rowwise() - center));
97 } // end of should_normalize
98
99 // SelfAdjointEigenSolver is guaranteed to return the eigenvalues sorted
100 // in increasing order. Note that this is not true for the general EigenValueSolver.
101 Eigen::SelfAdjointEigenSolver<MatrixXS> eigs(covariance);
102 // Unless something is going wrong, the covariance matrix should
103 // be well behaved. Let's double check.
104 la_runtime_assert(eigs.info() == Eigen::Success, "Eigen decomposition failed");
105
106 ComputePointcloudPCAOutput<Scalar> output;
107 output.center = center.transpose();
108 output.components = eigs.eigenvectors(); // Columns are eigenvectors
109 output.weights = eigs.eigenvalues();
110
111 // make sure components follow the right hand rule
112 if (output.components.determinant() < 0.) {
113 output.components.col(0) *= -1;
114 }
115
116 return output;
117}
