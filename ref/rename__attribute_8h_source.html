23template <typename MeshType>
24void rename_vertex_attribute(
25 MeshType& mesh,
26 const std::string& old_name,
27 const std::string& new_name)
28{
29 static_assert(MeshTrait<MeshType>::is_mesh(), "MeshType is not a mesh");
30
31 using AttributeArray = typename MeshType::AttributeArray;
32 la_runtime_assert(mesh.has_vertex_attribute(old_name));
33
34 AttributeArray attr_values;
35 mesh.export_vertex_attribute(old_name, attr_values);
36 mesh.remove_vertex_attribute(old_name);
37 mesh.add_vertex_attribute(new_name);
38 mesh.import_vertex_attribute(new_name, attr_values);
39}
