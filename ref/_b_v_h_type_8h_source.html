19enum class BVHType { //
20 NANOFLANN,
21 IGL,
22 UNKNOWN
23};
24
25inline std::string bvhtype_to_string(const BVHType in)
26{
27 switch (in) {
28 case BVHType::NANOFLANN: return "NANOFLANN";
29 case BVHType::IGL: return "IGL";
30 case BVHType::UNKNOWN:
31 default: return "UNKNOWN";
32 }
33}
34
35inline BVHType string_to_bvhtype(const std::string in)
36{
37 if (in == "NANOFLANN") {
38 return BVHType::NANOFLANN;
39 } else if (in == "IGL") {
40 return BVHType::IGL;
41 } else {
42 return BVHType::UNKNOWN;
43 }
44}
