82template <typename MeshType, typename GridType = openvdb::FloatGrid>
83auto mesh_to_volume(const MeshType &mesh, double voxel_size) -> typename GridType::Ptr
84{
85 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
86
87 openvdb::initialize();
88
89 const openvdb::Vec3d offset(voxel_size / 2.0, voxel_size / 2.0, voxel_size / 2.0);
90 auto transform = openvdb::math::Transform::createLinearTransform(voxel_size);
91 transform->postTranslate(offset);
92
93 MeshAdapter<MeshType> adapter(mesh, *transform);
94 try {
95 return openvdb::tools::meshToVolume<GridType, MeshAdapter<MeshType>>(adapter, *transform);
96 } catch (openvdb::ArithmeticError &) {
97 logger().error("Voxel size too small: {}", voxel_size);
98 throw;
99 }
100}
