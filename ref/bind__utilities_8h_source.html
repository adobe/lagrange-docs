104 [](MeshType& mesh,

107 std::optional<NormalOptions> normal_options) {
108 NormalOptions options;
109 if (normal_options.has_value()) {
110 options = std::move(normal_options.value());
111 }
112
113 if (cone_vertices.is_none()) {
114 return compute_normal<Scalar, Index>(mesh, feature_angle_threshold, {}, options);
