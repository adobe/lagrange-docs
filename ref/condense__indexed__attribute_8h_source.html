34template <typename MeshType>
35void condense_indexed_attribute(
36 MeshType& mesh,
37 const std::string& attr_name,
38 const std::string& new_attr_name = "")
39{
40 static_assert(MeshTrait<MeshType>::is_mesh(), "MeshType is not a mesh");
41 la_runtime_assert(mesh.has_indexed_attribute(attr_name));
42
43 using Index = typename MeshType::Index;
44 using AttributeArray = typename MeshType::AttributeArray;
45 using IndexArray = typename MeshType::IndexArray;
46 static_assert(AttributeArray::IsRowMajor, "Attribute array must be row major");
47
48 const auto num_vertices = mesh.get_num_vertices();
49 const auto num_facets = mesh.get_num_facets();
50 const auto& facets = mesh.get_facets();
51 const auto vertex_per_facet = mesh.get_vertex_per_facet();
52
53 const auto attr = mesh.get_indexed_attribute(attr_name);
54 const auto& attr_values = std::get<0>(attr);
55 const auto& attr_indices = std::get<1>(attr);
56 const Index num_cols = safe_cast<Index>(attr_values.cols());
57
58 using IndexEntry = std::array<Index, 2>; // facet id, corner id
59
65 auto index_pair_comp = [&](const IndexEntry& id0, const IndexEntry& id1) {
66 const Index offset_0 = attr_indices(id0[0], id0[1]) * num_cols;
67 const Index offset_1 = attr_indices(id1[0], id1[1]) * num_cols;
68 const auto base_ptr = attr_values.data();
69 return std::lexicographical_compare(
70 base_ptr + offset_0,
71 base_ptr + offset_0 + num_cols,
72 base_ptr + offset_1,
73 base_ptr + offset_1 + num_cols);
74 };
75
79 auto index_pair_eq = [&](const IndexEntry& id0, const IndexEntry& id1) -> bool {
80 return attr_values.row(attr_indices(id0[0], id0[1])) ==
81 attr_values.row(attr_indices(id1[0], id1[1]));
82 };
83
84 Index count = 0;
85 std::vector<Index> condensed_attr_values_index;
86 condensed_attr_values_index.reserve(attr_values.rows());
87 AttributeArray condensed_attr_values;
88 IndexArray condensed_attr_indices(num_facets, vertex_per_facet);
89 std::vector<IndexEntry> indices(num_facets * vertex_per_facet);
90 std::vector<Index> offsets(num_vertices + 2, 0);
91
92 for (auto fi : range(num_facets)) {
93 for (auto ci : range(vertex_per_facet)) {
94 offsets[facets(fi, ci) + 2]++;
95 }
96 }
97 std::partial_sum(offsets.begin(), offsets.end(), offsets.begin());
98
99 for (auto fi : range(num_facets)) {
100 for (auto ci : range(vertex_per_facet)) {
101 auto vi = facets(fi, ci);
102 indices[offsets[vi + 1]++] = {fi, ci};
103 }
104 }
105
106 tbb::parallel_for(
107 tbb::blocked_range<Index>(0, num_vertices),
108 [&](const tbb::blocked_range<Index>& tbb_range) {
109 for (auto vi = tbb_range.begin(); vi != tbb_range.end(); vi++) {
110 tbb::parallel_sort(
111 indices.data() + offsets[vi],
112 indices.data() + offsets[vi + 1],
113 index_pair_comp);
114 }
115 });
116
117 for (auto vi : range(num_vertices)) {
118 const IndexEntry* local_indices = indices.data() + offsets[vi];
119 const Index num_adj_facets = safe_cast<Index>(offsets[vi + 1] - offsets[vi]);
120 for (Index i = 0; i < num_adj_facets; i++) {
121 const auto& curr_entry = local_indices[i];
122 condensed_attr_indices(curr_entry[0], curr_entry[1]) = count;
123
124 for (Index j = i; j < num_adj_facets; j++) {
125 const auto& next_entry = local_indices[j];
126 if (index_pair_eq(curr_entry, next_entry)) {
127 condensed_attr_indices(next_entry[0], next_entry[1]) = count;
128 i = j;
129 } else {
130 break;
131 }
132 }
133 condensed_attr_values_index.push_back(attr_indices(curr_entry[0], curr_entry[1]));
134 count++;
135 }
136 }
137
138 assert(safe_cast<Index>(condensed_attr_values_index.size()) == count);
139
140 condensed_attr_values.resize(count, num_cols);
141 for (auto i : range(count)) {
142 condensed_attr_values.row(i) = attr_values.row(condensed_attr_values_index[i]);
143 }
144
145 if (new_attr_name == "" || new_attr_name == attr_name) {
146 mesh.import_indexed_attribute(attr_name, condensed_attr_values, condensed_attr_indices);
147 } else {
148 mesh.add_indexed_attribute(new_attr_name);
149 mesh.import_indexed_attribute(new_attr_name, condensed_attr_values, condensed_attr_indices);
150 }
151}
