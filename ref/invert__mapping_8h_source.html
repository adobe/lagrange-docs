43template <typename Index>
44auto invert_mapping(span<Index> old2new, Index num_target_entries)
45 -> std::tuple<std::vector<Index>, std::vector<Index>>
46{
47 Index num_source_entries = static_cast<Index>(old2new.size());
48 std::vector<Index> mapping_offsets(num_target_entries + 1, 0);
49 std::vector<Index> mapping_data;
50
51 for (Index i = 0; i < num_source_entries; ++i) {
52 Index j = old2new[i];
53 if (j == invalid<Index>()) continue;
54 la_runtime_assert(
55 j < num_target_entries,
56 "Mapped element index cannot exceeds target number of elements!");
57 ++mapping_offsets[j + 1];
58 }
59
60 std::partial_sum(mapping_offsets.begin(), mapping_offsets.end(), mapping_offsets.begin());
61 mapping_data.resize(mapping_offsets.back());
62 for (Index i = 0; i < num_source_entries; i++) {
63 Index j = old2new[i];
64 if (j == invalid<Index>()) continue;
65 mapping_data[mapping_offsets[j]++] = i;
66 }
67
68 std::rotate(mapping_offsets.begin(), std::prev(mapping_offsets.end()), mapping_offsets.end());
69 mapping_offsets[0] = 0;
70
71 return {std::move(mapping_data), std::move(mapping_offsets)};
72}
