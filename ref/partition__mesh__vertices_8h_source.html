52template <typename DerivedF>
53Eigen::Matrix<index_t, Eigen::Dynamic, 1> partition_mesh_vertices(

55 index_t num_partitions)
56{
57 const auto num_elems = safe_cast<index_t>(facets.rows());
58 const auto num_nodes = safe_cast<index_t>(facets.maxCoeff() + 1);
59 auto res = convert_index_buffer(facets);
60 return partition_mesh_vertices_raw(
61 num_elems,
62 num_nodes,
63 res.first.get(),
64 res.second.get(),
