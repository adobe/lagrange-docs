35template <typename MeshType>
36void compute_uv_distortion(MeshType& mesh)
37{
38 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
39 if (mesh.get_vertex_per_facet() != 3) {
40 throw std::runtime_error("Input mesh must be a triangle mesh.");
41 }
42
43 using Scalar = typename MeshType::Scalar;
44 using Index = typename MeshType::Index;
45 using AttributeArray = typename MeshType::AttributeArray;
46
47 if (mesh.has_vertex_attribute("uv") && !mesh.has_corner_attribute("uv")) {
48 map_vertex_attribute_to_corner_attribute(mesh, "uv");
49 }
50 if (!mesh.has_corner_attribute("uv")) {
51 throw std::runtime_error("UV vertex attribute is missing.");
52 }
53
54 // const Index num_vertices = mesh->get_num_vertices();
55 const Index num_facets = mesh.get_num_facets();
56 const Index vertex_per_facet = mesh.get_vertex_per_facet();
57 const auto& vertices = mesh.get_vertices();
58 const auto& facets = mesh.get_facets();
59
60 const auto& uv = mesh.get_corner_attribute("uv");
61 la_runtime_assert(safe_cast<Index>(uv.rows()) == num_facets * vertex_per_facet);
62
63 const Scalar INVALID = std::numeric_limits<Scalar>::max();
64
65 AttributeArray distortion(num_facets, 1);
66 distortion.setConstant(INVALID);
67
68 for (Index i = 0; i < num_facets; i++) {
69 Eigen::Matrix<Index, 1, 3> f = facets.row(i);
70 Eigen::Matrix<Scalar, 1, 3> v0, v1, v2;
71 Eigen::Matrix<Scalar, 1, 2> V0, V1, V2;
72 v0 << vertices.row(f[0]);
73 v1 << vertices.row(f[1]);
74 v2 << vertices.row(f[2]);
75 V0 << uv.row(i * 3 + 0);
76 V1 << uv.row(i * 3 + 1);
77 V2 << uv.row(i * 3 + 2);
78
79 if (V0.minCoeff() < 0.0 || V0.maxCoeff() > 1.0) continue;
80 if (V1.minCoeff() < 0.0 || V1.maxCoeff() > 1.0) continue;
81 if (V2.minCoeff() < 0.0 || V2.maxCoeff() > 1.0) continue;
82
83 // AMIPS energy can be computed geometrically.
84 // Area ratio == det(F)
85 // Dirichlet energy == |F|^2
86 const Eigen::Matrix<Scalar, 1, 3> e0 = v1 - v2;
87 const Eigen::Matrix<Scalar, 1, 3> e1 = v2 - v0;
88 const Eigen::Matrix<Scalar, 1, 3> e2 = v0 - v1;
89 const Scalar l0 = e0.norm();
90 const Scalar l1 = e1.norm();
91 const Scalar l2 = e2.norm();
92 const Scalar s = 0.5f * (l0 + l1 + l2);
93 const Scalar a = sqrt(s * (s - l0) * (s - l1) * (s - l2));
94
95 const Eigen::Matrix<Scalar, 1, 2> E0 = V1 - V2;
96 const Eigen::Matrix<Scalar, 1, 2> E1 = V2 - V0;
97 const Eigen::Matrix<Scalar, 1, 2> E2 = V0 - V1;
98 const Scalar L0 = E0.norm();
99 const Scalar L1 = E1.norm();
100 const Scalar L2 = E2.norm();
101 const Scalar S = 0.5f * (L0 + L1 + L2);
102 const Scalar A = sqrt(S * (S - L0) * (S - L1) * (S - L2));
103
104 const Scalar area_ratio = A / a;
105
106 const Scalar cot_0 = e1.dot(-e2) / e1.cross(-e2).norm();
107 const Scalar cot_1 = e2.dot(-e0) / e2.cross(-e0).norm();
108 const Scalar cot_2 = e0.dot(-e1) / e0.cross(-e1).norm();
109
110 // Need to divid by the area since we want energy density.
111 const Scalar dirichlet = 0.5f * (cot_0 * L0 * L0 + cot_1 * L1 * L1 + cot_2 * L2 * L2) / a;
112
113 // Distortion measure is bounded from below by 2. It is achieved if
114 // and only if the mapping f is isometric.
115 distortion(i, 0) = dirichlet / area_ratio;
116 }
117
118 mesh.add_facet_attribute("distortion");
119 mesh.set_facet_attribute("distortion", distortion);
120}
