42template <typename MeshType>
43std::unique_ptr<MeshType> resolve_nonmanifoldness(MeshType& mesh)
44{
45 if (!mesh.is_connectivity_initialized()) {
46 mesh.initialize_connectivity();
47 }
48
49 mesh.initialize_edge_data();
50
51 using Index = typename MeshType::Index;
52 using VertexArray = typename MeshType::VertexArray;
53 using FacetArray = typename MeshType::FacetArray;
54
55 const Index num_vertices = mesh.get_num_vertices();
56 const Index num_facets = mesh.get_num_facets();
57 const Index vertex_per_facet = mesh.get_vertex_per_facet();
58 const auto& vertices = mesh.get_vertices();
59 const auto& facets = mesh.get_facets();
60
61
68 auto get_orientation = [&facets](const Index v0, const Index v1, const Index fid) -> bool {
69 const auto& f = facets.row(fid);
70 if (f[0] == v0 && f[1] == v1)
71 return true;
72 else if (f[1] == v0 && f[2] == v1)
73 return true;
74 else if (f[2] == v0 && f[0] == v1)
75 return true;
76 else
77 return false;
78 };
79
88 auto is_inconsistently_oriented = [&mesh, &get_orientation](const Index ei) -> bool {
89 const auto e = mesh.get_edge_vertices(ei);
90 std::array<bool, 2> orientations;
91 size_t count = 0;
92 mesh.foreach_facets_around_edge(ei, [&](Index fid) {
93 orientations[count] = get_orientation(e[0], e[1], fid);
94 count++;
95 });
96 return orientations[0] == orientations[1];
97 };
98
105 auto is_inconsistently_oriented_wrt_facets =
106 [&mesh, &get_orientation](const Index ei, const Index f0, const Index f1) {
107 const auto e = mesh.get_edge_vertices(ei);
108 return get_orientation(e[0], e[1], f0) == get_orientation(e[0], e[1], f1);
109 };
110
115 auto is_nonmanifold_edge = [&mesh, &is_inconsistently_oriented](const Index ei) {
116 auto edge_valence = mesh.get_num_facets_around_edge(ei);
117 if (edge_valence > 2) return true;
118 if (edge_valence <= 1) return false;
119 return is_inconsistently_oriented(ei);
120 };
121
122 // Flood fill color across manifold edges. The color field split the
123 // facets into locally manifold components. Edges and vertices adjacent
124 // to multiple colors will be split.
125 constexpr Index BLANK = 0;
126 std::vector<Index> colors(num_facets, BLANK);
127 Index curr_color = 1;
128 for (Index i = 0; i < num_facets; i++) {
129 if (colors[i] != BLANK) continue;
130 std::queue<Index> Q;
131 Q.push(i);
132 colors[i] = curr_color;
133 while (!Q.empty()) {
134 Index curr_fid = Q.front();
135 Q.pop();
136 for (Index j = 0; j < vertex_per_facet; j++) {
137 Index ei = mesh.get_edge(curr_fid, j);
138 if (is_nonmanifold_edge(ei)) continue;
139
140 mesh.foreach_facets_around_edge(ei, [&](Index adj_fid) {
141 if (colors[adj_fid] == BLANK) {
142 colors[adj_fid] = curr_color;
143 Q.push(adj_fid);
144 }
145 });
146 }
147 }
148 curr_color++;
149 }
150
151 Index vertex_count = num_vertices;
152 // Note:
153 // The goal of vertex_map is to split the 1-ring neighborhood of a
154 // non-manifold vertex based on the colors of its adjacent facets. All
155 // adjacent facets sharing the sanme color will share the same copy of
156 // this vertex.
157 //
158 // To achieve this, I store a color map for each non-manifold vertex,
159 // where vertex_map maps a non-manifold vertex to its color map.
160 // A color map maps specific color to the vertex index after the split.
161 std::unordered_map<Index, std::unordered_map<Index, Index>> vertex_map;
162 // Split non-manifold edges.
163 for (Index i = 0; i < mesh.get_num_edges(); ++i) {
164 const auto e = mesh.get_edge_vertices(i);
165
166 if (!is_nonmanifold_edge(i)) continue;
167 auto itr0 = vertex_map.find(e[0]);
168 auto itr1 = vertex_map.find(e[1]);
169
170 if (itr0 == vertex_map.end()) {
171 itr0 = vertex_map.insert({e[0], std::unordered_map<Index, Index>()}).first;
172 }
173 if (itr1 == vertex_map.end()) {
174 itr1 = vertex_map.insert({e[1], std::unordered_map<Index, Index>()}).first;
175 }
176
177 auto& color_map_0 = itr0->second;
178 auto& color_map_1 = itr1->second;
179
180 std::unordered_map<Index, std::list<Index>> color_count;
181 mesh.foreach_facets_around_edge(i, [&](Index fid) {
182 const auto c = colors[fid];
183
184 auto c_itr = color_count.find(c);
185 if (c_itr == color_count.end()) {
186 color_count.insert({c, {fid}});
187 } else {
188 c_itr->second.push_back(fid);
189 }
190
191 auto c_itr0 = color_map_0.find(c);
192 if (c_itr0 == color_map_0.end()) {
193 if (color_map_0.size() == 0) {
194 color_map_0.insert({c, e[0]});
195 } else {
196 color_map_0.insert({c, vertex_count});
197 vertex_count++;
198 }
199 }
200
201 auto c_itr1 = color_map_1.find(c);
202 if (c_itr1 == color_map_1.end()) {
203 if (color_map_1.size() == 0) {
204 color_map_1.insert({c, e[1]});
205 } else {
206 color_map_1.insert({c, vertex_count});
207 vertex_count++;
208 }
209 }
210 });
211
212 for (const auto& entry : color_count) {
213 // Corner case 1:
214 // Exact two facets share the same color around this edge, but
215 // they are inconsistently oriented. Thus, they needs to be
216 // detacted.
217 const bool inconsistent_edge =
218 (entry.second.size() == 2) &&
219 is_inconsistently_oriented_wrt_facets(i, entry.second.front(), entry.second.back());
220
221 // Corner case 2:
222 // Some facets around this non-manifold edge are connected via
223 // a chain of manifold edges. Thus, they have the same color.
224 // To resolve this, I am detacching all facets of this color
225 // adjacent to this edge.
226 const bool single_comp_nonmanifoldness = entry.second.size() > 2;
227
228 if (single_comp_nonmanifoldness || inconsistent_edge) {
229 // Each facet will be reconnect to a newly created edge.
230 // This solution is not ideal, but works.
231 for (const auto fid : entry.second) {
232 colors[fid] = curr_color;
233 color_map_0.insert({curr_color, vertex_count});
234 vertex_count++;
235 color_map_1.insert({curr_color, vertex_count});
236 vertex_count++;
237 curr_color++;
238 }
239 }
240 }
241 }
242
243 // Split non-manifold vertices
244 for (Index i = 0; i < num_vertices; i++) {
245 const auto& adj_facets = mesh.get_facets_adjacent_to_vertex(i);
246 std::set<Index> adj_colors;
247 for (auto adj_f : adj_facets) {
248 adj_colors.insert(colors[adj_f]);
249 }
250 if (adj_colors.size() <= 1) continue;
251
252 auto itr = vertex_map.find(i);
253 if (itr == vertex_map.end()) {
254 vertex_map.insert({i, {}});
255 }
256
257 auto& vi_map = vertex_map[i];
258 for (auto c : adj_colors) {
259 auto c_itr = vi_map.find(c);
260 if (c_itr == vi_map.end()) {
261 if (vi_map.size() == 0) {
262 vi_map.insert({c, i});
263 } else {
264 vi_map.insert({c, vertex_count});
265 vertex_count++;
266 }
267 }
268 }
269 }
270
271 VertexArray manifold_vertices(vertex_count, vertices.cols());
272 manifold_vertices.topRows(num_vertices) = vertices;
273
274 std::vector<Index> backward_vertex_map(vertex_count, 0);
275 std::iota(backward_vertex_map.begin(), backward_vertex_map.begin() + num_vertices, 0);
276
277 for (const auto& itr : vertex_map) {
278 Index vid = itr.first;
279 for (const auto& itr2 : itr.second) {
280 manifold_vertices.row(itr2.second) = vertices.row(vid);
281 backward_vertex_map[itr2.second] = vid;
282 }
283 }
284
285 FacetArray manifold_facets = facets;
286 for (Index i = 0; i < num_facets; i++) {
287 const Index c = colors[i];
288 for (Index j = 0; j < vertex_per_facet; j++) {
289 const auto& itr = vertex_map.find(manifold_facets(i, j));
290 if (itr != vertex_map.end()) {
291 manifold_facets(i, j) = itr->second[c];
292 }
293 }
294 }
295
296 auto out_mesh = create_mesh(std::move(manifold_vertices), std::move(manifold_facets));
297
298 map_attributes(mesh, *out_mesh, backward_vertex_map);
299
300 out_mesh->initialize_connectivity();
301 out_mesh->initialize_edge_data();
302
303 out_mesh = resolve_vertex_nonmanifoldness(*out_mesh);
304 out_mesh = remove_topologically_degenerate_triangles(*out_mesh);
305 out_mesh = remove_duplicate_facets(*out_mesh);
306 out_mesh = remove_isolated_vertices(*out_mesh);
307 return out_mesh;
308}
