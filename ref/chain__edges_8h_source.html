40template <typename Index, typename Allocator = std::list<EdgeType<Index>>>
41std::vector<std::list<Index>> chain_edges(const Allocator& edges, bool close_loop = false)
42{
43 using EdgeIterator = typename Allocator::const_iterator;
44 using VertexEdgeAdjList = std::unordered_map<Index, std::vector<EdgeIterator>>;
45
46 using Chain = std::list<Index>;
47 std::vector<Chain> chains;
48 VertexEdgeAdjList next;
49 VertexEdgeAdjList prev;
50 size_t num_edges = std::distance(edges.begin(), edges.end());
51 next.reserve(num_edges);
52 prev.reserve(num_edges);
53 std::vector<bool> visited(num_edges, false);
54
55 auto insert_or_append = [&](VertexEdgeAdjList& adj_list, Index key, EdgeIterator value) {
56 auto itr = adj_list.find(key);
57 if (itr == adj_list.end()) {
58 adj_list.insert({key, {std::move(value)}});
59 } else {
60 itr->second.push_back(std::move(value));
61 }
62 };
63
64 for (auto itr = edges.begin(); itr != edges.end(); ++itr) {
65 const auto& e = *itr;
66 insert_or_append(next, e[0], itr);
67 insert_or_append(prev, e[1], itr);
68 }
69
74 auto grow_chain_forward = [&](Chain& chain) -> void {
75 // Extend chain from the end.
76 assert(chain.size() > 0);
77 while (true) {
78 const auto curr_v = chain.back();
79 const auto itr = next.find(curr_v);
80 if (itr != next.end()) {
81 const auto& next_edges = itr->second;
82 if (next_edges.size() != 1) break;
83
84 auto e_itr = next_edges.front();
85 auto eid = std::distance(edges.begin(), e_itr);
86 if (visited[eid]) break;
87
88 const auto& e = *e_itr;
89 la_debug_assert(e[0] == curr_v);
90 chain.push_back(e[1]);
91 visited[eid] = true;
92 } else {
93 break;
94 }
95 }
96 };
97
102 auto grow_chain_backward = [&](Chain& chain) -> void {
103 // Extend chain from the beginning.
104 assert(chain.size() > 0);
105 while (true) {
106 const auto curr_v = chain.front();
107 const auto itr = prev.find(curr_v);
108 if (itr != prev.end()) {
109 const auto& prev_edges = itr->second;
110 if (prev_edges.size() != 1) break;
111
112 auto e_itr = prev_edges.front();
113 auto eid = std::distance(edges.begin(), e_itr);
114 if (visited[eid]) break;
115
116 const auto& e = *e_itr;
117 la_debug_assert(e[1] == curr_v);
118 chain.push_front(e[0]);
119 visited[eid] = true;
120 } else {
121 break;
122 }
123 }
124 };
125
126 size_t eid = 0;
127 for (auto itr = edges.begin(); itr != edges.end(); ++itr, eid++) {
128 if (visited[eid]) continue;
129
130 const auto& e = *itr;
131 Chain chain;
132 chain.push_back(e[0]);
133 chain.push_back(e[1]);
134 visited[eid] = true;
135
136 grow_chain_forward(chain);
137 grow_chain_backward(chain);
138
139 if (!close_loop && chain.back() == chain.front()) {
140 chain.pop_back();
141 }
142 chains.push_back(std::move(chain));
143 }
144 return chains;
145}

159template <typename Index, typename Allocator = std::vector<EdgeType<Index>>>
160std::vector<std::vector<Index>> chain_undirected_edges(
161 const Allocator& edges,
162 bool close_loop = false)
163{
164 using VertexEdgeAdjList = std::unordered_map<Index, std::vector<Index>>;
165
166 using Chain = std::vector<Index>;
167 std::vector<Chain> chains;
168 VertexEdgeAdjList adj_list;
169 size_t num_edges = std::distance(edges.begin(), edges.end());
170 adj_list.reserve(num_edges);
171 std::vector<bool> visited(num_edges, false);
172
173 auto add_adj_connection = [&](Index ei) {
174 const auto& e = edges[ei];
175 Index v0 = e[0];
176 Index v1 = e[1];
177
178 auto itr = adj_list.find(v0);
179 if (itr != adj_list.end()) {
180 itr->second.push_back(ei);
181 } else {
182 adj_list.insert({v0, {ei}});
183 }
184
185 itr = adj_list.find(v1);
186 if (itr != adj_list.end()) {
187 itr->second.push_back(ei);
188 } else {
189 adj_list.insert({v1, {ei}});
190 }
191 };
192
193 LA_IGNORE_RANGE_LOOP_ANALYSIS_BEGIN
194 for (auto ei : range(num_edges)) {
195 add_adj_connection(ei);
196 }
197 LA_IGNORE_RANGE_LOOP_ANALYSIS_END
198
199 auto grow_chain_forward = [&](Chain& chain) {
200 Index curr_v = chain.back();
201 Index prev_v = chain[chain.size() - 2];
202
203 while (true) {
204 const auto& adj_edges = adj_list[curr_v];
205 if (adj_edges.size() != 2) break;
206
207 bool updated = false;
208 for (auto ei : adj_edges) {
209 if (visited[ei]) continue;
210
211 const auto& e = edges[ei];
212 if (e[0] == curr_v) {
213 chain.push_back(e[1]);
214 } else {
215 la_debug_assert(e[1] == curr_v);
216 chain.push_back(e[0]);
217 }
218 prev_v = curr_v;
219 curr_v = chain.back();
220 visited[ei] = true;
221 updated = true;
222 }
223 if (!updated) break;
224 }
225 };
226
227 auto grow_chain_backward = [&](Chain& chain) {
228 std::reverse(chain.begin(), chain.end());
229 grow_chain_forward(chain);
230 std::reverse(chain.begin(), chain.end());
231 };
232
233 for (auto ei : range(num_edges)) {
234 if (visited[ei]) continue;
235
236 const auto& e = edges[ei];
237 visited[ei] = true;
238 Chain chain;
239 chain.reserve(num_edges);
240 chain.push_back(e[0]);
241 chain.push_back(e[1]);
242 grow_chain_forward(chain);
243 grow_chain_backward(chain);
244
245 if (!close_loop && chain.front() == chain.back()) {
246 chain.pop_back();
247 }
248 chains.push_back(std::move(chain));
249 }
250
251 return chains;
252}
