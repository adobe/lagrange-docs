32template <typename Index, typename Allocator = std::list<EdgeType<Index>>>
33std::vector<std::list<Index>> chain_edges(const Allocator& edges)
34{
35 using VertexAdjList = std::unordered_map<Index, Index>;
36
37 using Chain = std::list<Index>;
38 std::vector<Chain> chains;
39 VertexAdjList next;
40 VertexAdjList prev;
41 size_t num_edges = std::distance(edges.begin(), edges.end());
42 next.reserve(num_edges);
43 prev.reserve(num_edges);
44 std::unordered_set<Index> visited;
45 visited.reserve(num_edges * 2);
46
47LA_IGNORE_RANGE_LOOP_ANALYSIS_BEGIN
48 for (const auto& e : edges) {
49 next[e[0]] = e[1];
50 prev[e[1]] = e[0];
51 }
52LA_IGNORE_RANGE_LOOP_ANALYSIS_END

57 auto has_visited = [&visited](Index id) -> bool { return visited.find(id) != visited.end(); };
58
63 auto grow_chain_forward = [&visited, &next, &has_visited](Chain& chain) -> void {
64 // Extend chain from the end.
65 assert(chain.size() > 0);
66 while (true) {
67 const auto curr_v = chain.back();
68 const auto itr = next.find(curr_v);
69 if (itr != next.end() && !has_visited(itr->second)) {
70 chain.push_back(itr->second);
71 visited.insert(itr->second);
72 } else {
73 break;
74 }
75 }
76 };
77
82 auto grow_chain_backward = [&visited, &prev, &has_visited](Chain& chain) -> void {
83 // Extend chain from the beginning.
84 assert(chain.size() > 0);
85 while (true) {
86 const auto curr_v = chain.front();
87 const auto itr = prev.find(curr_v);
88 if (itr != prev.end() && !has_visited(itr->second)) {
89 chain.push_front(itr->second);
90 visited.insert(itr->second);
91 } else {
92 break;
93 }
94 }
95 };
96
97LA_IGNORE_RANGE_LOOP_ANALYSIS_BEGIN
98 for (const auto& e : edges) {
99 if (!has_visited(e[0])) {
100 chains.emplace_back();
101 auto& chain = chains.back();
102 chain.push_back(e[0]);
103 visited.insert(e[0]);
104
105 grow_chain_forward(chain);
106 grow_chain_backward(chain);
107 }
108 }
109LA_IGNORE_RANGE_LOOP_ANALYSIS_END
110 return chains;
111}
