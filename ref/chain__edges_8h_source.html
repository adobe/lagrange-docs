41template <typename Index, typename Allocator = std::list<EdgeType<Index>>>
42std::vector<std::list<Index>> chain_edges(const Allocator& edges, bool close_loop = false)
43{
44 using EdgeIterator = typename Allocator::const_iterator;
45 using VertexEdgeAdjList = std::unordered_map<Index, std::vector<EdgeIterator>>;
46
47 using Chain = std::list<Index>;
48 std::vector<Chain> chains;
49 VertexEdgeAdjList next;
50 VertexEdgeAdjList prev;
51 size_t num_edges = std::distance(edges.begin(), edges.end());
52 next.reserve(num_edges);
53 prev.reserve(num_edges);
54 std::vector<bool> visited(num_edges, false);
55
56 auto insert_or_append = [&](VertexEdgeAdjList& adj_list, Index key, EdgeIterator value) {
57 auto itr = adj_list.find(key);
58 if (itr == adj_list.end()) {
59 adj_list.insert({key, {std::move(value)}});
60 } else {
61 itr->second.push_back(std::move(value));
62 }
63 };
64
65 for (auto itr = edges.begin(); itr != edges.end(); ++itr) {
66 const auto& e = *itr;
67 insert_or_append(next, e[0], itr);
68 insert_or_append(prev, e[1], itr);
69 }
70
75 auto grow_chain_forward = [&](Chain& chain) -> void {
76 // Extend chain from the end.
77 assert(chain.size() > 0);
78 while (true) {
79 const auto curr_v = chain.back();
80 const auto itr = next.find(curr_v);
81 if (itr != next.end()) {
82 const auto& next_edges = itr->second;
83 if (next_edges.size() != 1) break;
84
85 auto e_itr = next_edges.front();
86 auto eid = std::distance(edges.begin(), e_itr);
87 if (visited[eid]) break;
88
89 const auto& e = *e_itr;
90 la_debug_assert(e[0] == curr_v);
91 chain.push_back(e[1]);
92 visited[eid] = true;
93 } else {
94 break;
95 }
96 }
97 };
98
103 auto grow_chain_backward = [&](Chain& chain) -> void {
104 // Extend chain from the beginning.
105 assert(chain.size() > 0);
106 while (true) {
107 const auto curr_v = chain.front();
108 const auto itr = prev.find(curr_v);
109 if (itr != prev.end()) {
110 const auto& prev_edges = itr->second;
111 if (prev_edges.size() != 1) break;
112
113 auto e_itr = prev_edges.front();
114 auto eid = std::distance(edges.begin(), e_itr);
115 if (visited[eid]) break;
116
117 const auto& e = *e_itr;
118 la_debug_assert(e[1] == curr_v);
119 chain.push_front(e[0]);
120 visited[eid] = true;
121 } else {
122 break;
123 }
124 }
125 };
126
127 size_t eid = 0;
128 for (auto itr = edges.begin(); itr != edges.end(); ++itr, eid++) {
129 if (visited[eid]) continue;
130
131 const auto& e = *itr;
132 Chain chain;
133 chain.push_back(e[0]);
134 chain.push_back(e[1]);
135 visited[eid] = true;
136
137 grow_chain_forward(chain);
138 grow_chain_backward(chain);
139
140 if (!close_loop && chain.back() == chain.front()) {
141 chain.pop_back();
142 }
143 chains.push_back(std::move(chain));
144 }
145 return chains;
146}

164// Eigen integer matrix with 2 columns, one edge per row
165template <typename VertexIndex, typename EdgeArray, typename EdgeIndex>
166std::pair<VertexIndex, VertexIndex> get_edge_endpoints(EdgeArray& edges, EdgeIndex edge_index)
167{
168 return std::make_pair(
169 static_cast<VertexIndex>(edges(static_cast<Eigen::Index>(edge_index), 0)),
170 static_cast<VertexIndex>(edges(static_cast<Eigen::Index>(edge_index), 1)));
171}

173// std::vector of edges
174template <typename VertexIndex, typename EdgeType, typename EdgeIndex>
175std::pair<VertexIndex, VertexIndex> get_edge_endpoints(
176 const std::vector<EdgeType>& edges,
177 EdgeIndex edge_index)
178{
179 const auto& e = edges[static_cast<size_t>(edge_index)];
180 return std::make_pair(static_cast<VertexIndex>(e[0]), static_cast<VertexIndex>(e[1]));

198template <typename Index, typename Allocator = std::vector<EdgeType<Index>>>
199std::vector<std::vector<Index>> chain_undirected_edges(
200 const Allocator& edges,
201 bool close_loop = false)
202{
203 using VertexEdgeAdjList = std::unordered_map<Index, std::vector<Index>>;
204
205 using Chain = std::vector<Index>;
206 std::vector<Chain> chains;
207 VertexEdgeAdjList adj_list;
208 size_t num_edges = detail::get_num_edges(edges);
209 adj_list.reserve(num_edges);
210 std::vector<bool> visited(num_edges, false);
211
212 auto add_adj_connection = [&](Index ei) {
213 auto v = detail::get_edge_endpoints<Index>(edges, ei);
214
215 auto itr = adj_list.find(v.first);
216 if (itr != adj_list.end()) {
217 itr->second.push_back(ei);
218 } else {
219 adj_list.insert({v.first, {ei}});
220 }
221
222 itr = adj_list.find(v.second);
223 if (itr != adj_list.end()) {
224 itr->second.push_back(ei);
225 } else {
226 adj_list.insert({v.second, {ei}});
227 }
228 };
229
230 LA_IGNORE_RANGE_LOOP_ANALYSIS_BEGIN
231 for (auto ei : range(num_edges)) {
232 add_adj_connection(ei);
233 }
234 LA_IGNORE_RANGE_LOOP_ANALYSIS_END
235
236 auto grow_chain_forward = [&](Chain& chain) {
237 Index curr_v = chain.back();
238
239 while (true) {
240 const auto& adj_edges = adj_list[curr_v];
241 if (adj_edges.size() != 2) break;
242
243 bool updated = false;
244 for (auto ei : adj_edges) {
245 if (visited[ei]) continue;
246
247 auto v = detail::get_edge_endpoints<Index>(edges, ei);
248 if (v.first == curr_v) {
249 chain.push_back(v.second);
250 } else {
251 la_debug_assert(v.second == curr_v);
252 chain.push_back(v.first);
253 }
254 curr_v = chain.back();
255 visited[ei] = true;
256 updated = true;
257 }
258 if (!updated) break;
259 }
260 };
261
262 auto grow_chain_backward = [&](Chain& chain) {
263 std::reverse(chain.begin(), chain.end());
264 grow_chain_forward(chain);
265 std::reverse(chain.begin(), chain.end());
266 };
267
268 for (auto ei : range(num_edges)) {
269 if (visited[ei]) continue;
270
271 auto v = detail::get_edge_endpoints<Index>(edges, ei);
272 visited[ei] = true;
273 Chain chain;
274 chain.reserve(num_edges);
275 chain.push_back(v.first);
276 chain.push_back(v.second);
277 grow_chain_forward(chain);
278 grow_chain_backward(chain);
279
280 if (!close_loop && chain.front() == chain.back()) {
281 chain.pop_back();
282 }
283 chains.push_back(std::move(chain));
284 }
285
286 return chains;
287}
