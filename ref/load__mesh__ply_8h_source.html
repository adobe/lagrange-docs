32template <typename MeshType>
33std::unique_ptr<MeshType> load_mesh_ply(const fs::path& filename)
34{
35 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
36 using VertexArray = typename MeshType::VertexArray;
37 using FacetArray = typename MeshType::FacetArray;
38 using AttributeArray = typename MeshType::AttributeArray;
39 using Index = typename MeshType::Index;
40
41 VertexArray V;
42 FacetArray F;
43 Eigen::Matrix<Index, Eigen::Dynamic, Eigen::Dynamic> E;
44 AttributeArray N;
45 AttributeArray UV;
46
47 AttributeArray VD;
48 std::vector<std::string> VDheader;
49
50 AttributeArray FD;
51 std::vector<std::string> FDheader;
52
53 AttributeArray ED;
54 std::vector<std::string> EDheader;
55 std::vector<std::string> comments;
56
57 // TODO: libigl's wrapper around tinyply doesn't allow to load attributes of different tyeps
58 // (e.g. char for colors + float for normals).
59 igl::readPLY(
60 filename.string(),
61 V,
62 F,
63 E,
64 N,
65 UV,
66 VD,
67 VDheader,
68 FD,
69 FDheader,
70 ED,
71 EDheader,
72 comments);
73
74 auto mesh = create_mesh(std::move(V), std::move(F));
75 if (N.rows() == mesh->get_num_vertices()) {
76 logger().debug("Setting vertex normal");
77 mesh->add_vertex_attribute("normal");
78 mesh->import_vertex_attribute("normal", N);
79 }
80
81 auto has_attribute = [&](const std::string& name) {
82 return std::find(VDheader.begin(), VDheader.end(), name) != VDheader.end();
83 };
84
85 if (VD.rows() == mesh->get_num_vertices()) {
86 if (has_attribute("red") && has_attribute("green") && has_attribute("blue")) {
87 bool has_alpha = has_attribute("alpha");
88 int n = (has_alpha ? 4 : 3);
89 AttributeArray color(mesh->get_num_vertices(), n);
90 color.setZero();
91 for (size_t i = 0; i < VDheader.size(); ++i) {
92 if (VDheader[i] == "red") {
93 la_runtime_assert(n >= 1);
94 color.col(0) = VD.col(i);
95 } else if (VDheader[i] == "green") {
96 la_runtime_assert(n >= 2);
97 color.col(1) = VD.col(i);
98 } else if (VDheader[i] == "blue") {
99 la_runtime_assert(n >= 3);
100 color.col(2) = VD.col(i);
101 } else if (VDheader[i] == "alpha") {
102 la_runtime_assert(n >= 4);
103 color.col(3) = VD.col(i);
104 } else {
105 logger().warn("Unknown vertex attribute: {}", VDheader[i]);
106 }
107 }
108 logger().debug("Setting vertex color");
109 mesh->add_vertex_attribute("color");
110 mesh->import_vertex_attribute("color", color);
111 }
112 }
113
114 return mesh;
115}
