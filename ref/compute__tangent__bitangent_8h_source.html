48template <typename Scalar>
49Eigen::Matrix<Scalar, 2, 3> triangle_tangent_bitangent(
50 const Eigen::Matrix<Scalar, 3, 3>& vertices,
51 const Eigen::Matrix<Scalar, 3, 2>& uvs,
52 bool& sign)
53{
54 const auto edge0 = (vertices.row(1) - vertices.row(0)).eval();
55 const auto edge1 = (vertices.row(2) - vertices.row(0)).eval();
56
57 const auto edge_uv0 = (uvs.row(1) - uvs.row(0)).eval();
58 const auto edge_uv1 = (uvs.row(2) - uvs.row(0)).eval();
59
60 const Scalar uv_signed_area = (edge_uv0.x() * edge_uv1.y() - edge_uv0.y() * edge_uv1.x());
61 const Scalar r = (uv_signed_area == Scalar(0) ? Scalar(1) : Scalar(1) / uv_signed_area);
62
63 sign = (uv_signed_area > 0);
64
65 Eigen::Matrix<Scalar, 2, 3> T_BT;
66 T_BT.row(0) = ((edge0 * edge_uv1.y() - edge1 * edge_uv0.y()) * r).stableNormalized();
67 T_BT.row(1) = ((edge1 * edge_uv0.x() - edge0 * edge_uv1.x()) * r).stableNormalized();
68 return T_BT;
69}

80template <typename Scalar>
81bool triangle_uv_orientation(const Eigen::Matrix<Scalar, 3, 2>& uvs)
82{
83 const auto edge_uv0 = (uvs.row(1) - uvs.row(0)).eval();
84 const auto edge_uv1 = (uvs.row(2) - uvs.row(0)).eval();
85
86 const Scalar uv_signed_area = (edge_uv0.x() * edge_uv1.y() - edge_uv0.y() * edge_uv1.x());
87
88 return uv_signed_area > 0;
89}

103template <typename Scalar>
104Eigen::Matrix<Scalar, 2, 3> quad_tangent_bitangent(
105 const Eigen::Matrix<Scalar, 4, 3>& vertices,
106 const Eigen::Matrix<Scalar, 4, 2>& uvs,
107 bool& sign)
108{
109 // Check whether there are colocated vertices in UV space
110 Eigen::Vector3i indices = Eigen::Vector3i(0, 1, 2);
111 if ((uvs.row(0) - uvs.row(1)).isZero()) {
112 indices = Eigen::Vector3i(1, 2, 3);
113 } else if ((uvs.row(1) - uvs.row(2)).isZero()) {
114 indices = Eigen::Vector3i(2, 3, 0);
115 } else if ((uvs.row(2) - uvs.row(3)).isZero()) {
116 indices = Eigen::Vector3i(3, 0, 1);
117 }
118
119 Eigen::Matrix<Scalar, 3, 3> triangle_vertices;
120 triangle_vertices.row(0) = vertices.row(indices(0));
121 triangle_vertices.row(1) = vertices.row(indices(1));
122 triangle_vertices.row(2) = vertices.row(indices(2));
123
124 Eigen::Matrix<Scalar, 3, 2> triangle_uvs;
125 triangle_uvs.row(0) = uvs.row(indices(0));
126 triangle_uvs.row(1) = uvs.row(indices(1));
127 triangle_uvs.row(2) = uvs.row(indices(2));
128
129 // Calculate from triangle
130 return triangle_tangent_bitangent(triangle_vertices, triangle_uvs, sign);
131}

147template <typename MeshType, typename DerivedT, typename DerivedB>
148void accumulate_tangent_bitangent(
149 MeshType& mesh,
150 const Eigen::MatrixBase<DerivedT>& tangents,
151 const Eigen::MatrixBase<DerivedB>& bitangents)
152{
153 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
154 using Scalar = typename MeshType::Scalar;
155 using Index = typename MeshType::Index;
156 using FacetArray = typename MeshType::FacetArray;
157 using AttributeArray = typename MeshType::AttributeArray;
158 using IndexArray = Eigen::Matrix<Index, Eigen::Dynamic, 1>;
159 using RowVector3r = Eigen::Matrix<Scalar, 1, 3>;
160
161 la_runtime_assert(mesh.get_vertex_per_facet() == 3, "Only triangle meshes are supported for this.");
162 la_runtime_assert(safe_cast<Index>(tangents.rows()) == mesh.get_num_facets() * 3);
163 la_runtime_assert(safe_cast<Index>(bitangents.rows()) == mesh.get_num_facets() * 3);
164 la_runtime_assert(tangents.cols() == bitangents.cols());
165 la_runtime_assert(tangents.cols() == 3 || tangents.cols() == 4);
166
167 // Compute edge information
168 logger().trace("Corner to edge mapping");
169 IndexArray c2e;
170 corner_to_edge_mapping(mesh.get_facets(), c2e);
171 logger().trace("Chain corners around edges");
172 IndexArray e2c;
173 IndexArray next_corner_around_edge;
174 chain_corners_around_edges(mesh.get_facets(), c2e, e2c, next_corner_around_edge);
175 logger().trace("Chain corners around vertices");
176 IndexArray v2c;
177 IndexArray next_corner_around_vertex;
178 chain_corners_around_vertices(
179 mesh.get_num_vertices(),
180 mesh.get_facets(),
181 v2c,
182 next_corner_around_vertex);
183
184 const auto nvpf = mesh.get_vertex_per_facet();
185 const auto& vertices = mesh.get_vertices();
186 const auto& facets = mesh.get_facets();
187 const auto& uv_values = std::get<0>(mesh.get_indexed_attribute("uv"));
188 const auto& uv_indices = std::get<1>(mesh.get_indexed_attribute("uv"));
189 const auto& normal_values = std::get<0>(mesh.get_indexed_attribute("normal"));
190 const auto& normal_indices = std::get<1>(mesh.get_indexed_attribute("normal"));
191
192 // Check the uv triangle orientation
193 logger().trace("Is orient preserving?");
194 std::vector<unsigned char> is_orient_preserving(mesh.get_num_facets());
195 tbb::parallel_for(Index(0), mesh.get_num_facets(), [&](Index f) {
196 Eigen::Matrix<Scalar, 3, 2> uvs;
197 uvs.row(0) = uv_values.row(uv_indices(f, 0));
198 uvs.row(1) = uv_values.row(uv_indices(f, 1));
199 uvs.row(2) = uv_values.row(uv_indices(f, 2));
200 is_orient_preserving[f] = triangle_uv_orientation(uvs);
201 });
202
203 // Iterate over facets incident to eij until fi and fj are found. The edge is considered smooth
204 // if both corners at (eij, fi) and (eij, fj) share the same uv and normal indices.
205 auto is_edge_smooth = [&](Index eij, Index fi, Index fj) {
206 std::array<Index, 2> v, n, uv;
207 std::fill(v.begin(), v.end(), invalid<Index>());
208 for (Index c = e2c[eij]; c != invalid<Index>(); c = next_corner_around_edge[c]) {
209 Index f = c / 3;
210 Index lv = c % 3;
211 if (f == fi || f == fj) {
212 Index v0 = facets(f, lv);
213 Index v1 = facets(f, (lv + 1) % 3);
214 Index n0 = normal_indices(f, lv);
215 Index n1 = normal_indices(f, (lv + 1) % 3);
216 Index uv0 = uv_indices(f, lv);
217 Index uv1 = uv_indices(f, (lv + 1) % 3);
218 if (v[0] == invalid<Index>()) {
219 v[0] = v0;
220 v[1] = v1;
221 n[0] = n0;
222 n[1] = n1;
223 uv[0] = uv0;
224 uv[1] = uv1;
225 continue;
226 }
227 if (v0 != v[0]) {
228 std::swap(v0, v1);
229 std::swap(n0, n1);
230 std::swap(uv0, uv1);
231 }
232 assert(v0 == v[0]);
233 return (n0 == n[0] && n1 == n[1] && uv0 == uv[0] && uv1 == uv[1]);
234 }
235 }
236 assert(false);
237 return false;
238 };
239
240 // Check if two face vertices are collapsed
241 auto is_face_degenerate = [&](Index f) {
242 for (Index lv = 0; lv < nvpf; ++lv) {
243 if (facets(f, lv) == facets(f, (lv + 1) % nvpf)) {
244 return true;
245 }
246 }
247 return false;
248 };
249
250 // STEP 1: For each vertex v, iterate over each pair of incident facet (fi, fj) that share a
251 // common edge eij.
252 logger().trace("Loop to unify corner indices");
253 DisjointSets<Index> unified_indices((Index)tangents.rows());
254 tbb::parallel_for(Index(0), Index(mesh.get_num_vertices()), [&](Index v) {
255 for (Index ci = v2c[v]; ci != invalid<Index>(); ci = next_corner_around_vertex[ci]) {
256 Index eij = c2e[ci];
257 Index fi = ci / 3;
258 Index lvi = ci % 3;
259 Index vi = facets(fi, lvi);
260 if (is_face_degenerate(fi)) continue;
261
262 auto si = is_orient_preserving[fi];
263 for (Index cj = e2c[eij]; cj != invalid<Index>(); cj = next_corner_around_edge[cj]) {
264 Index fj = cj / 3;
265 Index lvj = cj % 3;
266 auto sj = is_orient_preserving[fj];
267 if (fi == fj) continue;
268 if (si != sj) continue;
269 Index vj = facets(fj, lvj);
270 if (vi != vj) {
271 lvj = (lvj + 1) % 3;
272 vj = facets(fj, lvj);
273 assert(vi == vj);
274 }
275
276 if (is_edge_smooth(eij, fi, fj)) {
277 unified_indices.merge(ci, fj * 3 + lvj);
278 }
279 }
280 }
281 });
282
283 // STEP 2: Perform averaging and reindex attribute
284 logger().trace("Compute new indices");
285 const Index num_corners = (Index)tangents.rows();
286 std::vector<Index> repr(num_corners, invalid<Index>());
287 Index num_indices = 0;
288 for (Index n = 0; n < num_corners; ++n) {
289 Index r = unified_indices.find(n);
290 if (repr[r] == invalid<Index>()) {
291 repr[r] = num_indices++;
292 }
293 repr[n] = repr[r];
294 }
295 logger().trace("Compute offsets");
296 std::vector<Index> indices(num_corners);
297 std::vector<Index> offsets(num_indices + 1, 0);
298 for (Index c = 0; c < num_corners; ++c) {
299 offsets[repr[c] + 1]++;
300 }
301 for (Index r = 1; r <= num_indices; ++r) {
302 offsets[r] += offsets[r - 1];
303 }
304 {
305 // Bucket sort for corner indices
306 std::vector<Index> counter = offsets;
307 for (Index c = 0; c < num_corners; ++c) {
308 indices[counter[repr[c]]++] = c;
309 }
310 }
311
312 logger().trace("Project and average tangent/bitangent");
313 FacetArray tangent_indices(mesh.get_num_facets(), 3);
314 AttributeArray tangent_values(num_indices, tangents.cols());
315 AttributeArray bitangent_values(num_indices, bitangents.cols());
316 tangent_values.setZero();
317 bitangent_values.setZero();
318 tbb::parallel_for(Index(0), Index(offsets.size() - 1), [&](Index r) {
319 for (Index i = offsets[r]; i < offsets[r + 1]; ++i) {
320 Index c = indices[i];
321 Index f = c / 3;
322 Index lv = c % 3;
323 assert(repr[c] == r);
324
325 // Project tangent frame onto the tangent plane
326 RowVector3r nrm = normal_values.row(normal_indices(f, lv));
327 RowVector3r t = tangents.row(c).template head<3>();
328 RowVector3r bt = bitangents.row(c).template head<3>();
329 t = project_on_plane(t, nrm).stableNormalized();
330 bt = project_on_plane(bt, nrm).stableNormalized();
331
332 // Compute weight as the angle between the (projected) edges
333 RowVector3r v0 = vertices.row(facets(f, lv));
334 RowVector3r v1 = vertices.row(facets(f, (lv + 1) % 3));
335 RowVector3r v2 = vertices.row(facets(f, (lv + 2) % 3));
336 RowVector3r e01 = v1 - v0;
337 RowVector3r e02 = v2 - v0;
338 Scalar w = projected_angle_between(e01, e02, nrm);
339
340 tangent_values.row(r).template head<3>() += t * w;
341 bitangent_values.row(r).template head<3>() += bt * w;
342 if (tangents.cols() == 4) {
343 if (tangent_values(r, 3) == 0) {
344 tangent_values(r, 3) = tangents(c, 3);
345 bitangent_values(r, 3) = bitangents(c, 3);
346 } else {
347 assert(tangent_values(r, 3) == tangents(c, 3));
348 assert(bitangent_values(r, 3) == bitangents(c, 3));
349 }
350 }
351
352 tangent_indices(f, lv) = r;
353 }
354 });
355 logger().trace("Normalize vectors");
356 tbb::parallel_for(Index(0), Index(tangent_values.rows()), [&](Index c) {
357 tangent_values.row(c).template head<3>().stableNormalize();
358 bitangent_values.row(c).template head<3>().stableNormalize();
359 });
360
361 logger().trace("Write attributes");
362 FacetArray bitangent_indices = tangent_indices;
363 mesh.add_indexed_attribute("tangent");
364 mesh.add_indexed_attribute("bitangent");
365 mesh.import_indexed_attribute("tangent", tangent_values, tangent_indices);
366 mesh.import_indexed_attribute("bitangent", bitangent_values, bitangent_indices);
367}

383template <typename MeshType, typename DerivedT, typename DerivedB>
384void corner_tangent_bitangent_raw(
385 const MeshType& mesh,
386 Eigen::PlainObjectBase<DerivedT>& T,
387 Eigen::PlainObjectBase<DerivedB>& BT,
388 bool pad_with_sign)
389{
390 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
391
392 la_runtime_assert(mesh.get_vertices().cols() == 3, "Mesh must be 3D");
393 la_runtime_assert(mesh.is_uv_initialized(), "UVs must be initialized");
394
395 using Scalar = typename MeshType::Scalar;
396 using Index = typename MeshType::Index;
397
398 const auto& F = mesh.get_facets();
399 const auto& V = mesh.get_vertices();
400 const auto& UV_val = mesh.get_uv();
401 const auto& UV_indices = mesh.get_uv_indices();
402
403 const auto n_faces = mesh.get_num_facets();
404 const auto per_facet = mesh.get_vertex_per_facet();
405
406 T.resize(n_faces * per_facet, pad_with_sign ? 4 : 3);
407 BT.resize(n_faces * per_facet, pad_with_sign ? 4 : 3);
408
409 // Triangular mesh
410 if (per_facet == 3) {
411 tbb::parallel_for(Index(0), Index(F.rows()), [&](Index i) {
412 const auto& indices = F.row(i);
413 Eigen::Matrix<Scalar, 3, 3> vertices;
414 vertices.row(0) = V.row(indices(0));
415 vertices.row(1) = V.row(indices(1));
416 vertices.row(2) = V.row(indices(2));
417
418 Eigen::Matrix<Scalar, 3, 2> uvs;
419 uvs.row(0) = UV_val.row(UV_indices(i, 0));
420 uvs.row(1) = UV_val.row(UV_indices(i, 1));
421 uvs.row(2) = UV_val.row(UV_indices(i, 2));
422
423 bool sign;
424 auto T_BT = triangle_tangent_bitangent(vertices, uvs, sign);
425
426 for (Index k = 0; k < per_facet; k++) {
427 T.row(i * per_facet + k).template head<3>() = T_BT.row(0);
428 BT.row(i * per_facet + k).template head<3>() = T_BT.row(1);
429 if (pad_with_sign) {
430 T.row(i * per_facet + k).w() = Scalar(sign ? 1 : -1);
431 BT.row(i * per_facet + k).w() = Scalar(sign ? 1 : -1);
432 }
433 }
434 });
435 }
436 // Quad or mixed mesh
437 else if (per_facet == 4) {
438 tbb::parallel_for(Index(0), Index(F.rows()), [&](Index i) {
439 const auto& indices = F.row(i);
440
441 bool sign;
442 Eigen::Matrix<Scalar, 2, 3> T_BT;
443
444 // Triangle in mixed mesh
445 if (indices(3) == invalid<typename MeshType::Index>()) {
446 Eigen::Matrix<Scalar, 3, 3> vertices;
447 vertices.row(0) = V.row(indices(0));
448 vertices.row(1) = V.row(indices(1));
449 vertices.row(2) = V.row(indices(2));
450
451 Eigen::Matrix<Scalar, 3, 2> uvs;
452 uvs.row(0) = UV_val.row(UV_indices(i, 0));
453 uvs.row(1) = UV_val.row(UV_indices(i, 1));
454 uvs.row(2) = UV_val.row(UV_indices(i, 2));
455
456 T_BT = triangle_tangent_bitangent(vertices, uvs, sign);
457 }
458 // Quad
459 else {
460 Eigen::Matrix<Scalar, 4, 3> vertices;
461 vertices.row(0) = V.row(indices(0));
462 vertices.row(1) = V.row(indices(1));
463 vertices.row(2) = V.row(indices(2));
464 vertices.row(3) = V.row(indices(3));
465
466 Eigen::Matrix<Scalar, 4, 2> uvs;
467 uvs.row(0) = UV_val.row(UV_indices(i, 0));
468 uvs.row(1) = UV_val.row(UV_indices(i, 1));
469 uvs.row(2) = UV_val.row(UV_indices(i, 2));
470 uvs.row(3) = UV_val.row(UV_indices(i, 3));
471
472 T_BT = quad_tangent_bitangent(vertices, uvs, sign);
473 }
474
475 for (Index k = 0; k < per_facet; k++) {
476 if (indices(k) == invalid<typename MeshType::Index>()) break;
477 T.row(i * per_facet + k).template head<3>() = T_BT.row(0);
478 BT.row(i * per_facet + k).template head<3>() = T_BT.row(1);
479 if (pad_with_sign) {
480 T.row(i * per_facet + k).w() = Scalar(sign ? 1 : -1);
481 BT.row(i * per_facet + k).w() = Scalar(sign ? 1 : -1);
482 }
483 }
484 });
485 }
486}

503template <typename MeshType>
504void compute_corner_tangent_bitangent(MeshType& mesh, bool pad_with_sign = false)

517 mesh.import_corner_attribute("tangent", T);
518 mesh.import_corner_attribute("bitangent", BT);

532template <typename MeshType>
533void compute_indexed_tangent_bitangent(MeshType& mesh, bool pad_with_sign = false)
534{
535 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
536
537 la_runtime_assert(mesh.get_vertices().cols() == 3, "Mesh must be 3D");

544 logger().debug("Compute corner tangent info");
545 detail::corner_tangent_bitangent_raw(mesh, T, BT, pad_with_sign);
546 logger().debug("Accumulate tangent info");
547 detail::accumulate_tangent_bitangent(mesh, T, BT);
548}
