104template <typename BufferComponent>
105bool set_mesh_geometry_layout(
106 const Registry& registry,
107 const Entity mesh_geometry_entity,
108 const int& location,
109 VertexData& vertex_data)
110{
111 const auto* buffer_component = registry.try_get<BufferComponent>(mesh_geometry_entity);
112
113 vertex_data.attribute_dimensions[location] = BufferComponent::dimension;
114
115 if (buffer_component && buffer_component->buffer) {
116 vertex_data.attribute_buffers[location] = buffer_component->buffer;
117 return true;
118 } else {
119 vertex_data.attribute_buffers[location] = nullptr;
120 return false;
121 }
122}
