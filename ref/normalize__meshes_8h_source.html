26template <typename MeshTypePtr>
27void normalize_meshes(const std::vector<MeshTypePtr>& meshes)
28{
29 static_assert(MeshTrait<MeshTypePtr>::is_mesh_ptr(), "Input type is not a Mesh pointer type");
30 using MeshType = typename std::pointer_traits<MeshTypePtr>::element_type;
31 using Scalar = typename MeshType::Scalar;
32 using VertexArray = typename MeshType::VertexArray;
33 using VectorType = Eigen::Matrix<Scalar, 1, VertexArray::ColsAtCompileTime>;
34
35 if (meshes.empty()) {
36 return;
37 }
38
39 const auto DIM = meshes.front()->get_vertices().cols();
40
41 VectorType min_pos = VectorType::Constant(DIM, std::numeric_limits<Scalar>::max());
42 VectorType max_pos = VectorType::Constant(DIM, std::numeric_limits<Scalar>::lowest());
43
44 for (const auto& mesh : meshes) {
45 const auto& V = mesh->get_vertices();
46 min_pos = min_pos.cwiseMin(V.colwise().minCoeff());
47 max_pos = max_pos.cwiseMax(V.colwise().maxCoeff());
48 }
49
50 const Scalar scaling = Scalar(1) / (max_pos - min_pos).maxCoeff();
51 const auto origin = Scalar(0.5) * (min_pos + max_pos);
52
53 for (auto& mesh : meshes) {
54 VertexArray vertices;
55 mesh->export_vertices(vertices);
56 vertices = (vertices.array().rowwise() - origin.array()) * scaling;
57 mesh->import_vertices(vertices);
58 }
59}

61template <typename MeshType>
62void normalize_mesh(MeshType& mesh)
63{
64 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
65 std::vector<MeshType*> vec = {&mesh};
66 normalize_meshes(vec);
67}
