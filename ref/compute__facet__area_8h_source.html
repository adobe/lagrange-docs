34template <typename MeshType>
35auto compute_triangle_areas(const MeshType& mesh) -> AttributeArrayOf<MeshType>
36{
37 using Scalar = typename MeshType::Scalar;
38 using Index = typename MeshType::Index;
39 using Vector = Eigen::Matrix<Scalar, 3, 1>;
40
41 const Index dim = mesh.get_dim();
42 const Index num_facets = mesh.get_num_facets();
43 const auto& vertices = mesh.get_vertices();
44 const auto& facets = mesh.get_facets();
45 AttributeArrayOf<MeshType> areas(num_facets, 1);
46
47 if (dim == 2) {
48 for (Index i = 0; i < num_facets; i++) {
49 const Index i0 = facets(i, 0);
50 const Index i1 = facets(i, 1);
51 const Index i2 = facets(i, 2);
52 Vector v0, v1, v2;
53 v0 << vertices(i0, 0), vertices(i0, 1), 0.0;
54 v1 << vertices(i1, 0), vertices(i1, 1), 0.0;
55 v2 << vertices(i2, 0), vertices(i2, 1), 0.0;
56 Vector e1 = v1 - v0;
57 Vector e2 = v2 - v0;
58 areas(i, 0) = Scalar(0.5) * (e1.cross(e2)).norm();
59 }
60 } else if (dim == 3) {
61 for (Index i = 0; i < num_facets; i++) {
62 const Index i0 = facets(i, 0);
63 const Index i1 = facets(i, 1);
64 const Index i2 = facets(i, 2);
65 Vector v0, v1, v2;
66 v0 << vertices(i0, 0), vertices(i0, 1), vertices(i0, 2);
67 v1 << vertices(i1, 0), vertices(i1, 1), vertices(i1, 2);
68 v2 << vertices(i2, 0), vertices(i2, 1), vertices(i2, 2);
69 Vector e1 = v1 - v0;
70 Vector e2 = v2 - v0;
71 areas(i, 0) = Scalar(0.5) * (e1.cross(e2)).norm();
72 }
73 } else {
74 throw std::runtime_error("Unsupported dimention.");
75 }
76
77 return areas;
78}

89template <typename MeshType>
90auto compute_quad_areas(const MeshType& mesh) -> AttributeArrayOf<MeshType>
91{
92 using Scalar = typename MeshType::Scalar;
93 using Index = typename MeshType::Index;
94 using Vector = Eigen::Matrix<Scalar, 3, 1>;
95
96 const Index dim = mesh.get_dim();
97 const Index num_facets = mesh.get_num_facets();
98 const auto& vertices = mesh.get_vertices();
99 const auto& facets = mesh.get_facets();
100 AttributeArrayOf<MeshType> areas(num_facets, 1);
101
102 if (dim == 2) {
103 for (Index i = 0; i < num_facets; i++) {
104 const Index i0 = facets(i, 0);
105 const Index i1 = facets(i, 1);
106 const Index i2 = facets(i, 2);
107 const Index i3 = facets(i, 3);
108 Vector v0, v1, v2, v3;
109 v0 << vertices(i0, 0), vertices(i0, 1), 0.0f;
110 v1 << vertices(i1, 0), vertices(i1, 1), 0.0f;
111 v2 << vertices(i2, 0), vertices(i2, 1), 0.0f;
112 v3 << vertices(i3, 0), vertices(i3, 1), 0.0f;
113 Vector e10 = v1 - v0;
114 Vector e30 = v3 - v0;
115 Vector e12 = v1 - v2;
116 Vector e32 = v3 - v2;
117 areas(i, 0) = 0.5f * (e10.cross(e30)).norm() + 0.5f * (e12.cross(e32)).norm();
118 }
119 } else if (dim == 3) {
120 for (Index i = 0; i < num_facets; i++) {
121 const Index i0 = facets(i, 0);
122 const Index i1 = facets(i, 1);
123 const Index i2 = facets(i, 2);
124 const Index i3 = facets(i, 3);
125 Vector v0, v1, v2, v3;
126 v0 << vertices(i0, 0), vertices(i0, 1), vertices(i0, 2);
127 v1 << vertices(i1, 0), vertices(i1, 1), vertices(i1, 2);
128 v2 << vertices(i2, 0), vertices(i2, 1), vertices(i2, 2);
129 v3 << vertices(i3, 0), vertices(i3, 1), vertices(i3, 2);
130 Vector e10 = v1 - v0;
131 Vector e30 = v3 - v0;
132 Vector e12 = v1 - v2;
133 Vector e32 = v3 - v2;
134 areas(i, 0) = 0.5f * (e10.cross(e30)).norm() + 0.5f * (e12.cross(e32)).norm();
135 }
136 } else {
137 throw std::runtime_error("Unsupported dimention.");
138 }
139
140 return areas;
141}

153template <typename MeshType>
154auto compute_facet_area_raw(const MeshType& mesh) -> AttributeArrayOf<MeshType>
155{
156 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
157 using Index = typename MeshType::Index;
158 const Index vertex_per_facet = mesh.get_vertex_per_facet();
159 if (vertex_per_facet == 3) {
160 return internal::compute_triangle_areas(mesh);
161 } else if (vertex_per_facet == 4) {
162 return internal::compute_quad_areas(mesh);
163 } else {
164 throw std::runtime_error("Unsupported facet type.");
165 }
166}

175template <typename MeshType>
176void compute_facet_area(MeshType& mesh)
177{
178 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
179 auto areas = compute_facet_area_raw(mesh);
180 mesh.add_facet_attribute("area");
181 mesh.import_facet_attribute("area", areas);
182}

195template <typename DerivedUV, typename DerivedF>
196Eigen::Matrix<typename DerivedUV::Scalar, Eigen::Dynamic, 1> compute_uv_area_raw(
197 const Eigen::MatrixBase<DerivedUV>& uv,
198 const Eigen::MatrixBase<DerivedF>& triangles)
199{
200 using Scalar = typename DerivedUV::Scalar;
201 using Index = typename DerivedF::Scalar;
202
203 la_runtime_assert(uv.cols() == 2);
204 la_runtime_assert(triangles.cols() == 3);
205 const auto num_triangles = triangles.rows();
206
207 auto compute_single_triangle_area = [&uv, &triangles](Index i) {
208 const auto& f = triangles.row(i);
209 const auto& v0 = uv.row(f[0]);
210 const auto& v1 = uv.row(f[1]);
211 const auto& v2 = uv.row(f[2]);
212 return 0.5f * (v0[0] * v1[1] + v1[0] * v2[1] + v2[0] * v0[1] - v0[0] * v2[1] -
213 v1[0] * v0[1] - v2[0] * v1[1]);
214 };
215
216 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> areas(num_triangles);
217 for (auto i : range(safe_cast<Index>(num_triangles))) {
218 areas[i] = compute_single_triangle_area(i);
219 }
220 return areas;
221}
