45template <Policy policy, typename Scalar, typename Index, typename InputMeshType>
46SurfaceMesh<Scalar, Index> to_surface_mesh_internal(InputMeshType&& mesh)
47{
48 using CMeshType = std::remove_reference_t<InputMeshType>;
49 using MeshType = std::decay_t<InputMeshType>;
50 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
51 using AttributeArray = typename MeshType::AttributeArray;
52 using IndexArray = typename MeshType::IndexArray;
53 using MeshScalar = typename MeshType::Scalar;
54 using MeshIndex = typename MeshType::Index;
55
56 if constexpr (policy == Policy::Wrap) {
57 using VertexArray = typename MeshType::VertexArray;
58 using FacetArray = typename MeshType::FacetArray;
59 static_assert(
60 std::is_lvalue_reference_v<InputMeshType&&>,
61 "Input mesh type must be a lvalue reference when wrapping a mesh object.");
62 static_assert(VertexArray::IsRowMajor, "Input vertex array must have row major storage");
63 static_assert(FacetArray::IsRowMajor, "Input facet array must have row major storage");
64 }
65
66#define LA_X_to_surface_mesh_scalar(_, S) || std::is_same_v<Scalar, S>
67 static_assert(
68 false LA_SURFACE_MESH_SCALAR_X(to_surface_mesh_scalar, 0),
69 "Scalar type should be float or double");
70#undef LA_X_to_surface_mesh_scalar
71
72#define LA_X_to_surface_mesh_index(_, I) || std::is_same_v<Index, I>
73 static_assert(
74 false LA_SURFACE_MESH_INDEX_X(to_surface_mesh_index, 0),
75 "Index type should be uint32_t or uint64_t");
76#undef LA_X_to_surface_mesh_index
77
78 // 1st - Transfer vertices positions and facets indices
79 SurfaceMesh<Scalar, Index> new_mesh(static_cast<Index>(mesh.get_dim()));
80 if constexpr (policy == Policy::Copy) {
81 // Copy vertex/facet buffer
82 new_mesh.add_vertices(static_cast<Index>(mesh.get_num_vertices()));
83 new_mesh.add_polygons(static_cast<Index>(mesh.get_num_facets()),
84 static_cast<Index>(mesh.get_vertex_per_facet()));
85 if (mesh.get_num_vertices() > 0) {
86 vertex_ref(new_mesh) = mesh.get_vertices().template cast<Scalar>();
87 }
88 if (mesh.get_num_facets() > 0) {
89 facet_ref(new_mesh) = mesh.get_facets().template cast<Index>();
90 }
91 } else if constexpr (policy == Policy::Wrap) {
92 // Wrap vertex/facet buffer, but only if types are compatible
93 static_assert(std::is_same_v<Scalar, MeshScalar>, "Mesh scalar type mismatch");
94 static_assert(std::is_same_v<Index, MeshIndex>, "Mesh index type mismatch");
95 if constexpr (std::is_const_v<CMeshType>) {
96 const auto& vertices = mesh.get_vertices();
97 const auto& facets = mesh.get_facets();
98 new_mesh.wrap_as_const_vertices(
99 {vertices.data(), safe_cast<size_t>(vertices.size())},
100 mesh.get_num_vertices());
101 new_mesh.wrap_as_const_facets(
102 {facets.data(), safe_cast<size_t>(facets.size())},
103 mesh.get_num_facets(),
104 mesh.get_vertex_per_facet());
105 } else {
106 auto& vertices = mesh.ref_vertices();
107 auto& facets = mesh.ref_facets();
108 new_mesh.wrap_as_vertices(
109 {vertices.data(), safe_cast<size_t>(vertices.size())},
110 mesh.get_num_vertices());
111 new_mesh.wrap_as_facets(
112 {facets.data(), safe_cast<size_t>(facets.size())},
113 mesh.get_num_facets(),
114 mesh.get_vertex_per_facet());
115 }
116 }
117
118 // 2nd - Transfer edge indices
119 if (mesh.is_edge_data_initialized()) {
120 new_mesh.initialize_edges(
121 static_cast<Index>(mesh.get_num_edges()),
122 [&](Index e) -> std::array<Index, 2> {
123 auto v = mesh.get_edge_vertices(static_cast<typename MeshType::Index>(e));
124 return {static_cast<Index>(v[0]), static_cast<Index>(v[1])};
125 });
126 }
127
128 // 3rd - Transfer attributes
129 auto usage_from_name = [](std::string_view name) {
130 if (starts_with(name, AttributeName::normal)) {
131 return AttributeUsage::Normal;
132 }
133 // "uv" is for legacy code with hardcoded "uv" name.
134 if (starts_with(name, AttributeName::texcoord) || starts_with(name, "uv")) {
135 return AttributeUsage::UV;
136 }
137 if (starts_with(name, AttributeName::color)) {
138 return AttributeUsage::Color;
139 }
140 return AttributeUsage::Vector;
141 };
142
143 auto get_unique_name = [&](auto name) -> std::string {
144 if (!new_mesh.has_attribute(name)) {
145 return name;
146 } else {
147 std::string new_name;
148 for (int cnt = 0; cnt < 1000; ++cnt) {
149 new_name = fmt::format("{}.{}", name, cnt);
150 if (!new_mesh.has_attribute(new_name)) {
151 return new_name;
152 }
153 }
154 throw Error(fmt::format("Could not assign a unique attribute name for: {}", name));
155 }
156 };
157
158 auto transfer_attribute = [&](auto&& name, auto&& array, AttributeElement elem) {
159 std::string new_name = get_unique_name(name);
160 decltype(auto) attr = array->template get<AttributeArray>();
161 if constexpr (policy == Policy::Copy) {
162 new_mesh.template create_attribute<MeshScalar>(
163 new_name,
164 elem,
165 usage_from_name(name),
166 attr.cols());
167 attribute_matrix_ref<MeshScalar>(new_mesh, new_name) = attr;
168 } else if constexpr (policy == Policy::Wrap) {
169 if constexpr (std::is_const_v<CMeshType>) {
170 new_mesh.template wrap_as_const_attribute<MeshScalar>(
171 new_name,
172 elem,
173 usage_from_name(name),
174 attr.cols(),
175 {attr.data(), static_cast<size_t>(attr.size())});
176
177 } else {
178 new_mesh.template wrap_as_attribute<MeshScalar>(
179 new_name,
180 elem,
181 usage_from_name(name),
182 attr.cols(),
183 {attr.data(), static_cast<size_t>(attr.size())});
184 }
185 }
186 };
187
188 // Transfer vertex attributes
189 for (const auto& name : mesh.get_vertex_attribute_names()) {
190 transfer_attribute(name, mesh.get_vertex_attribute_array(name), AttributeElement::Vertex);
191 }
192
193 // Transfer facet attributes
194 for (const auto& name : mesh.get_facet_attribute_names()) {
195 transfer_attribute(name, mesh.get_facet_attribute_array(name), AttributeElement::Facet);
196 }
197
198 // Transfer corner attributes
199 for (const auto& name : mesh.get_corner_attribute_names()) {
200 transfer_attribute(name, mesh.get_corner_attribute_array(name), AttributeElement::Corner);
201 }
202
203 // Transfer edge attributes
204 for (const auto& name : mesh.get_edge_attribute_names()) {
205 transfer_attribute(name, mesh.get_edge_attribute_array(name), AttributeElement::Edge);
206 }
207
208 // Transfer indexed attributes
209 for (const auto& name : mesh.get_indexed_attribute_names()) {
210 decltype(auto) attr = mesh.get_indexed_attribute_array(name);
211 decltype(auto) values = std::get<0>(attr)->template get<AttributeArray>();
212 decltype(auto) indices = std::get<1>(attr)->template get<IndexArray>();
213 std::string new_name = get_unique_name(name);
214
215 if constexpr (policy == Policy::Wrap) {
216 static_assert(std::is_same_v<Index, MeshIndex>, "Mesh attribute index type mismatch");
217 if (std::is_const_v<CMeshType>) {
218 new_mesh.template wrap_as_const_indexed_attribute<MeshScalar>(
219 new_name,
220 usage_from_name(name),
221 values.rows(),
222 values.cols(),
223 {values.data(), static_cast<size_t>(values.size())},
224 {indices.data(), static_cast<size_t>(indices.size())});
225 } else {
226 new_mesh.template wrap_as_indexed_attribute<MeshScalar>(
227 new_name,
228 usage_from_name(name),
229 values.rows(),
230 values.cols(),
231 {values.data(), static_cast<size_t>(values.size())},
232 {indices.data(), static_cast<size_t>(indices.size())});
233 }
234 } else if constexpr (policy == Policy::Copy) {
235 auto id = new_mesh.template create_attribute<MeshScalar>(
236 new_name,
237 AttributeElement::Indexed,
238 usage_from_name(name),
239 values.cols());
240 auto& new_attr = new_mesh.template ref_indexed_attribute<MeshScalar>(id);
241 new_attr.values().resize_elements(values.rows());
242 matrix_ref(new_attr.values()) = values;
243 reshaped_ref(new_attr.indices(), indices.cols()) = indices.template cast<Index>();
244 }
245 }
246
247 return new_mesh;
248}

278template <typename Index, typename Func>
279std::vector<Index> fast_edge_sort(
280 Index num_edges,
281 Index num_vertices,
282 Func get_edge,
283 span<Index> vertex_to_first_edge = {})
284{
285 std::vector<Index> local_buffer;
286 if (vertex_to_first_edge.empty()) {
287 local_buffer.assign(num_vertices + 1, 0);
288 vertex_to_first_edge = local_buffer;
289 } else {
290 std::fill(vertex_to_first_edge.begin(), vertex_to_first_edge.end(), Index(0));
291 }
292 la_runtime_assert(vertex_to_first_edge.size() == static_cast<size_t>(num_vertices) + 1);
293 // Count number of edges starting at each vertex
294 for (Index e = 0; e < num_edges; ++e) {
295 std::array<Index, 2> v = get_edge(e);
296 if (v[0] > v[1]) {
297 std::swap(v[0], v[1]);
298 }
299 vertex_to_first_edge[v[0] + 1]++;
300 }
301 // Prefix sum to compute actual offsets
302 std::partial_sum(
303 vertex_to_first_edge.begin(),
304 vertex_to_first_edge.end(),
305 vertex_to_first_edge.begin());
306 la_runtime_assert(vertex_to_first_edge.back() == num_edges);
307 // Bucket each edge id to its respective starting vertex
308 std::vector<Index> edge_ids(num_edges);
309 for (Index e = 0; e < num_edges; ++e) {
310 std::array<Index, 2> v = get_edge(e);
311 if (v[0] > v[1]) {
312 std::swap(v[0], v[1]);
313 }
314 edge_ids[vertex_to_first_edge[v[0]]++] = e;
315 }
316 // Shift back the offset buffer 'vertex_to_first_edge' (can use std::shift_right in C++20 :p)
317 std::rotate(
318 vertex_to_first_edge.rbegin(),
319 vertex_to_first_edge.rbegin() + 1,
320 vertex_to_first_edge.rend());
321 vertex_to_first_edge.front() = 0;
322 // Sort each bucket in parallel
323 tbb::parallel_for(Index(0), num_vertices, [&](Index v) {
324 tbb::parallel_sort(
325 edge_ids.begin() + vertex_to_first_edge[v],
326 edge_ids.begin() + vertex_to_first_edge[v + 1],
327 [&](Index ei, Index ej) {
328 auto vi = get_edge(ei);
329 auto vj = get_edge(ej);
330 if (vi[0] > vi[1]) std::swap(vi[0], vi[1]);
331 if (vj[0] > vj[1]) std::swap(vj[0], vj[1]);
332 return vi < vj;
333 });
334 });
335 return edge_ids;
336}

340template <typename MeshType, typename Scalar, typename Index>
341std::unique_ptr<MeshType> to_legacy_mesh(const SurfaceMesh<Scalar, Index>& mesh)
342{
343 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
344 using VertexArray = typename MeshType::VertexArray;
345 using FacetArray = typename MeshType::FacetArray;
346 using IndexArray = typename MeshType::IndexArray;
347 using AttributeArray = typename MeshType::AttributeArray;
348 using MeshScalar = typename MeshType::Scalar;
349 using MeshIndex = typename MeshType::Index;
350
351#define LA_X_to_legacy_mesh_scalar(_, S) || std::is_same_v<Scalar, S>
352 static_assert(
353 false LA_SURFACE_MESH_SCALAR_X(to_legacy_mesh_scalar, 0),
354 "Scalar type should be float or double");
355#undef LA_X_to_legacy_mesh_scalar
356
357#define LA_X_to_legacy_mesh_index(_, I) || std::is_same_v<Index, I>
358 static_assert(
359 false LA_SURFACE_MESH_INDEX_X(to_legacy_mesh_index, 0),
360 "Index type should be uint32_t or uint64_t");
361#undef LA_X_to_legacy_mesh_index
362
363 la_runtime_assert(mesh.is_regular(), "Input polygonal mesh is not regular");
364
365 LA_IGNORE(mesh);
366 auto new_mesh = [&] {
367 VertexArray vertices;
368 FacetArray facets;
369 if (mesh.get_num_vertices() > 0) {
370 vertices = vertex_view(mesh).template cast<MeshScalar>();
371 }
372 if (mesh.get_num_facets() > 0) {
373 facets = facet_view(mesh).template cast<MeshIndex>();
374 }
375 return create_mesh<VertexArray, FacetArray>(std::move(vertices), std::move(facets));
376 }();
377
378 // If mesh contains edges, attempt to transfer them as well
379 std::vector<Index> old_edge_ids;
380 std::vector<MeshIndex> new_edge_ids;
381 if (mesh.has_edges()) {
382 logger().warn("Mesh contain edges information. A possible reordering may occur.");
383 new_mesh->initialize_edge_data();
384 la_runtime_assert(
385 mesh.get_num_edges() == static_cast<Index>(new_mesh->get_num_edges()),
386 "Number of edges do not match");
387 const auto num_vertices = static_cast<size_t>(mesh.get_num_vertices());
388 auto buffer = std::make_unique<uint8_t[]>(
389 (num_vertices + 1) * std::max(sizeof(Index), sizeof(MeshIndex)));
390 old_edge_ids = mesh_convert_detail::fast_edge_sort(
391 mesh.get_num_edges(),
392 mesh.get_num_vertices(),
393 [&](Index e) -> std::array<Index, 2> { return mesh.get_edge_vertices(e); },
394 {reinterpret_cast<Index*>(buffer.get()), num_vertices + 1});
395 new_edge_ids = mesh_convert_detail::fast_edge_sort(
396 new_mesh->get_num_edges(),
397 new_mesh->get_num_vertices(),
398 [&](MeshIndex e) -> std::array<MeshIndex, 2> { return new_mesh->get_edge_vertices(e); },
399 {reinterpret_cast<MeshIndex*>(buffer.get()), num_vertices + 1});
400 }
401
402 // Transfer non-indexed attributes
403 seq_foreach_named_attribute_read<~AttributeElement::Indexed>(
404 mesh,
405 [&](std::string_view name_view, auto&& attr) {
406 using AttributeType = std::decay_t<decltype(attr)>;
407 using ValueType = typename AttributeType::ValueType;
408 if (mesh.attr_name_is_reserved(name_view)) {
409 return;
410 }
411 std::string name(name_view);
412 AttributeArray vals =
413 attribute_matrix_view<ValueType>(mesh, name).template cast<MeshScalar>();
414 switch (attr.get_element_type()) {
415 case AttributeElement::Vertex: {
416 new_mesh->add_vertex_attribute(name);
417 new_mesh->import_vertex_attribute(name, std::move(vals));
418 break;
419 }
420 case AttributeElement::Facet: {
421 new_mesh->add_facet_attribute(name);
422 new_mesh->import_facet_attribute(name, std::move(vals));
423 break;
424 }
425 case AttributeElement::Corner: {
426 new_mesh->add_corner_attribute(name);
427 new_mesh->import_corner_attribute(name, std::move(vals));
428 break;
429 }
430 case AttributeElement::Edge: {
431 // Permute rows
432 auto old_vals = attribute_matrix_view<ValueType>(mesh, name);
433 for (Index e = 0; e < mesh.get_num_edges(); ++e) {
434 vals.row(new_edge_ids[e]) =
435 old_vals.row(old_edge_ids[e]).template cast<MeshScalar>();
436 }
437 new_mesh->add_edge_attribute(name);
438 new_mesh->import_edge_attribute(name, std::move(vals));
439 break;
440 }
441 case AttributeElement::Value:
442 logger().warn("Cannot transfer value attribute: {}", name);
443 break;
444 case AttributeElement::Indexed:
445 // Not possible by construction
446 break;
447 }
448 });
449
450 // Transfer indexed attributes
451 const auto nvpf = static_cast<size_t>(new_mesh->get_vertex_per_facet());
452 seq_foreach_named_attribute_read<AttributeElement::Indexed>(
453 mesh,
454 [&](std::string_view name_view, auto&& attr) {
455 if (mesh.attr_name_is_reserved(name_view)) {
456 return;
457 }
458 std::string name(name_view);
459 IndexArray indices = reshaped_view(attr.indices(), nvpf).template cast<MeshIndex>();
460 AttributeArray values = matrix_view(attr.values()).template cast<MeshScalar>();
461 new_mesh->add_indexed_attribute(name);
462 new_mesh->import_indexed_attribute(name, std::move(values), std::move(indices));
463 });
464
465 return new_mesh;
466}
