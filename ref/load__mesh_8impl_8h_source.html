31template <typename MeshType>
32std::unique_ptr<MeshType> load_mesh_basic(const fs::path& filename)
33{
34 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
35 typename MeshType::VertexArray vertices;
36 typename MeshType::FacetArray facets;
37
38 igl::read_triangle_mesh(filename.string(), vertices, facets);
39
40 return create_mesh(vertices, facets);
41}

45template <typename MeshType>
46std::vector<std::unique_ptr<MeshType>> load_obj_meshes(const fs::path& filename)
47{
48 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
49
50 if (!std::is_same<tinyobj::real_t, double>::value) {
51 logger().warn("Tinyobjloader compiled to use float, loss of precision may occur.");
52 }
53
54 return load_mesh_ext<MeshType>(filename).meshes;
55}

59template <typename MeshType>
60std::unique_ptr<MeshType> load_obj_mesh(const fs::path& filename)
61{
62 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
63
64 auto res = load_obj_meshes<MeshType>(filename);
65 if (res.empty()) {
66 return nullptr;
67 } else if (res.size() == 1) {
68 return std::move(res[0]);
69 } else {
70 logger().debug("Combining {} meshes into one.", res.size());
71 return combine_mesh_list(res, true);
72 }
73}
74extern template std::unique_ptr<TriangleMesh3D> load_obj_mesh(const fs::path&);
75extern template std::unique_ptr<TriangleMesh3Df> load_obj_mesh(const fs::path&);
76
77template <typename MeshType>
78std::unique_ptr<MeshType> load_mesh(const fs::path& filename)
79{
80 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
81
82 if (filename.extension() == ".obj") {
83 return load_obj_mesh<MeshType>(filename);
84 } else if (filename.extension() == ".ply") {
85 return load_mesh_ply<MeshType>(filename);
86 } else {
87 return load_mesh_basic<MeshType>(filename);
88 }
89}
