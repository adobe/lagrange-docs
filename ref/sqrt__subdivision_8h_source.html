40template <typename DerivedVI, typename DerivedFI, typename DerivedVO, typename DerivedFO>
41void sqrt_subdivision(
42 const Eigen::MatrixBase<DerivedVI>& VI,
43 const Eigen::MatrixBase<DerivedFI>& FI,
44 Eigen::PlainObjectBase<DerivedVO>& VO,
45 Eigen::PlainObjectBase<DerivedFO>& FO)
46{
47 using Scalar = typename DerivedVI::Scalar;
48 using Index = typename DerivedFI::Scalar;
49 using MatrixXs = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>;
50 using MatrixXi = Eigen::Matrix<Index, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>;
51 using RowVector3s = Eigen::Matrix<Scalar, 1, 3>;
52
53 // Step 1: new vertices at barycenter of every face
54 MatrixXs BC;
55 igl::barycenter(VI, FI, BC);
56
57 // Step 2: move old vertices to new positions
58 VO.resize(VI.rows() + FI.rows(), VI.cols());
59 VO.topRows(VI.rows()) = VI;
60 VO.bottomRows(BC.rows()) = BC;
61 std::vector<std::vector<Index>> VV;
62 igl::adjacency_list(FI, VV);
63 for (Index i = 0; i < (Index)VI.rows(); ++i) {
64 Scalar n = static_cast<Scalar>(VV[i].size());
65 Scalar an = static_cast<Scalar>((4.0 - 2.0 * std::cos(2.0 * M_PI / n)) / 9.0);
66 RowVector3s sum_vi = RowVector3s::Zero();
67 for (auto j : VV[i]) {
68 sum_vi += VI.row(j);
69 }
70 VO.row(i) = (1 - an) * VI.row(i) + an / n * sum_vi;
71 }
72
73 // Step 3: compute a new set of faces, and flip edges as required
74 FO.resize(3 * FI.rows(), FI.cols());
75 for (Index f = 0; f < (Index)FI.rows(); ++f) {
76 FO.row(3 * f + 0) << VI.rows() + f, FI(f, 0), FI(f, 1);
77 FO.row(3 * f + 1) << VI.rows() + f, FI(f, 1), FI(f, 2);
78 FO.row(3 * f + 2) << VI.rows() + f, FI(f, 2), FI(f, 0);
79 }
80 MatrixXi TT, TTi;
81 igl::triangle_triangle_adjacency(FI, TT, TTi);
82 for (Index f = 0; f < (Index)FI.rows(); ++f) {
83 for (Index i = 0; i < (Index)FI.cols(); ++i) {
84 if (TT(f, i) != static_cast<Index>(-1) && TT(f, i) > f) {
85 Index g = TT(f, i);
86 Index j = TTi(f, i);
87 FO(3 * f + i, 2) = VI.rows() + g;
88 FO(3 * g + j, 2) = VI.rows() + f;
89 }
90 }
91 }
92}

109template <typename MeshType>
110std::unique_ptr<MeshType> sqrt_subdivision(const MeshType& mesh)
111{
112 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
113
114 using VertexArray = typename MeshType::VertexArray;
115 using FacetArray = typename MeshType::FacetArray;
116
117 VertexArray V;
118 FacetArray F;
119 sqrt_subdivision(mesh.get_vertices(), mesh.get_facets(), V, F);
120 return lagrange::create_mesh(V, F);
121}
