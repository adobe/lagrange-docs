33void bind_attribute(nanobind::module_& m)
34{
35 namespace nb = nanobind;
36 using namespace nb::literals;
37
38 auto attr_class = nb::class_<PyAttribute>(m, "Attribute");
39 attr_class.def_property_readonly("element_type", [](PyAttribute& self) {
40 return self->get_element_type();
41 });
42 attr_class.def_property_readonly("usage", [](PyAttribute& self) { return self->get_usage(); });
43 attr_class.def_property_readonly("num_channels", [](PyAttribute& self) {
44 return self->get_num_channels();
45 });
46
47 attr_class.def_property(
48 "default_value",
49 [](PyAttribute& self) {
50 return self.process([](auto& attr) { return nb::cast(attr.get_default_value()); });
51 },
52 [](PyAttribute& self, double val) {
53 self.process([&](auto& attr) {
54 using ValueType = typename std::decay_t<decltype(attr)>::ValueType;
55 attr.set_default_value(static_cast<ValueType>(val));
56 });
57 });
58 attr_class.def_property(
59 "growth_policy",
60 [](PyAttribute& self) {
61 return self.process([](auto& attr) { return attr.get_growth_policy(); });
62 },
63 [](PyAttribute& self, AttributeGrowthPolicy policy) {
64 self.process([&](auto& attr) { attr.set_growth_policy(policy); });
65 });
66 attr_class.def_property(
67 "write_policy",
68 [](PyAttribute& self) {
69 return self.process([](auto& attr) { return attr.get_write_policy(); });
70 },
71 [](PyAttribute& self, AttributeWritePolicy policy) {
72 self.process([&](auto& attr) { attr.set_write_policy(policy); });
73 });
74 attr_class.def_property(
75 "copy_policy",
76 [](PyAttribute& self) {
77 return self.process([](auto& attr) { return attr.get_copy_policy(); });
78 },
79 [](PyAttribute& self, AttributeCopyPolicy policy) {
80 self.process([&](auto& attr) { attr.set_copy_policy(policy); });
81 });
82 attr_class.def("create_internal_copy", [](PyAttribute& self) {
83 self.process([](auto& attr) { attr.create_internal_copy(); });
84 });
85 attr_class.def("clear", [](PyAttribute& self) {
86 self.process([](auto& attr) { attr.clear(); });
87 });
88 attr_class.def("reserve_entries", [](PyAttribute& self, size_t s) {
89 self.process([=](auto& attr) { attr.reserve_entries(s); });
90 });
91 attr_class.def("insert_elements", [](PyAttribute& self, size_t num_elements) {
92 self.process([=](auto& attr) { attr.insert_elements(num_elements); });
93 });
94 attr_class.def(
95 "insert_elements",
96 [](PyAttribute& self, GenericTensor tensor) {
97 auto insert_elements = [&](auto& attr) {
98 using ValueType = typename std::decay_t<decltype(attr)>::ValueType;
99 if (tensor.dtype() != nb::dtype<ValueType>()) {
100 throw nb::type_error(fmt::format(
101 "Tensor has a unexpected dtype. Expecting {}.",
102 internal::string_from_scalar<ValueType>()));
103 }
104 Tensor<ValueType> local_tensor(tensor.handle());
105 auto [data, shape, stride] = tensor_to_span(local_tensor);
106 la_runtime_assert(is_dense(shape, stride));
107 attr.insert_elements(data);
108 };
109 self.process(insert_elements);
110 },
111 "tensor"_a);
112 attr_class.def("empty", [](PyAttribute& self) {
113 return self.process([](auto& attr) { return attr.empty(); });
114 });
115 attr_class.def_property_readonly("num_elements", [](PyAttribute& self) {
116 return self.process([](auto& attr) { return attr.get_num_elements(); });
117 });
118 attr_class.def_property_readonly("external", [](PyAttribute& self) {
119 return self.process([](auto& attr) { return attr.is_external(); });
120 });
121 attr_class.def_property_readonly("readonly", [](PyAttribute& self) {
122 return self.process([](auto& attr) { return attr.is_read_only(); });
123 });
124 attr_class.def_property(
125 "data",
126 [](PyAttribute& self) {
127 return self.process(
128 [&](auto& attr) { return attribute_to_tensor(attr, nb::cast(&self)); });
129 },
130 [](PyAttribute& self, GenericTensor tensor) {
131 auto wrap_tensor = [&](auto& attr) {
132 using ValueType = typename std::decay_t<decltype(attr)>::ValueType;
133 if (tensor.dtype() != nb::dtype<ValueType>()) {
134 throw nb::type_error(fmt::format(
135 "Tensor has a unexpected dtype. Expecting {}.",
136 internal::string_from_scalar<ValueType>()));
137 }
138 Tensor<ValueType> local_tensor(tensor.handle());
139 auto [data, shape, stride] = tensor_to_span(local_tensor);
140 la_runtime_assert(is_dense(shape, stride));
141 la_runtime_assert(shape.size() == 1 ? 1 == attr.get_num_channels() : true);
142 la_runtime_assert(shape.size() == 2 ? shape[1] == attr.get_num_channels() : true);
143 const size_t num_elements = shape[0];
144
145 auto owner = std::make_shared<nb::object>(nb::cast(tensor));
146 attr.wrap(make_shared_span(owner, data.data(), data.size()), num_elements);
147 };
148 self.process(wrap_tensor);
149 });
150}
