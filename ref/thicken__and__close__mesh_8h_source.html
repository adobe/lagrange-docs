46template <typename MeshType>
47std::unique_ptr<MeshType> thicken_and_close_mesh(
48 const MeshType& input_mesh,
49 bool use_direction_and_mirror,
50 Eigen::Matrix<ScalarOf<MeshType>, 3, 1> direction,
51 ScalarOf<MeshType> offset_amount,
52 ScalarOf<MeshType> mirror_amount,
53 typename MeshType::Index num_segments = 1)
54{
55 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
56
57 using Scalar = typename MeshType::Scalar;
58 using Index = typename MeshType::Index;
59 using VertexArray = typename MeshType::VertexArray;
60 using AttributeArray = typename MeshType::AttributeArray;
61 using UVArray = typename MeshType::UVArray;
62 using UVIndices = typename MeshType::UVIndices;
63 using FacetArray = typename MeshType::FacetArray;
64 using Vector3s = Eigen::Matrix<Scalar, 3, 1>;
65
66 la_runtime_assert(input_mesh.get_dim() == 3, "This function only supports 3D meshes.");
67 la_runtime_assert(
68 input_mesh.get_vertex_per_facet() == 3,
69 "This function only supports triangle meshes.");
70 la_runtime_assert(
71 input_mesh.is_edge_data_initialized(),
72 "This function requires the mesh to have edge data pre-initialized.");
73
74 // compute offset vertex
75 auto compute_vertex = [](const Vector3s& vertex,
76 const Vector3s& offset_vector,
77 const Vector3s& mirror_vector,
78 const Vector3s& target_direction,
79 bool with_direction_and_mirror,
80 Scalar amount) {
81 Vector3s offset_vertex = vertex + amount * offset_vector;
82 if (with_direction_and_mirror) {
83 offset_vertex -= (offset_vertex.dot(target_direction) * amount) * mirror_vector;
84 }
85 return offset_vertex;
86 };
87
88 // Sanitize parameters
89 if (num_segments < 1) num_segments = 1;
90 direction.stableNormalize();
91
92 // Prepare ancillary data
93 Vector3s offset_vector(0.0, 1.0, 0.0);
94 Vector3s mirror_vector(0.0, 0.0, 0.0);
95 AttributeArray vertex_normal;
96 if (use_direction_and_mirror) {
97 offset_vector = offset_amount * direction;
98 mirror_vector = (Scalar(1) - mirror_amount) * direction;
99 } else {
100 // compute vertex normal vector
101 std::unique_ptr<MeshType> copied_mesh = wrap_with_mesh<VertexArray, FacetArray>(
102 input_mesh.get_vertices(),
103 input_mesh.get_facets());
104 compute_vertex_normal(*copied_mesh);
105 copied_mesh->export_vertex_attribute("normal", vertex_normal);
106 la_runtime_assert(vertex_normal.rows() == input_mesh.get_num_vertices());
107 }
108 const Index num_input_vertices = input_mesh.get_num_vertices();
109 const Index num_input_facets = input_mesh.get_num_facets();
110 const bool has_uvs = input_mesh.is_uv_initialized();
111 const Index num_input_uvs = static_cast<Index>(has_uvs ? input_mesh.get_uv().rows() : 0);
112
113 // Count boundary edges and vertices in the input mesh
114 Index num_boundary_edges = 0;
115 Index num_boundary_vertices = 0;
116 std::unordered_map<Index, Index>
117 boundary_vertices; // (key is vertex index, value is a boundary-relative index)
118 for (Index e = 0; e < input_mesh.get_num_edges(); ++e) {
119 if (input_mesh.is_boundary_edge(e)) {
120 ++num_boundary_edges;
121 for (auto v : input_mesh.get_edge_vertices(e)) { // for each of two vertices
122 boundary_vertices.emplace(v, Index(boundary_vertices.size()));
123 }
124 }
125 }
126 num_boundary_vertices = static_cast<Index>(boundary_vertices.size());
127
128 // Vertices
129 // output vertices are packed as follows:
130 // 1. original vertices (num_input_vertices)
131 // 2. offset vertices (num_input_vertices)
132 // 3. stitch vertices ((num_segments - 1) * num_boundary_vertices), packed by segment
133 VertexArray offset_vertices(
134 num_input_vertices * 2 + (num_segments - 1) * num_boundary_vertices,
135 3);
136 for (Index v = 0; v < num_input_vertices; ++v) {
137 if (!use_direction_and_mirror) {
138 // compute per vertex normal and use it as offset direction
139 // TODO: a smarter version could compensate for tight angles with disjoint
140 // indexed normals and amplify this vector to preserve the apparent thickness
141 // of the resulting solid.
142 offset_vector = -vertex_normal.row(v).template head<3>() * offset_amount;
143 }
144 Vector3s vertex = input_mesh.get_vertices().row(v).template head<3>();
145 // copy original vertices
146 offset_vertices.row(v) = vertex;
147 // also opposite face
148 auto offset_vertex = compute_vertex(
149 vertex,
150 offset_vector,
151 mirror_vector,
152 direction,
153 use_direction_and_mirror,
154 1.0);
155 offset_vertices.row(num_input_vertices + v) = offset_vertex;
156
157 // is this a boundary vertex? Add intermediate vertices for the stitch
158 Scalar segment_increment = Scalar(1) / static_cast<Scalar>(num_segments);
159 if (num_segments > 1) {
160 auto vb = boundary_vertices.find(v);
161 if (vb != boundary_vertices.end()) {
162 for (Index is = 1; is < num_segments; ++is) {
163 Scalar offset_ratio = static_cast<Scalar>(is) * segment_increment;
164 assert(offset_ratio < 1.0);
165 offset_vertices.row(
166 num_input_vertices * 2 + // original + offset vertices
167 (is - 1) * num_boundary_vertices + // segment row
168 vb->second // index in the boundary
169 ) =
170 compute_vertex(
171 vertex,
172 offset_vector,
173 mirror_vector,
174 direction,
175 use_direction_and_mirror,
176 offset_ratio);
177 }
178 }
179 }
180 }
181
182
183 // Facets
184 // output facets are packed as follows:
185 // 1. original facets interleaved with flipped facets (num_input_facets*2)
186 // 2. stitch vertices (num_boundary_edges * num_boundary_vertices), packed by segment
187 FacetArray offset_facets((num_input_facets + num_boundary_edges * num_segments) * 2, 3);
188 for (Index f = 0; f < num_input_facets; ++f) {
189 const auto& facet = input_mesh.get_facets().row(f);
190 offset_facets.row(2 * f) = facet;
191 offset_facets.row(2 * f + 1) << facet[0] + num_input_vertices,
192 facet[2] + num_input_vertices,
193 facet[1] + num_input_vertices; // this face is flipped: order of vertices is reversed
194 }
195
196 // 2. Stitch
197 Index vbstart = num_input_vertices * 2;
198 for (Index e = 0, f = 2 * num_input_facets; e < input_mesh.get_num_edges(); ++e) {
199 if (input_mesh.is_boundary_edge(e)) {
200 auto edge_vertices = input_mesh.get_edge_vertices(e);
201 assert(boundary_vertices.find(edge_vertices[0]) != boundary_vertices.end());
202 assert(boundary_vertices.find(edge_vertices[1]) != boundary_vertices.end());
203 assert(f + 1 < offset_facets.rows());
204
205 for (Index is = 0; is < num_segments; ++is) {
206 Index vbstart_on_this_segment = vbstart + (is - 1) * num_boundary_vertices;
207 Index vbstart_on_next_segment = vbstart + is * num_boundary_vertices;
208 bool is_first_segment = (is == 0);
209 bool is_last_segment = (is == num_segments - 1);
210 Index v0 = is_first_segment
211 ? edge_vertices[0]
212 : vbstart_on_this_segment + boundary_vertices[edge_vertices[0]];
213 Index v1 = is_first_segment
214 ? edge_vertices[1]
215 : vbstart_on_this_segment + boundary_vertices[edge_vertices[1]];
216 Index v2 = is_last_segment
217 ? edge_vertices[0] + num_input_vertices
218 : vbstart_on_next_segment + boundary_vertices[edge_vertices[0]];
219 Index v3 = is_last_segment
220 ? edge_vertices[1] + num_input_vertices
221 : vbstart_on_next_segment + boundary_vertices[edge_vertices[1]];
222
223 assert(v0 < offset_vertices.rows());
224 assert(v1 < offset_vertices.rows());
225 assert(v2 < offset_vertices.rows());
226 assert(v3 < offset_vertices.rows());
227
228 offset_facets.row(f++) << v0, v2, v1;
229 offset_facets.row(f++) << v1, v2, v3;
230 }
231 }
232 }
233
234 auto offset_mesh = lagrange::create_mesh(std::move(offset_vertices), std::move(offset_facets));
235
236 if (has_uvs) {
237 const auto& input_uv_values = input_mesh.get_uv();
238 const auto& input_uv_indices = input_mesh.get_uv_indices();
239
240 // UV values
241 UVArray uv_values(num_input_uvs * 2 + (num_segments - 1) * num_boundary_vertices, 2);
242 for (Index u = 0; u < num_input_uvs; ++u) {
243 // copy original uv and opposite face
244 uv_values.row(u) = uv_values.row(u + num_input_uvs) = input_uv_values.row(u);
245 }
246 // additional uv values for num_segments
247 for (Index f = 0; f < input_mesh.get_facets().rows(); ++f) {
248 // find individual uvs indices
249 const auto& uv_facet = input_mesh.get_uv_indices().row(f);
250 // get face indices for this same facet
251 const auto& facet = input_mesh.get_facets().row(f);
252 // for each uv on the face:
253 for (Index fv = 0; fv < 3; ++fv) {
254 Index v = facet[fv];
255 auto vb = boundary_vertices.find(v);
256 if (vb != boundary_vertices.end()) {
257 // this is a boundary vertex.
258 Index uv_index = uv_facet[fv];
259 auto uv_value = input_mesh.get_uv().row(uv_index);
260
261 // add overlapping identical values for uvs at the stitch...
262 // at some point we might want to have non overlapping uvs. Not sure if it's in
263 // scope for this.
264 for (Index is = 1; is < num_segments; ++is) {
265 uv_values.row(
266 num_input_vertices * 2 + // original + offset vertices
267 (is - 1) * num_boundary_vertices + // segment row
268 vb->second) = uv_value;
269 }
270 }
271 }
272 }
273
274 // UV facets
275 UVIndices uv_facets((input_uv_indices.rows() + num_boundary_edges * num_segments) * 2, 3);
276 for (Index u = 0; u < input_uv_indices.rows(); ++u) {
277 const auto& uv_facet = input_uv_indices.row(u);
278 uv_facets.row(2 * u) = uv_facet;
279 uv_facets.row(2 * u + 1) << uv_facet[0] + num_input_uvs, uv_facet[2] + num_input_uvs,
280 uv_facet[1] + num_input_uvs;
281 }
282
283 // Stitch
284 Index uvbstart = static_cast<Index>(2 * input_uv_values.rows());
285 for (Index e = 0, f_uv = 2 * static_cast<Index>(input_uv_indices.rows());
286 e < input_mesh.get_num_edges();
287 ++e) {
288 if (input_mesh.is_boundary_edge(e)) {
289 // Find first and only face on this edge
290 const Index f = input_mesh.get_one_facet_around_edge(e);
291 assert(f != lagrange::invalid<Index>());
292 const auto& facet = input_mesh.get_facets().row(f);
293 assert((facet.array() < num_input_vertices).all());
294 const auto& uv_facet = input_uv_indices.row(f);
295 assert((uv_facet.array() < num_input_uvs).all());
296 // Find vertices on this edge
297 auto edge_vertices = input_mesh.get_edge_vertices(e);
298 // Now find corresponding uvs on this edge
299 Index uv_index_0 = lagrange::invalid<Index>();
300 Index uv_index_1 = lagrange::invalid<Index>();
301 for (Index i = 0; i < 3; ++i) {
302 Index vtx_index = facet[i];
303 Index uv_index = uv_facet[i];
304 if (vtx_index == edge_vertices[0]) {
305 uv_index_0 = uv_index;
306 } else if (vtx_index == edge_vertices[1]) {
307 uv_index_1 = uv_index;
308 }
309 }
310 assert(uv_index_0 != lagrange::invalid<Index>());
311 assert(uv_index_1 != lagrange::invalid<Index>());
312
313 assert(boundary_vertices.find(edge_vertices[0]) != boundary_vertices.end());
314 assert(boundary_vertices.find(edge_vertices[1]) != boundary_vertices.end());
315 assert(f_uv + 1 < uv_facets.rows());
316
317 for (Index is = 0; is < num_segments; ++is) {
318 Index uvbstart_on_this_segment = uvbstart + (is - 1) * num_boundary_vertices;
319 Index uvbstart_on_next_segment = uvbstart + is * num_boundary_vertices;
320 bool is_first_segment = (is == 0);
321 bool is_last_segment = (is == num_segments - 1);
322
323 Index uv0 = is_first_segment ? uv_index_0
324 : uvbstart_on_this_segment +
325 boundary_vertices[edge_vertices[0]];
326 Index uv1 = is_first_segment ? uv_index_1
327 : uvbstart_on_this_segment +
328 boundary_vertices[edge_vertices[1]];
329 Index uv2 = is_last_segment ? uv_index_0 + num_input_vertices
330 : uvbstart_on_next_segment +
331 boundary_vertices[edge_vertices[0]];
332 Index uv3 = is_last_segment ? uv_index_1 + num_input_vertices
333 : uvbstart_on_next_segment +
334 boundary_vertices[edge_vertices[1]];
335
336 assert(uv0 < uv_values.rows());
337 assert(uv1 < uv_values.rows());
338 assert(uv2 < uv_values.rows());
339 assert(uv3 < uv_values.rows());
340
341 uv_facets.row(f_uv++) << uv0, uv2, uv1;
342 uv_facets.row(f_uv++) << uv1, uv2, uv3;
343 }
344 }
345 }
346
347 // Apply UVs
348 assert(uv_values.rows() == num_input_uvs * 2 + (num_segments - 1) * num_boundary_vertices);
349 offset_mesh->initialize_uv(std::move(uv_values), std::move(uv_facets));
350 }
351
352 // sanity check
353 assert(
354 offset_mesh->get_num_vertices() ==
355 num_input_vertices * 2 + num_boundary_vertices * (num_segments - 1));
356
357 return offset_mesh;
358}

380template <typename MeshType>

382 const MeshType& input_mesh,

384 ScalarOf<MeshType> offset_amount,
