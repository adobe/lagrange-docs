109 template <typename Derived>
110 std::shared_ptr<Derived> release_ptr()
111 {
112 ensure_unique_owner<Derived>();
113 auto ptr = static_cast<Derived*>(m_data.get());
114 la_debug_assert(dynamic_cast<Derived*>(m_data.get()));
115 auto ret = std::make_shared<Derived>(std::move(*ptr));
116 m_data.reset();
117 return ret;
118 }

120public:

127
133 ::lagrange::internal::weak_ptr<T> _get_weak_ptr() { return m_data; }
134
135protected:
136 ::lagrange::internal::shared_ptr<T> m_data;
137
139 template <typename Derived>
140 void ensure_unique_owner()
141 {
142 if (m_data.use_count() != 1) {
143 auto ptr = static_cast<const Derived*>(m_data.get());
144 la_debug_assert(dynamic_cast<const Derived*>(m_data.get()));
145 m_data = ::lagrange::internal::make_shared<Derived>(*ptr);
146 }
147 }
