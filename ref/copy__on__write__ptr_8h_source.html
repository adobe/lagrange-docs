29template <typename T>
30class copy_on_write_ptr
31{
32public:

39public:
41 copy_on_write_ptr(std::shared_ptr<T>&& ptr = nullptr)
42 : m_data(std::move(ptr))
43 {
44 if (m_data) {

47 }

56 template <typename Derived>
57 copy_on_write_ptr(copy_on_write_ptr<Derived>&& ptr)
58 : m_data{std::dynamic_pointer_cast<T>(std::move(ptr.m_data))}
59 {}

62 copy_on_write_ptr(const copy_on_write_ptr&) = default;

66
68 const T* read() const { return m_data.get(); }

71 template <typename Derived>
72 const Derived* dynamic_read() const
73 {
74 return dynamic_cast<const Derived*>(m_data.get());
75 }
76
78 template <typename Derived>
79 const Derived* static_read() const
80 {
81 return static_cast<const Derived*>(m_data.get());
82 }
83
85 template <typename Derived>
86 Derived* static_write()
87 {
88 ensure_unique_owner<Derived>();
89 return static_cast<Derived*>(m_data.get());
90 }
91
93 template <typename Derived>
94 Derived* dynamic_write()
95 {
96 ensure_unique_owner<Derived>();
97 return dynamic_cast<Derived*>(m_data.get());
98 }
99
101 template <typename Derived>
102 std::shared_ptr<Derived> release_ptr()
103 {
104 ensure_unique_owner<Derived>();
105 return std::dynamic_pointer_cast<Derived>(std::move(m_data));
106 }
107
108protected:
110 std::shared_ptr<T> m_data;
111
113 template <typename Derived>
114 void ensure_unique_owner()
115 {
116 if (m_data.use_count() != 1) {
117 auto ptr = dynamic_cast<const Derived*>(m_data.get());
118 la_debug_assert(ptr);
119 m_data = std::dynamic_pointer_cast<T>(std::make_shared<Derived>(*ptr));
120 }
121 }
122};
