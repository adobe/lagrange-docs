37template <typename MeshType>
38std::unique_ptr<MeshType> resolve_vertex_nonmanifoldness(MeshType& mesh)
39{
40 if (!mesh.is_connectivity_initialized()) {
41 mesh.initialize_connectivity();
42 }
43 if (mesh.get_vertex_per_facet() != 3) {
44 throw std::runtime_error(
45 "Resolve vertex nonmanifoldness is only implemented for triangle mesh");
46 }
47
48 using Index = typename MeshType::Index;
49 using Edge = typename MeshType::Edge;
50 using VertexArray = typename MeshType::VertexArray;
51 using FacetArray = typename MeshType::FacetArray;
52
53 const auto dim = mesh.get_dim();
54 const Index num_vertices = mesh.get_num_vertices();
55 const auto& vertices = mesh.get_vertices();
56 const auto& facets = mesh.get_facets();
57
58 auto get_opposite_edge = [&](Index fid, Index vid) {
59 const auto& f = facets.row(fid);
60 if (f[0] == vid) {
61 return Edge(f[1], f[2]);
62 } else if (f[1] == vid) {
63 return Edge(f[2], f[0]);
64 } else if (f[2] == vid) {
65 return Edge(f[0], f[1]);
66 } else {
67 throw Error(
68 "Facet " + std::to_string(fid) + " does not contain vertex " + std::to_string(vid));
69 }
70 };
71
72 FacetArray out_facets(facets);
73 Index vertex_count = mesh.get_num_vertices();
74
75 // this is a backward vertex map, we initialize with iota as {0, 1, 2, 3, ...}
76 // later we add elements gradually, since we don't know the final size in advance
77 // This sounds really bad, but in practice there are not many non-manifold vertices,
78 // and because of std::vector's memory allocation strategy, copying will typically
79 // never happen, or at most once.
80 std::vector<Index> backward_vertex_map(vertex_count);
81 std::iota(backward_vertex_map.begin(), backward_vertex_map.end(), 0);
82
83 for (Index i = 0; i < num_vertices; i++) {
84 const auto& adj_facets = mesh.get_facets_adjacent_to_vertex(i);
85 std::vector<Index> rim_edges;
86 rim_edges.reserve(adj_facets.size() * 2);
87 for (Index fid : adj_facets) {
88 const auto e = get_opposite_edge(fid, i);
89 rim_edges.push_back(e[0]);
90 rim_edges.push_back(e[1]);
91 }
92 auto [loops, chains] = chain_directed_edges<Index>({rim_edges.data(), rim_edges.size()});
93 if (loops.size() + chains.size() > 1) {
94 std::unordered_map<Index, Index> comp_map;
95 Index comp_count = 0;
96 for (const auto& loop : loops) {
97 for (const auto vid : loop) {
98 comp_map[vid] = comp_count;
99 }
100 comp_count++;
101 }
102 for (const auto& chain : chains) {
103 for (const auto vid : chain) {
104 comp_map[vid] = comp_count;
105 }
106 comp_count++;
107 }
108
109 // Assign a new vertex for each additional rim loops.
110 for (Index fid : adj_facets) {
111 auto& f = facets.row(fid);
112 for (Index j = 0; j < 3; j++) {
113 if (f[j] == i) {
114 const Index next = (j + 1) % 3;
115 const Index prev = (j + 2) % 3;
116 assert(comp_map.find(f[next]) != comp_map.end());
117 assert(comp_map.find(f[prev]) != comp_map.end());
118 if (comp_map[f[next]] != comp_map[f[prev]]) {
119 throw std::runtime_error(
120 "Complex edge loop detected. Vertex " + std::to_string(i) +
121 "'s one ring neighborhood must contain nonmanifold_edges!");
122 }
123 assert(comp_map[f[next]] == comp_map[f[prev]]);
124 const Index comp_id = comp_map[f[next]];
125 if (comp_id > 0) {
126 const Index new_vertex_index = vertex_count + comp_id - 1;
127 out_facets(fid, j) = new_vertex_index;
128 if (safe_cast<Index>(backward_vertex_map.size()) <= new_vertex_index) {
129 backward_vertex_map.resize(new_vertex_index + 1, invalid<Index>());
130 }
131 backward_vertex_map[new_vertex_index] = i;
132 }
133 break;
134 }
135 }
136 }
137 vertex_count += safe_cast<Index>(loops.size() + chains.size()) - 1;
138 }
139 }
140 assert(out_facets.rows() == 0 || out_facets.maxCoeff() == vertex_count - 1);
141
142 // all vertices between 0 and num_vertices are the same, so we can block-copy
143 VertexArray out_vertices(vertex_count, dim);
144 out_vertices.block(0, 0, num_vertices, dim) = vertices;
145 for (Index i = num_vertices; i < vertex_count; i++) {
146 out_vertices.row(i) = vertices.row(backward_vertex_map[i]);
147 }
148
149 auto out_mesh = create_mesh(std::move(out_vertices), std::move(out_facets));
150
151 map_attributes(mesh, *out_mesh, backward_vertex_map);
152
153 return out_mesh;
154}
