44template <typename MeshType>
45std::unique_ptr<MeshType> resolve_vertex_nonmanifoldness(MeshType& mesh)
46{
47 if (!mesh.is_connectivity_initialized()) {
48 mesh.initialize_connectivity();
49 }
50 if (mesh.get_vertex_per_facet() != 3) {
51 throw std::runtime_error(
52 "Resolve vertex nonmanifoldness is only implemented for triangle mesh");
53 }
54
55 using Index = typename MeshType::Index;
56 using Edge = typename MeshType::Edge;
57 using VertexArray = typename MeshType::VertexArray;
58 using FacetArray = typename MeshType::FacetArray;
59
60 const auto dim = mesh.get_dim();
61 const Index num_vertices = mesh.get_num_vertices();
62 const auto& vertices = mesh.get_vertices();
63 const auto& facets = mesh.get_facets();
64
65 FacetArray out_facets(facets);
66 Index vertex_count = mesh.get_num_vertices();
67
68 // this is a backward vertex map, we initialize with iota as {0, 1, 2, 3, ...}
69 // later we add elements gradually, since we don't know the final size in advance
70 // This sounds really bad, but in practice there are not many non-manifold vertices,
71 // and because of std::vector's memory allocation strategy, copying will typically
72 // never happen, or at most once.
73 std::vector<Index> backward_vertex_map(vertex_count);
74 std::iota(backward_vertex_map.begin(), backward_vertex_map.end(), 0);
75
76 for (Index i = 0; i < num_vertices; i++) {
77 const auto& adj_facets = mesh.get_facets_adjacent_to_vertex(i);
78 std::list<Edge> rim_edges;
79 for (Index fid : adj_facets) {
80 rim_edges.push_back(get_opposite_edge(facets, fid, i));
81 }
82 auto chains = chain_edges<Index>(rim_edges);
83 if (chains.size() > 1) {
84 std::unordered_map<Index, Index> comp_map;
85 Index comp_count = 0;
86 for (const auto& chain : chains) {
87 for (const auto vid : chain) {
88 comp_map[vid] = comp_count;
89 }
90 comp_count++;
91 }
92
93 // Assign a new vertex for each additional rim loops.
94 for (Index fid : adj_facets) {
95 auto& f = facets.row(fid);
96 for (Index j = 0; j < 3; j++) {
97 if (f[j] == i) {
98 const Index next = (j + 1) % 3;
99 const Index prev = (j + 2) % 3;
100 assert(comp_map.find(f[next]) != comp_map.end());
101 assert(comp_map.find(f[prev]) != comp_map.end());
102 if (comp_map[f[next]] != comp_map[f[prev]]) {
103 logger().trace("vertex: {}", i);
104 for (auto fi : adj_facets) {
105 logger().trace("{}", facets.row(fi));
106 }
107 throw std::runtime_error(
108 "Complex edge loop detected. Vertex " + std::to_string(i) +
109 "'s one ring neighborhood must contain nonmanifold_edges!");
110 }
111 assert(comp_map[f[next]] == comp_map[f[prev]]);
112 const Index comp_id = comp_map[f[next]];
113 if (comp_id > 0) {
114 const Index new_vertex_index = vertex_count + comp_id - 1;
115 out_facets(fid, j) = new_vertex_index;
116 if (safe_cast<Index>(backward_vertex_map.size()) <= new_vertex_index) {
117 backward_vertex_map.resize(new_vertex_index + 1, invalid<Index>());
118 }
119 backward_vertex_map[new_vertex_index] = i;
120 }
121 break;
122 }
123 }
124 }
125 vertex_count += safe_cast<Index>(chains.size()) - 1;
126 }
127 }
128 assert(out_facets.rows() == 0 || out_facets.maxCoeff() == vertex_count - 1);
129
130 // all vertices between 0 and num_vertices are the same, so we can block-copy
131 VertexArray out_vertices(vertex_count, dim);
132 out_vertices.block(0, 0, num_vertices, dim) = vertices;
133 for (Index i = num_vertices; i < vertex_count; i++) {
134 out_vertices.row(i) = vertices.row(backward_vertex_map[i]);
135 }
136
137 auto out_mesh = create_mesh(std::move(out_vertices), std::move(out_facets));
138
139 map_attributes(mesh, *out_mesh, backward_vertex_map);
140
141 return out_mesh;
142}
