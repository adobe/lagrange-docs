34template <typename MeshType>
35std::unique_ptr<MeshType> unify_index_buffer(
36 const MeshType& mesh,
37 const std::vector<std::string>& indexed_attribute_names)
38{
39 using VertexArray = typename MeshType::VertexArray;
40 using AttributeArray = typename MeshType::AttributeArray;
41 using Index = typename MeshType::Index;
42 using IndexArray = typename MeshType::IndexArray;
43
44 const Index num_attrs = safe_cast<Index>(indexed_attribute_names.size());
45 std::vector<std::tuple<const AttributeArray&, const IndexArray&>> attrs;
46 attrs.reserve(num_attrs);
47 for (const auto& attr_name : indexed_attribute_names) {
48 attrs.emplace_back(mesh.get_indexed_attribute(attr_name));
49 }
50
51 const Index num_vertices = mesh.get_num_vertices();
52 const Index num_facets = mesh.get_num_facets();
53 const Index vertex_per_facet = mesh.get_vertex_per_facet();
54 const auto& vertices = mesh.get_vertices();
55 const auto& facets = mesh.get_facets();
56 const Index num_corners = num_facets * vertex_per_facet;
57 std::vector<Index> corner_indices(num_corners);
58 std::iota(corner_indices.begin(), corner_indices.end(), 0);
59
60 // Lexicographical comparison of corner indices.
61 auto corner_index_comp = [&](Index i, Index j) -> bool {
62 const Index fi = i / vertex_per_facet;
63 const Index ci = i % vertex_per_facet;
64 const Index fj = j / vertex_per_facet;
65 const Index cj = j % vertex_per_facet;
66
67 if (facets(fi, ci) == facets(fj, cj)) {
68 for (auto k : range(num_attrs)) {
69 const auto& indices = std::get<1>(attrs[k]);
70 if (indices(fi, ci) != indices(fj, cj)) {
71 return indices(fi, ci) < indices(fj, cj);
72 }
73 }
74 } else {
75 return facets(fi, ci) < facets(fj, cj);
76 }
77 return false;
78 };
79
80 auto corner_index_eq = [&](Index i, Index j) -> bool {
81 const Index fi = i / vertex_per_facet;
82 const Index ci = i % vertex_per_facet;
83 const Index fj = j / vertex_per_facet;
84 const Index cj = j % vertex_per_facet;
85
86 if (facets(fi, ci) == facets(fj, cj)) {
87 for (auto k : range(num_attrs)) {
88 const auto& indices = std::get<1>(attrs[k]);
89 if (indices(fi, ci) != indices(fj, cj)) {
90 return false;
91 }
92 }
93 } else {
94 return false;
95 }
96 return true;
97 };
98
99 tbb::parallel_sort(corner_indices.begin(), corner_indices.end(), corner_index_comp);
100
101 std::vector<bool> visited(num_vertices, false);
102 std::vector<Index> new_vertices;
103 new_vertices.reserve(num_vertices);
104 auto unified_facets = facets;
105
106 for (Index i = 0; i < num_corners;) {
107 const Index fi = corner_indices[i] / vertex_per_facet;
108 const Index ci = corner_indices[i] % vertex_per_facet;
109
110 Index vi = facets(fi, ci);
111 if (!visited[vi]) {
112 visited[vi] = true;
113 } else {
114 new_vertices.push_back(vi);
115 vi = (Index)(new_vertices.size() + num_vertices - 1);
116 }
117
118 Index j = i;
119 while (j < num_corners && corner_index_eq(corner_indices[i], corner_indices[j])) {
120 const Index fj = corner_indices[j] / vertex_per_facet;
121 const Index cj = corner_indices[j] % vertex_per_facet;
122 unified_facets(fj, cj) = vi;
123 j++;
124 }
125 i = j;
126 }
127
128 const Index num_added_vertices = safe_cast<Index>(new_vertices.size());
129 VertexArray unified_vertices(num_vertices + num_added_vertices, mesh.get_dim());
130 unified_vertices.topRows(num_vertices) = vertices;
131 tbb::parallel_for(Index(0), num_added_vertices, [&](Index i) {
132 unified_vertices.row(i + num_vertices) = vertices.row(new_vertices[i]);
133 });
134
135 auto unified_mesh = create_mesh(std::move(unified_vertices), std::move(unified_facets));
136 const auto& new_facets = unified_mesh->get_facets();
137
138 for (auto i : range(num_attrs)) {
139 const auto& attr_name = indexed_attribute_names[i];
140 unified_mesh->add_vertex_attribute(attr_name);
141 const auto& values = std::get<0>(attrs[i]);
142 const auto& indices = std::get<1>(attrs[i]);
143
144 AttributeArray unified_attr(num_vertices + num_added_vertices, values.cols());
145 unified_attr.setZero();
146 for (Index fi = 0; fi < num_facets; ++fi) {
147 for (auto ci : range(vertex_per_facet)) {
148 unified_attr.row(new_facets(fi, ci)) = values.row(indices(fi, ci));
149 }
150 }
151 unified_mesh->import_vertex_attribute(attr_name, unified_attr);
152 }
153
154 // Map vertex, facet and corner attribute over.
155 {
156 std::vector<Index> vertex_map(num_vertices + num_added_vertices);
157 tbb::parallel_for(Index(0), num_vertices, [&](Index i) { vertex_map[i] = i; });
158 tbb::parallel_for(Index(0), num_added_vertices, [&](Index i) {
159 vertex_map[num_vertices + i] = new_vertices[i];
160 });
161 map_vertex_attributes(mesh, *unified_mesh, vertex_map);
162 }
163 map_facet_attributes(mesh, *unified_mesh);
164 map_corner_attributes(mesh, *unified_mesh);
165
166 return unified_mesh;
167}
