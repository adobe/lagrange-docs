49template <typename TargetType, typename SourceType>
50constexpr auto safe_cast(SourceType value)
51 -> std::enable_if_t<!std::is_same<SourceType, TargetType>::value, TargetType>
52{
53 TargetType value_2 = static_cast<TargetType>(value);
54 SourceType value_3 = static_cast<SourceType>(value_2);
55
56 if ((value_2 >= 0) != (value >= 0)) {
57 // Sign changed. Not good.
58 logger().error("Casting failed: from {} to {} causes a sign change...", value, value_2);
59 throw BadCastError();
60 } else if (value_3 == value) {
61 // Lossless cast. :D
62 return value_2;
63 } else {
64 // Lossy cast... Check for casting error.
65 constexpr SourceType EPS =
66 static_cast<SourceType>(std::numeric_limits<TargetType>::epsilon());
67
68 // Generates warning C4146: "unary minus operator applied to unsigned type, result still
69 // unsigned" this cannot happen, as we check for unsigned types above.
70 LA_DISABLE_WARNING_BEGIN
71 LA_DISABLE_WARNING_MSVC(4146)
72 const SourceType value_abs = value_3 > 0 ? value_3 : -value_3;
73 LA_DISABLE_WARNING_END
74
75 const SourceType scaled_eps = value_abs >= SourceType(1) ? EPS * value_abs : EPS;
76 if (value_3 > value && value_3 < value + scaled_eps) {
77 return value_2;
78 } else if (value_3 < value && value_3 + scaled_eps > value) {
79 return value_2;
80 } else {
81 // Cast is likely not valid...
82 logger().error(
83 "Casting failed: from {} to {} will incur error ({}) larger than {}",
84 value,
85 value_2,
86 value - static_cast<SourceType>(value_2),
87 scaled_eps);
88 throw BadCastError();
89 }
90 }
91}

135template <typename T, typename U>
136constexpr T safe_cast_enum(const U u)
137{
138 static_assert(
139 std::is_enum<T>::value || std::is_enum<U>::value,
140 "At least one of the types should be an enum");
141 static_assert(
142 (std::is_enum<T>::value && std::is_enum<U>::value) == std::is_same<T, U>::value,
143 "Casting one enum to another is prohibitied");
144
145 const auto tu = static_cast<T>(static_cast<std::int64_t>(u));
146 const auto utu = static_cast<U>(static_cast<std::int64_t>(tu));
147 const auto tutu = static_cast<T>(static_cast<std::int64_t>(utu));
148 if (u == utu && tutu == tu) {
149 return tu;
150 } else {
151 throw BadCastError();
152 return T();
153 }
154}
