21template <typename MeshType>
22auto compute_triangle_normal_const(const MeshType& mesh) -> typename MeshType::AttributeArray
23{
24 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
25 using Index = IndexOf<MeshType>;
26 using Scalar = ScalarOf<MeshType>;
27 using AttributeArray = typename MeshType::AttributeArray;
28 using RowVector3s = Eigen::Matrix<Scalar, 1, 3>;
29
30 if (mesh.get_vertex_per_facet() != 3) {
31 throw std::runtime_error("Input mesh is not triangle mesh.");
32 }
33 if (mesh.get_dim() != 3) {
34 throw std::runtime_error("Input mesh vertices should have dimension 3.");
35 }
36
37 const auto& vertices = mesh.get_vertices();
38 const auto& facets = mesh.get_facets();
39 AttributeArray normals(facets.rows(), 3);
40
41 // Iterate over facets in parallel
42 tbb::parallel_for(Index(0), mesh.get_num_facets(), [&](Index f) {
43 const RowVector3s p0 = vertices.row(facets(f, 0));
44 const RowVector3s p1 = vertices.row(facets(f, 1));
45 const RowVector3s p2 = vertices.row(facets(f, 2));
46 normals.row(f) = (p1 - p0).cross(p2 - p0).stableNormalized();
47 });
48
49 return normals;
50}
51
52template <typename MeshType>
53void compute_triangle_normal(MeshType& mesh)
54{
55 mesh.add_facet_attribute("normal");
56 mesh.import_facet_attribute("normal", compute_triangle_normal_const(mesh));
57}
