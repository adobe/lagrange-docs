24template <typename MeshType>
25std::vector<std::unique_ptr<MeshType>> extract_component_submeshes(
26 MeshType& mesh,
27 std::vector<std::vector<typename MeshType::Index>>* vertex_mappings = nullptr,
28 std::vector<std::vector<typename MeshType::Index>>* facet_mappings = nullptr)
29{
30 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
31
32 if (!mesh.is_components_initialized()) {
33 mesh.initialize_components();
34 }
35
36 return extract_submeshes(mesh, mesh.get_components(), vertex_mappings, facet_mappings);
37}

39template <typename MeshType, typename Index>
40std::vector<std::unique_ptr<MeshType>> extract_submeshes(
41 const MeshType& mesh,
42 const std::vector<std::vector<Index>>& facet_groups,
43 std::vector<std::vector<Index>>* vertex_mappings = nullptr,
44 std::vector<std::vector<Index>>* facet_mappings = nullptr)
45{
46 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
47
48 if (vertex_mappings) {
49 vertex_mappings->resize(facet_groups.size());
50 }
51 if (facet_mappings) {
52 facet_mappings->resize(facet_groups.size());
53 }
54
55 std::vector<std::unique_ptr<MeshType>> extracted_meshes;
56 for (auto i : range(facet_groups.size())) {
57 extracted_meshes.emplace_back(extract_submesh(
58 mesh,
59 facet_groups[i],
60 vertex_mappings ? &((*vertex_mappings)[i]) : nullptr,
61 facet_mappings ? &((*facet_mappings)[i]) : nullptr));
62 }
63 return extracted_meshes;
64}

66template <typename MeshType, typename Index>
67std::unique_ptr<MeshType> extract_submesh(
68 const MeshType& mesh,
69 const std::vector<Index>& selected_facets,
70 std::vector<Index>* vertex_mapping = nullptr,
71 std::vector<Index>* facet_mapping = nullptr)
72{
73 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
74
75 using MeshIndex = typename MeshType::Index;
76 using VertexArray = typename MeshType::VertexArray;
77 using FacetArray = typename MeshType::FacetArray;
78 const MeshIndex num_selected_facets = safe_cast<MeshIndex>(selected_facets.size());
79 const auto vertex_per_facet = mesh.get_vertex_per_facet();
80
81 const auto& vertices = mesh.get_vertices();
82 const auto& facets = mesh.get_facets();
83 const auto num_facets = mesh.get_num_facets();
84 const MeshIndex min_num_vertices = std::min(
85 safe_cast<MeshIndex>(num_selected_facets * vertex_per_facet),
86 mesh.get_num_vertices());
87
88 FacetArray sub_facets(num_selected_facets, vertex_per_facet);
89 std::unordered_map<MeshIndex, MeshIndex> sub_vertex_indices;
90 sub_vertex_indices.reserve(min_num_vertices);
91
92 if (facet_mapping) {
93 facet_mapping->resize(num_selected_facets);
94 }
95
96 MeshIndex count = 0;
97 for (auto i : range(num_selected_facets)) {
98 la_runtime_assert(
99 selected_facets[i] >= 0 && safe_cast<MeshIndex>(selected_facets[i]) < num_facets,
100 "Facet index out of bound for submesh extraction");
101 for (auto j : range(vertex_per_facet)) {
102 const MeshIndex idx = facets(selected_facets[i], j);
103 if (sub_vertex_indices.find(idx) == sub_vertex_indices.end()) {
104 sub_vertex_indices[idx] = count;
105 count++;
106 }
107 sub_facets(i, j) = sub_vertex_indices[idx];
108 }
109 if (facet_mapping) {
110 (*facet_mapping)[i] = selected_facets[i];
111 }
112 }
113
114 VertexArray sub_vertices(count, mesh.get_dim());
115 if (vertex_mapping) {
116 vertex_mapping->resize(count);
117 }
118 for (const auto& item : sub_vertex_indices) {
119 // item.second = index in submesh
120 // item.first = index in original mesh
121 sub_vertices.row(item.second) = vertices.row(item.first);
122 if (vertex_mapping) {
123 (*vertex_mapping)[item.second] = item.first;
124 }
125 }
126
127 return lagrange::create_mesh(std::move(sub_vertices), std::move(sub_facets));
128}
