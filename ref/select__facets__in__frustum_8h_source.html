43template <typename MeshType, typename Point3D>
44bool select_facets_in_frustum(
45 MeshType& mesh,
46 const Eigen::PlainObjectBase<Point3D>& n0,
47 const Eigen::PlainObjectBase<Point3D>& p0,
48 const Eigen::PlainObjectBase<Point3D>& n1,
49 const Eigen::PlainObjectBase<Point3D>& p1,
50 const Eigen::PlainObjectBase<Point3D>& n2,
51 const Eigen::PlainObjectBase<Point3D>& p2,
52 const Eigen::PlainObjectBase<Point3D>& n3,
53 const Eigen::PlainObjectBase<Point3D>& p3,

65 // Per-thread buffers to store intermediate results.
66 struct LocalBuffers
67 {
68 Point3D v0, v1, v2; // Triangle vertices.
69 Point3D q0, q1, q2, q3; // Intermediate tet vertices.
70 std::array<double, 2> r0, r1, r2; // Triangle projections.

74 auto edge_overlap_with_negative_octant = [](const Point3D& q0, const Point3D& q1) -> bool {
75 Scalar t_min = 0, t_max = 1;
76 const Point3D e = q0 - q1;
77 if (e[0] > 0) {
78 t_max = std::min(t_max, -q1[0] / e[0]);
79 } else if (e[0] < 0) {
80 t_min = std::max(t_min, -q1[0] / e[0]);
81 } else {
82 if (q1[0] >= 0) return false;
83 }
84
85 if (e[1] > 0) {
86 t_max = std::min(t_max, -q1[1] / e[1]);
87 } else if (e[1] < 0) {
88 t_min = std::max(t_min, -q1[1] / e[1]);
89 } else {
90 if (q1[1] >= 0) return false;
91 }
92
93 if (e[2] > 0) {
94 t_max = std::min(t_max, -q1[2] / e[2]);
95 } else if (e[2] < 0) {
96 t_min = std::max(t_min, -q1[2] / e[2]);
97 } else {
98 if (q1[2] >= 0) return false;
99 }
100
101 return t_max >= t_min;
102 };

104 auto compute_plane =
105 [](const Point3D& q0, const Point3D& q1, const Point3D& q2) -> std::pair<Point3D, Scalar> {
106 const Point3D n = {
107 q0[2] * (q2[1] - q1[1]) + q1[2] * (q0[1] - q2[1]) + q2[2] * (q1[1] - q0[1]),
108 q0[0] * (q2[2] - q1[2]) + q1[0] * (q0[2] - q2[2]) + q2[0] * (q1[2] - q0[2]),
109 q0[1] * (q2[0] - q1[0]) + q1[1] * (q0[0] - q2[0]) + q2[1] * (q1[0] - q0[0])};
110 const Scalar c = (n.dot(q0) + n.dot(q1) + n.dot(q2)) / 3;
111 return {n, c};
112 };

114 // Compute the orientation of 2D traingle (v0, v1, O), where O = (0, 0).
115 auto orient2D_inexact = [](const std::array<double, 2>& v0,
116 const std::array<double, 2>& v1) -> int {
117 const auto r = v0[0] * v1[1] - v0[1] * v1[0];
118 if (r > 0) return 1;
119 if (r < 0) return -1;
120 return 0;
121 };

123 auto triangle_intersects_negative_axis = [&](const Point3D& q0,
124 const Point3D& q1,
125 const Point3D& q2,
126 const Point3D& n,
127 const Scalar c,
128 const int axis) -> bool {
129 auto& r0 = temp_vars.local().r0;
130 auto& r1 = temp_vars.local().r1;
131 auto& r2 = temp_vars.local().r2;
132
133 r0[0] = q0[(axis + 1) % 3];
134 r0[1] = q0[(axis + 2) % 3];
135 r1[0] = q1[(axis + 1) % 3];
136 r1[1] = q1[(axis + 2) % 3];
137 r2[0] = q2[(axis + 1) % 3];
138 r2[1] = q2[(axis + 2) % 3];
139
140 auto o01 = orient2D_inexact(r0, r1);
141 auto o12 = orient2D_inexact(r1, r2);
142 auto o20 = orient2D_inexact(r2, r0);
143 if (o01 == o12 && o01 == o20) {
144 if (o01 == 0) {
145 // Triangle projection is degenerate.
146 // Note that the case where axis is coplanar with the triangle
147 // is treated as no intersection (which is debatale).
148 return false;
149 } else {
150 // Triangle projection contains the origin.
151 // Check axis intercept.
152 return (c < 0 && n[axis] > 0) || (c > 0 && n[axis] < 0);
153 }
154 } else {
155 // Note that we treat the case where the axis intersect triangle at
156 // its boundary as no intersection.
157 return false;
158 }
159 };
160
161 auto triangle_intersects_negative_axes =
162 [&](const Point3D& q0, const Point3D& q1, const Point3D& q2) -> bool {
163 const auto r = compute_plane(q0, q1, q2);
164 if (triangle_intersects_negative_axis(q0, q1, q2, r.first, r.second, 0)) return true;
165 if (triangle_intersects_negative_axis(q0, q1, q2, r.first, r.second, 1)) return true;
166 if (triangle_intersects_negative_axis(q0, q1, q2, r.first, r.second, 2)) return true;
167 return false;
168 };

170 auto tet_overlap_with_negative_octant =
171 [&](const Point3D& q0, const Point3D& q1, const Point3D& q2, const Point3D& q3) -> bool {
172 // Check 1: Check if any tet vertices is in negative octant.
173 if ((q0.array() < 0).all()) return true;
174 if ((q1.array() < 0).all()) return true;
175 if ((q2.array() < 0).all()) return true;
176 if ((q3.array() < 0).all()) return true;
177
178 // Check 2: Check if any tet edges cross the negative octant.
179 if (edge_overlap_with_negative_octant(q0, q1)) return true;
180 if (edge_overlap_with_negative_octant(q0, q2)) return true;
181 if (edge_overlap_with_negative_octant(q0, q3)) return true;
182 if (edge_overlap_with_negative_octant(q1, q2)) return true;
183 if (edge_overlap_with_negative_octant(q1, q3)) return true;
184 if (edge_overlap_with_negative_octant(q2, q3)) return true;
185
186 // Check 3: Check if -X, -Y or -Z axis intersect the tet.
187 if (triangle_intersects_negative_axes(q0, q1, q2)) return true;
188 if (triangle_intersects_negative_axes(q1, q2, q3)) return true;
189 if (triangle_intersects_negative_axes(q2, q3, q0)) return true;
190 if (triangle_intersects_negative_axes(q3, q0, q1)) return true;
191
192 // All check failed iff tet does not intersect the negative octant.
193 return false;
194 };

196 AttributeArray attr;
197 if (!greedy) {
198 attr.resize(num_facets, 1);
199 attr.setZero();
200 }
201 std::atomic_bool r{false};
202
203 tbb::parallel_for(
204 tbb::blocked_range<Index>(0, num_facets),
205 [&](const tbb::blocked_range<Index>& tbb_range) {
206 for (auto fi = tbb_range.begin(); fi != tbb_range.end(); fi++) {
207 if (tbb_utils::is_cancelled()) break;
208
209 // Triangle (v0, v1, v2) intersect the cone defined by 4 planes
210 // iff the tetrahedron (q0, q1, q2, q3) does not intersect the negative octant.
211 // This can be proved by Farcus' lemma.
212
213 auto& v0 = temp_vars.local().v0;
214 auto& v1 = temp_vars.local().v1;
215 auto& v2 = temp_vars.local().v2;
216
217 auto& q0 = temp_vars.local().q0;
218 auto& q1 = temp_vars.local().q1;
219 auto& q2 = temp_vars.local().q2;
220 auto& q3 = temp_vars.local().q3;
221
222 v0 = vertices.row(facets(fi, 0));
223 v1 = vertices.row(facets(fi, 1));
224 v2 = vertices.row(facets(fi, 2));
225
226 q0 = {(v0 - p0).dot(n0), (v1 - p0).dot(n0), (v2 - p0).dot(n0)};
227 q1 = {(v0 - p1).dot(n1), (v1 - p1).dot(n1), (v2 - p1).dot(n1)};
228 q2 = {(v0 - p2).dot(n2), (v1 - p2).dot(n2), (v2 - p2).dot(n2)};
229 q3 = {(v0 - p3).dot(n3), (v1 - p3).dot(n3), (v2 - p3).dot(n3)};
230
231 const auto ri = !tet_overlap_with_negative_octant(q0, q1, q2, q3);
232
233 if (ri) r = true;
234
235 if (!greedy) {
236 attr(fi, 0) = ri;
237 }
238
239 if (greedy && ri) {
240 tbb_utils::cancel_group_execution();
241 break;
242 }
243 }
244 });
245
246 if (!greedy) {
247 mesh.add_facet_attribute("is_selected");
248 mesh.import_facet_attribute("is_selected", attr);
249 }
250 return r.load();
251}
