28//
29// The parameters for the function
30//
31template <typename MeshType>
32struct SelectFacetsByNormalSimilarityParameters
33{
34 using Scalar = typename MeshType::Scalar;
35 using Index = typename MeshType::Index;
36
37 //
38 // This are input and have to be set
39 //
40 // Increasing this would select a larger region
41 Scalar flood_error_limit = invalid<Scalar>();
42 // This tries to smooth the selection boundary (reduce ears)
43 bool should_smooth_boundary = true;
44
45 //
46 // These parameters have default values
47 // Default value comes from segmentation/Constants.h
48 //
49 // If this is not empty, then only faces are considered which are marked as true
50 std::vector<bool> is_facet_selectable = std::vector<bool>();
51 // Internal param...
52 Scalar flood_second_to_first_order_limit_ratio = 1.0 / 6.0;
53 // Number of iterations for smoothing the boundary
54 Index num_smooth_iterations = 3;
55 // Use DFS or BFS search
56 enum class SearchType { BFS = 0, DFS = 1 };
57 SearchType search_type = SearchType::DFS;
58
59 // Set selectable facets using a vector of indices
60 void set_selectable_facets(MeshType& mesh_ref, const std::vector<Index>& selectable_facets)
61 {
62 if (selectable_facets.empty()) {
63 is_facet_selectable.clear();
64 } else {
65 is_facet_selectable.resize(mesh_ref.get_num_facets(), false);
66 for (auto facet_id : range_facets(mesh_ref, selectable_facets)) {
67 is_facet_selectable[facet_id] = true;
68 }
69 }
70 }
71
72}; // SelectFaceByNormalSimilarityParameters

74//
75// Perform the selection
76// mesh, input: the mesh
77// seed_facet_id, input: the index of the seed
78// parameters, input: the parameters above
79// returns a vector that contains true for facets that are selected
80//
81template <typename MeshType>
82std::vector<bool> select_facets_by_normal_similarity(
83 MeshType& mesh,
84 const typename MeshType::Index seed_facet_id,
85 const SelectFacetsByNormalSimilarityParameters<MeshType>& parameters)
86{
87 // ===================================================
88 // Helper typedef and lambdas
89 // ===================================================
90
91 using VertexType = typename MeshType::VertexType;
92 using Index = typename MeshType::Index;
93 using Scalar = typename MeshType::Scalar;
94 using IndexList = typename MeshType::IndexList;
95 using AttributeArray = typename MeshType::AttributeArray;
96 using Parameters = SelectFacetsByNormalSimilarityParameters<MeshType>;
97
98 // Check if a face is selectable
99 auto is_facet_selectable = [&mesh, &parameters](const Index facet_id) -> bool {
100 if (parameters.is_facet_selectable.empty()) {
101 return true;
102 } else {
103 la_runtime_assert(facet_id < mesh.get_num_facets());
104 return parameters.is_facet_selectable[facet_id];
105 }
106 };
107
108 // The energy between two normals
109 auto normal_direction_error = [](const VertexType& n1, const VertexType& n2) -> Scalar {
110 return static_cast<Scalar>(1.0) - std::abs(n1.dot(n2));
111 };
112
113 // ===================================================
114 // Let's begin
115 // current code just copied from
116 // Segmentation/SuperTriangulation/SingleFloodFromSeed()
117 // ===================================================
118
119 if (mesh.get_vertex_per_facet() != 3) {
120 throw std::runtime_error("Input mesh must be a triangle mesh.");
121 }
122
123 // We need this things
124 if (!mesh.is_connectivity_initialized()) {
125 mesh.initialize_connectivity();
126 }
127 if (!mesh.has_facet_attribute("normal")) {
128 compute_triangle_normal(mesh);
129 }
130
131 // This would be the final value that will be returned
132 std::vector<bool> is_facet_selected(mesh.get_num_facets(), false);
133
134 // Normals (and the seed normal)
135 const AttributeArray& facet_normals = mesh.get_facet_attribute("normal");
136 const VertexType seed_n = facet_normals.row(seed_facet_id);
137
138 // DFS search utility
139 std::vector<bool> is_facet_processed(mesh.get_num_facets(), false);
140 std::deque<Index> search_queue;
141
142 //
143 // (0): Add the seed neighbors to the queue to initialize the queue
144 //
145 is_facet_processed[seed_facet_id] = true;
146 is_facet_selected[seed_facet_id] = true;
147 {
148 const IndexList& facets_adjacent_to_seed = mesh.get_facets_adjacent_to_facet(seed_facet_id);
149 for (Index j = 0; j < static_cast<Index>(facets_adjacent_to_seed.size()); j++) {
150 const Index ne_fid = facets_adjacent_to_seed[j];
151 const VertexType ne_n = facet_normals.row(ne_fid);
152
153 if ((!is_facet_processed[ne_fid]) && is_facet_selectable(ne_fid)) {
154 Scalar error = normal_direction_error(seed_n, ne_n);
155 if (error < parameters.flood_error_limit) {
156 is_facet_selected[ne_fid] = true;
157 search_queue.push_back(ne_fid);
158 }
159 }
160 } // End of for over neighbours of seed
161 } // end of step 0
162
163 //
164 // (1): Run through all neighbors, process them, and push them into a queue
165 // for further flood
166 //
167 while (!search_queue.empty()) {
168 Index fid = invalid<Index>();
169 if (parameters.search_type == Parameters::SearchType::DFS) {
170 fid = search_queue.front();
171 search_queue.pop_front();
172 } else {
173 fid = search_queue.back();
174 search_queue.pop_back();
175 }
176
177 const VertexType center_n = facet_normals.row(fid);
178
179
180 const IndexList& facets_adjacent_to_candidate = mesh.get_facets_adjacent_to_facet(fid);
181 for (Index j = 0; j < static_cast<Index>(facets_adjacent_to_candidate.size()); j++) {
182 const Index ne_fid = facets_adjacent_to_candidate[j];
183 const VertexType ne_n = facet_normals.row(ne_fid);
184
185 if ((!is_facet_processed[ne_fid]) && is_facet_selectable(ne_fid)) {
186 const Scalar error_1 = normal_direction_error(seed_n, ne_n);
187 const Scalar error_2 = normal_direction_error(center_n, ne_n);
188 is_facet_processed[ne_fid] = true;
189
190 if ((error_1 < parameters.flood_error_limit) &&
191 (error_2 < parameters.flood_error_limit)) {
192 is_facet_selected[ne_fid] = true;
193 search_queue.push_back(ne_fid);
194 } else if (
195 error_2 < parameters.flood_error_limit *
196 parameters.flood_second_to_first_order_limit_ratio) {
197 // Second order approximation
198 is_facet_selected[ne_fid] = true;
199 search_queue.push_back(ne_fid);
200 }
201 } // end of is neighbour selectable
202 } // end of loops over neighbours
203 } // end of dfs stack
204
205 //
206 // Smooth the selection
207 //
208 if (parameters.should_smooth_boundary) {
209 for (Index st_iter_2 = 0; st_iter_2 < parameters.num_smooth_iterations; st_iter_2++) {
210 // Go through boundary faces - check spikes
211 for (Index i = 0; i < mesh.get_num_facets(); i++) {
212 // std::cout << facet_one_ring_faces[i].size() << std::endl;
213
214 const IndexList& facets_adjacent_to_candidate =
215 mesh.get_facets_adjacent_to_facet(i);
216
217 if (is_facet_selectable(i) && (facets_adjacent_to_candidate.size() > 2)) {
218 const bool select_flag = is_facet_selected[i];
219 Index neighbor_diff_flag = 0;
220
221 const Index ne_fid_0 = facets_adjacent_to_candidate[0];
222 const bool ne_select_0 = is_facet_selected[ne_fid_0];
223 if (select_flag != ne_select_0) neighbor_diff_flag++;
224
225 const Index ne_fid_1 = facets_adjacent_to_candidate[1];
226 const bool ne_select_1 = is_facet_selected[ne_fid_1];
227 if (select_flag != ne_select_1) neighbor_diff_flag++;
228
229 const Index ne_fid_2 = facets_adjacent_to_candidate[2];
230 const bool ne_select_2 = is_facet_selected[ne_fid_2];
231 if (select_flag != ne_select_2) neighbor_diff_flag++;
232
233 if (neighbor_diff_flag > 1) {
234 const VertexType self_n = facet_normals.row(i);
235 const VertexType ne_n_0 = facet_normals.row(ne_fid_0);
236 const VertexType ne_n_1 = facet_normals.row(ne_fid_1);
237 const VertexType ne_n_2 = facet_normals.row(ne_fid_2);
238
239 const Scalar e_0 = normal_direction_error(self_n, ne_n_0);
240 const Scalar e_1 = normal_direction_error(self_n, ne_n_1);
241 const Scalar e_2 = normal_direction_error(self_n, ne_n_2);
242
243 const bool e_0_ok = e_0 < parameters.flood_error_limit;
244 const bool e_1_ok = e_1 < parameters.flood_error_limit;
245 const bool e_2_ok = e_2 < parameters.flood_error_limit;
246
247 if (ne_select_0 && ne_select_1 && (e_0_ok || e_1_ok)) {
248 is_facet_selected[i] = true;
249 } else if (ne_select_1 && ne_select_2 && (e_1_ok || e_2_ok)) {
250 is_facet_selected[i] = true;
251 } else if (ne_select_2 && ne_select_0 && (e_2_ok || e_0_ok)) {
252 is_facet_selected[i] = true;
253 }
254 }
255 } // end of non selected facet that has two selectable neighbours
256 } // end of loop over faces
257 } // end of for(num_smooth_iterations)
258 } // end of should smooth boundary
259
260
261 return is_facet_selected;
262
263} // select_faces_by_normal_similarity
