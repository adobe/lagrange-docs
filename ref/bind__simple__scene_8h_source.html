31template <typename Scalar, typename Index>
32void bind_simple_scene(nb::module_& m)
33{
34 using MeshInstance3D = lagrange::scene::MeshInstance<Scalar, Index, 3>;
35 nb::class_<MeshInstance3D>(m, "MeshInstance3D", "A single mesh instance in a scene")
36 .def(nb::init<>())
37 .def_readwrite(
38 "mesh_index",
39 &MeshInstance3D::mesh_index)
40 .def_property(
41 "transform",
42 [](MeshInstance3D& self) {
43 auto& M = self.transform.matrix();
44 using MatrixType = std::decay_t<decltype(M)>;
45 static_assert(!MatrixType::IsRowMajor, "Transformation matrix is not column major");
46
47 span<Scalar> data(M.data(), M.size());
48 size_t shape[2]{static_cast<size_t>(M.rows()), static_cast<size_t>(M.cols())};
49 int64_t stride[2]{1, 4};
50 return span_to_tensor(data, shape, stride, nb::cast(&self));
51 },
52 [](MeshInstance3D& self, Tensor<Scalar> tensor) {
53 auto [values, shape, stride] = tensor_to_span(tensor);
54 auto& M = self.transform.matrix();
55 using MatrixType = std::decay_t<decltype(M)>;
56 static_assert(!MatrixType::IsRowMajor, "Transformation matrix is not column major");
57
58 la_runtime_assert(is_dense(shape, stride));
59 la_runtime_assert(check_shape(shape, 4, 4));
60 if (stride[0] == 1) {
61 // Tensor is col major.
62 std::copy(values.begin(), values.end(), M.data());
63 } else {
64 // Tensor is row major.
65 M(0, 0) = values[0];
66 M(0, 1) = values[1];
67 M(0, 2) = values[2];
68 M(0, 3) = values[3];
69
70 M(1, 0) = values[4];
71 M(1, 1) = values[5];
72 M(1, 2) = values[6];
73 M(1, 3) = values[7];
74
75 M(2, 0) = values[8];
76 M(2, 1) = values[9];
77 M(2, 2) = values[10];
78 M(2, 3) = values[11];
79
80 M(3, 0) = values[12];
81 M(3, 1) = values[13];
82 M(3, 2) = values[14];
83 M(3, 3) = values[15];
84 }
85 });
86
87 using SimpleScene3D = lagrange::scene::SimpleScene<Scalar, Index, 3>;
88 nb::class_<SimpleScene3D>(m, "SimpleScene3D", "Simple scene container for instanced meshes")
89 .def(nb::init<>())
90 .def_property_readonly(
91 "num_meshes",
92 &SimpleScene3D::get_num_meshes,
93 "Number of meshes in the scene")
94 .def(
95 "num_instances",
96 &SimpleScene3D::get_num_instances,
97 "mesh_index"_a)
98 .def_property_readonly(
99 "total_num_instances",
100 &SimpleScene3D::compute_num_instances,
101 "Total number of instances for all meshes in the scene")
102 .def(
103 "get_mesh",
104 &SimpleScene3D::get_mesh,
105 "mesh_index"_a)
106 .def(
107 "ref_mesh",
108 &SimpleScene3D::ref_mesh,
109 "mesh_index"_a)
110 .def(
111 "get_instance",
112 &SimpleScene3D::get_instance,
113 "mesh_index"_a,
114 "instance_index"_a)
115 .def(
116 "reserve_meshes",
117 &SimpleScene3D::reserve_meshes,
118 "num_meshes"_a)
119 .def(
120 "add_mesh",
121 &SimpleScene3D::add_mesh,
122 "mesh"_a)
123 .def(
124 "reserve_instances",
125 &SimpleScene3D::reserve_instances,
126 "mesh_index"_a,
127 "num_instances"_a)
128 .def(
129 "add_instance",
130 &SimpleScene3D::add_instance,
131 "instance"_a);
132}
