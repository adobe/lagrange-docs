31template <typename Scalar, typename Index>
32void bind_simple_scene(nb::module_& m)
33{
34 using MeshInstance3D = lagrange::scene::MeshInstance<Scalar, Index, 3>;
35 nb::class_<MeshInstance3D>(m, "MeshInstance3D", "A single mesh instance in a scene")
36 .def(nb::init<>())
37 .def_rw("mesh_index", &MeshInstance3D::mesh_index)
38 .def_prop_rw(
39 "transform",
40 [](MeshInstance3D& self) {
41 auto& M = self.transform.matrix();
42 using MatrixType = std::decay_t<decltype(M)>;
43 static_assert(!MatrixType::IsRowMajor, "Transformation matrix is not column major");
44
45 span<Scalar> data(M.data(), M.size());
46 size_t shape[2]{static_cast<size_t>(M.rows()), static_cast<size_t>(M.cols())};
47 int64_t stride[2]{1, 4};
48 return span_to_tensor(data, shape, stride, nb::cast(&self));
49 },
50 [](MeshInstance3D& self, Tensor<Scalar> tensor) {
51 auto [values, shape, stride] = tensor_to_span(tensor);
52 auto& M = self.transform.matrix();
53 using MatrixType = std::decay_t<decltype(M)>;
54 static_assert(!MatrixType::IsRowMajor, "Transformation matrix is not column major");
55
56 la_runtime_assert(is_dense(shape, stride));
57 la_runtime_assert(check_shape(shape, 4, 4));
58 if (stride[0] == 1) {
59 // Tensor is col major.
60 std::copy(values.begin(), values.end(), M.data());
61 } else {
62 // Tensor is row major.
63 M(0, 0) = values[0];
64 M(0, 1) = values[1];
65 M(0, 2) = values[2];
66 M(0, 3) = values[3];
67
68 M(1, 0) = values[4];
69 M(1, 1) = values[5];
70 M(1, 2) = values[6];
71 M(1, 3) = values[7];
72
73 M(2, 0) = values[8];
74 M(2, 1) = values[9];
75 M(2, 2) = values[10];
76 M(2, 3) = values[11];
77
78 M(3, 0) = values[12];
79 M(3, 1) = values[13];
80 M(3, 2) = values[14];
81 M(3, 3) = values[15];
82 }
83 });
84
85 using SimpleScene3D = lagrange::scene::SimpleScene<Scalar, Index, 3>;
86 nb::class_<SimpleScene3D>(m, "SimpleScene3D", "Simple scene container for instanced meshes")
87 .def(nb::init<>())
88 .def_prop_ro("num_meshes", &SimpleScene3D::get_num_meshes, "Number of meshes in the scene")
89 .def("num_instances", &SimpleScene3D::get_num_instances, "mesh_index"_a)
90 .def_prop_ro(
91 "total_num_instances",
92 &SimpleScene3D::compute_num_instances,
93 "Total number of instances for all meshes in the scene")
94 .def("get_mesh", &SimpleScene3D::get_mesh, "mesh_index"_a)
95 .def("ref_mesh", &SimpleScene3D::ref_mesh, "mesh_index"_a)
96 .def("get_instance", &SimpleScene3D::get_instance, "mesh_index"_a, "instance_index"_a)
97 .def("reserve_meshes", &SimpleScene3D::reserve_meshes, "num_meshes"_a)
98 .def("add_mesh", &SimpleScene3D::add_mesh, "mesh"_a)
99 .def(
100 "reserve_instances",
101 &SimpleScene3D::reserve_instances,
102 "mesh_index"_a,
103 "num_instances"_a)
104 .def("add_instance", &SimpleScene3D::add_instance, "instance"_a);
105}
