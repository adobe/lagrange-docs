18template <typename Derived>
19void ArrayBase::set(const Eigen::MatrixBase<Derived>& data)
20{
21 using EvalType = std::decay_t<decltype(data.eval())>;
22 if (auto ptr = dynamic_cast<EigenArray<EvalType>*>(this)) {
23 ptr->set(data);
24 } else if (
25 auto ptr2 = dynamic_cast<RawArray<
26 typename EvalType::Scalar,
27 EvalType::RowsAtCompileTime,
28 EvalType::ColsAtCompileTime,
29 EvalType::Options>*>(this)) {
30 ptr2->set(data);
31 } else if (auto ptr3 = dynamic_cast<EigenArrayRef<EvalType>*>(this)) {
32 ptr3->set(data);
33 } else if (is_compatible<EvalType>(true)) {
34 // The Derived type is not an exact match to the true type, fall back to
35 // brute force copying.
36 if (EvalType::IsRowMajor == is_row_major()) {
37 resize(data.rows(), data.cols());
38 view<EvalType>() = data;
39 } else {
40 // Type is compatible except storage order.
41 using TransposedType = std::decay_t<decltype(data.transpose().eval())>;
42 la_runtime_assert(is_compatible<TransposedType>());
43 resize(data.rows(), data.cols());
44 view<TransposedType>() = data;
45 }
46 } else {
47 throw std::runtime_error(string_format(
48 "Unsupported type passed to ArrayBase::set(). Expecting {}",
49 type_name()));
50 }
51}

53template <typename Derived>
54void ArrayBase::set(Eigen::MatrixBase<Derived>&& data)
55{
56 using EvalType = std::decay_t<decltype(data.eval())>;
57 if (auto ptr = dynamic_cast<EigenArray<EvalType>*>(this)) {
58 ptr->set(std::move(data.derived()));
59 } else if (
60 auto ptr2 = dynamic_cast<RawArray<
61 typename EvalType::Scalar,
62 EvalType::RowsAtCompileTime,
63 EvalType::ColsAtCompileTime,
64 EvalType::Options>*>(this)) {
65 // Fall back to copying because RawArray cannot change its data memory
66 // pointer.
67 ptr2->set(data);
68 } else if (auto ptr3 = dynamic_cast<EigenArrayRef<EvalType>*>(this)) {
69 ptr3->set(std::move(data.derived()));
70 } else if (is_compatible<EvalType>(true)) {
71 // The Derived type is not an exact match to the true type, fall back to
72 // brute force copying.
73 if (EvalType::IsRowMajor == is_row_major()) {
74 resize(data.rows(), data.cols());
75 view<EvalType>() = data;
76 } else {
77 // Type is compatible except storage order.
78 using TransposedType = std::decay_t<decltype(data.transpose().eval())>;
79 la_runtime_assert(is_compatible<TransposedType>());
80 resize(data.rows(), data.cols());
81 view<TransposedType>() = std::move(data);
82 }
83 } else {
84 throw std::runtime_error(string_format(
85 "Unsupported type passed to ArrayBase::set(). Expecting {}",
86 type_name()));
87 }

122template <typename TargetType>
123bool ArrayBase::is_compatible(bool ignore_storage_order) const
124{
125 using Scalar = typename TargetType::Scalar;
126 constexpr int RowsAtCompileTime = TargetType::RowsAtCompileTime;
127 constexpr int ColsAtCompileTime = TargetType::ColsAtCompileTime;
128 constexpr int row_major = TargetType::IsRowMajor;
129
130 if (RowsAtCompileTime > 0 && this->rows() != RowsAtCompileTime) return false;
131 if (ColsAtCompileTime > 0 && this->cols() != ColsAtCompileTime) return false;
132 if (ScalarToEnum_v<Scalar> != m_scalar_type) return false;
133 if (!ignore_storage_order && this->rows() != 1 && this->cols() != 1 &&
134 row_major != is_row_major()) {
135 lagrange::logger().error(
136 "Target storage order ({}) does not match array storage order ({}).",
137 row_major == 1 ? "RowMajor" : "ColMajor",
138 is_row_major() == 1 ? "RowMajor" : "ColMajor");
139 return false;
140 }
141
142 return true;
143}

146template <typename Derived>
147std::unique_ptr<ArrayBase> ArrayBase::row_slice_impl(
148 const Eigen::MatrixBase<Derived>& matrix,
149 Index num_rows,
150 const IndexFunction& mapping_fn)
151{
152 using Scalar = typename Derived::Scalar;
153 using OutEigenType = Eigen::Matrix<
154 Scalar,
155 Eigen::Dynamic,
156 Derived::ColsAtCompileTime,
157 Derived::IsRowMajor ? Eigen::RowMajor : Eigen::ColMajor>;
158
159 const auto num_cols = matrix.cols();
160 OutEigenType out_matrix(num_rows, num_cols);
161 tbb::parallel_for(
162 tbb::blocked_range<size_t>(0, num_rows),
163 [&](const tbb::blocked_range<size_t>& r) {
164 for (auto i = r.begin(); i != r.end(); i++) {
165 out_matrix.row(i) = matrix.row(mapping_fn(i));
166 }
167 });
168
169 return std::make_unique<EigenArray<OutEigenType>>(std::move(out_matrix));
170}

172template <typename Derived>
173std::enable_if_t<std::is_integral<typename Derived::Scalar>::value, std::unique_ptr<ArrayBase>>
174ArrayBase::row_slice_impl(
175 const Eigen::MatrixBase<Derived>& matrix,
176 Index num_rows,
177 const WeightedIndexFunction& mapping_fn)
178{
179 using Scalar = typename Derived::Scalar;
180 using OutEigenType = Eigen::Matrix<
181 Scalar,
182 Eigen::Dynamic,
183 Derived::ColsAtCompileTime,
184 Derived::IsRowMajor ? Eigen::RowMajor : Eigen::ColMajor>;
185 const auto num_cols = matrix.cols();
186 tbb::enumerable_thread_specific<std::vector<std::pair<Index, double>>> weights;
187
188 using DoubleEigenType = Eigen::Matrix<
189 double,
190 Eigen::Dynamic,
191 Derived::ColsAtCompileTime,
192 Derived::IsRowMajor ? Eigen::RowMajor : Eigen::ColMajor>;
193 DoubleEigenType out_matrix(num_rows, num_cols);
194 out_matrix.setZero();
195
196 tbb::parallel_for(
197 tbb::blocked_range<Index>(0, num_rows),
198 [&](const tbb::blocked_range<Index>& r) {
199 for (auto i = r.begin(); i != r.end(); i++) {
200 auto& entries = weights.local();
201 mapping_fn(i, entries);
202 for (const auto& entry : entries) {
203 out_matrix.row(i) +=
204 matrix.row(entry.first).template cast<double>() * entry.second;
205 }
206 }
207 });
208 return std::make_unique<EigenArray<OutEigenType>>(
209 std::move(out_matrix.array().round().template cast<Scalar>().matrix().eval()));
210}
