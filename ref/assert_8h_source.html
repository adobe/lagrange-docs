142// Note: In order for the `la_xxx_assert()` macro to behave as a function, it needs to expand to an
143// expression. This means we cannot use `if` or `do ... while` statements. The only options are
144// ternary operators ?: or logical boolean using &&.
145#define LA_INTERNAL_ASSERT_ARGS_2(condition, message) \
146 ((void)(!static_cast<bool>((condition)) && ::lagrange::assertion_failed(LA_ASSERT_FUNCTION, __FILE__, __LINE__, #condition, message)))
147#define LA_INTERNAL_ASSERT_ARGS_1(condition) LA_INTERNAL_ASSERT_ARGS_2(condition, "")
148
149#define LA_INTERNAL_IGNORE_ARGS_2(condition, message) ((void)(0))
150#define LA_INTERNAL_IGNORE_ARGS_1(condition) LA_INTERNAL_IGNORE_ARGS_2(condition, "")
