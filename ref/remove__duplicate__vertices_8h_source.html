35// A much lighter version of igl::unique_rows()
36template <typename DerivedA, typename Index>
37void unique_rows(
38 const Eigen::DenseBase<DerivedA>& A,
39 Index& num_unique_vertices,
40 std::vector<Index>& forward_mapping)
41{
42 // Sort the rows
43 Eigen::VectorXi IM;
44 {
45 DerivedA __; // sorted A
46 constexpr bool is_ascending = true; // does not matter really
47 igl::sortrows(A, is_ascending, __, IM);
48 }
49
50 // Build the forward mapping and the number of unique rows
51 const Index num_rows = safe_cast<Index>(A.rows());
52 forward_mapping.resize(num_rows);
53 num_unique_vertices = 0;
54 forward_mapping[IM(0)] = num_unique_vertices;
55 for (auto i : range<Index>(1, num_rows)) {
56 if ((A.row(IM(i)) != A.row(IM(i - 1)))) {
57 ++num_unique_vertices;
58 }
59 forward_mapping[IM(i)] = static_cast<Index>(num_unique_vertices);
60 }
61 ++num_unique_vertices; // num_of_xx = last index + 1
62}
63} // namespace internal

78template <typename MeshType>
79std::unique_ptr<MeshType> remove_duplicate_vertices(
80 const MeshType& mesh,
81 const std::vector<std::string>& vertex_attribute_names,
82 const std::vector<std::string>& indexed_attribute_names = {})
83{
84 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
85 la_runtime_assert(
86 mesh.get_vertex_per_facet() == 3,
87 std::string("vertex per facet is ") + std::to_string(mesh.get_vertex_per_facet()));
88
89 using Scalar = typename MeshType::Scalar;
90 using Index = typename MeshType::Index;
91 using Entries = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>;
92
93 const auto& vertices = mesh.get_vertices();
94 const auto& facets = mesh.get_facets();
95 const Index num_vertices = mesh.get_num_vertices();
96 const Index num_facets = mesh.get_num_facets();
97
98 const Index dim = mesh.get_dim();
99 Index num_cols = dim;
100 for (const auto& attr_name : vertex_attribute_names) {
101 la_runtime_assert(mesh.has_vertex_attribute(attr_name));
102 num_cols += mesh.get_vertex_attribute(attr_name).cols();
103 }
104 for (const auto& attr_name : indexed_attribute_names) {
105 la_runtime_assert(mesh.has_indexed_attribute(attr_name));
106 auto attr = mesh.get_indexed_attribute(attr_name);
107 num_cols += std::get<0>(attr).cols();
108 }
109
110 Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> vertices_and_keys;
111 vertices_and_keys.resize(num_vertices, num_cols);
112 vertices_and_keys.leftCols(dim) = vertices;
113 Index col_count = dim;
114
115 for (const auto& attr_name : vertex_attribute_names) {
116 const auto& attr = mesh.get_vertex_attribute(attr_name);
117 vertices_and_keys.block(0, col_count, num_vertices, attr.cols()) = attr;
118 col_count += attr.cols();
119 }
120 for (const auto& attr_name : indexed_attribute_names) {
121 const auto attr = mesh.get_indexed_attribute(attr_name);
122 const auto& attr_values = std::get<0>(attr);
123 const auto& attr_indices = std::get<1>(attr);
124 for (auto i : range(num_facets)) {
125 vertices_and_keys.row(facets(i, 0)).segment(col_count, attr_values.cols()) =
126 attr_values.row(attr_indices(i, 0));
127 vertices_and_keys.row(facets(i, 1)).segment(col_count, attr_values.cols()) =
128 attr_values.row(attr_indices(i, 1));
129 vertices_and_keys.row(facets(i, 2)).segment(col_count, attr_values.cols()) =
130 attr_values.row(attr_indices(i, 2));
131 }
132 col_count += attr_values.cols();
133 }
134
135 Entries unique_vertices_and_keys;
136 std::vector<Index> forward_mapping;
137 Index num_unique_vertices;
138 internal::unique_rows(vertices_and_keys, num_unique_vertices, forward_mapping);
139
140 la_runtime_assert(safe_cast<Eigen::Index>(forward_mapping.size()) == vertices.rows());
141
142 if (num_unique_vertices < mesh.get_num_vertices()) {
143 auto mesh2 = reorder_mesh_vertices(mesh, forward_mapping);
144 mesh2 = remove_topologically_degenerate_triangles(*mesh2);
145 return mesh2;
146 } else {
147 la_runtime_assert(num_unique_vertices == mesh.get_num_vertices());
148 auto mesh2 = create_mesh(mesh.get_vertices(), mesh.get_facets());
149 map_attributes(mesh, *mesh2);
150 return mesh2;
151 }
152}

171template <typename MeshType>
172std::unique_ptr<MeshType> remove_duplicate_vertices(
173 const MeshType& mesh,
174 const std::string& key_name = "",
175 bool with_uv = true)
176{
177 std::vector<std::string> vertex_attributes, indexed_attributes;
178 if (key_name != "") {
179 vertex_attributes.push_back(key_name);
180 }
181 if (with_uv && mesh.is_uv_initialized()) {
182 indexed_attributes.push_back("uv");
183 }
184 return remove_duplicate_vertices(mesh, vertex_attributes, indexed_attributes);
185}
