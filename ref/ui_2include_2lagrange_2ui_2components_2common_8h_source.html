40inline std::string get_name(const Registry& r, Entity e)
41{
42 if (!r.valid(e)) return lagrange::string_format("Invalid Entity (ID={})", e);
43 if (!r.all_of<Name>(e)) return lagrange::string_format("Unnamed Entity (ID={})", e);

47inline bool set_name(Registry& r, Entity e, const std::string& name)
