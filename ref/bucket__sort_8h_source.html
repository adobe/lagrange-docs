23template <typename Index>
24struct BucketSortResult
25{
27 Index num_representatives;
28
30 std::vector<Index> sorted_elements;
31
34 std::vector<Index> representative_offsets;
35};

52template <typename Index>
53BucketSortResult<Index> bucket_sort(
54 DisjointSets<Index>& unified_indices,
55 span<Index> element_representative)
56{
57 la_debug_assert(unified_indices.size() == element_representative.size());
58 BucketSortResult<Index> result;
59
60 auto& num_representatives = result.num_representatives;
61 auto& sorted_elements = result.sorted_elements;
62 auto& representative_offsets = result.representative_offsets;
63
64 Index num_elements = static_cast<Index>(element_representative.size());
65
66 // Calculate representative element for each bucket
67 num_representatives = 0;
68 std::fill(element_representative.begin(), element_representative.end(), invalid<Index>());
69 for (Index e = 0; e < num_elements; ++e) {
70 Index r = unified_indices.find(e);
71 if (element_representative[r] == invalid<Index>()) {
72 element_representative[r] = num_representatives++;
73 }
74 element_representative[e] = element_representative[r];
75 }
76
77 // Count number of items in each bucket
78 representative_offsets.assign(num_representatives + 1, 0);
79 for (Index e = 0; e < num_elements; ++e) {
80 representative_offsets[element_representative[e] + 1]++;
81 }
82
83 // Prefix sum to compute offsets into each bucket
84 std::partial_sum(
85 representative_offsets.begin(),
86 representative_offsets.end(),
87 representative_offsets.begin());
88
89 // Bucket sort element indices
90 sorted_elements.resize(num_elements);
91 for (Index e = 0; e < num_elements; ++e) {
92 sorted_elements[representative_offsets[element_representative[e]]++] = e;
93 }
94
95 // Shift back the offset buffer (can use std::shift_right in C++20 :p)
96 std::rotate(
97 representative_offsets.rbegin(),
98 representative_offsets.rbegin() + 1,
99 representative_offsets.rend());
100 representative_offsets.front() = 0;
101
102 return result;
103}
