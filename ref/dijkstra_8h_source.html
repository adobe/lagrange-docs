39template <typename MeshType, typename DistFunc, typename ProcessFunc>
40void dijkstra(
41 const MeshType& mesh,
42 const std::vector<typename MeshType::Index>& seed_vertices,
43 const std::vector<typename MeshType::Scalar>& seed_vertex_dist,
44 typename MeshType::Scalar radius,
45 const DistFunc& dist,
46 const ProcessFunc& process)
47{
48 using Scalar = typename MeshType::Scalar;
49 using Index = typename MeshType::Index;
50
51 la_runtime_assert(
52 mesh.get_vertex_per_facet() == 3,
53 "Only triangle meshes are supported for now.");
54 la_runtime_assert(
55 mesh.is_connectivity_initialized(),
56 "BSF requires mesh connectivity to be initialized.");
57
58 if (radius <= 0.0) {
59 radius = std::numeric_limits<Scalar>::max();
60 }
61
62 const Index num_vertices = mesh.get_num_vertices();
63
64 using Entry = std::pair<Index, Scalar>;
65 auto comp = [](const Entry& e1, const Entry& e2) { return e1.second > e2.second; };
66 std::priority_queue<Entry, std::vector<Entry>, decltype(comp)> Q(comp);
67 std::vector<bool> visited(num_vertices, false);
68
69 size_t num_seeds = seed_vertices.size();
70 la_runtime_assert(
71 num_seeds == seed_vertex_dist.size(),
72 "Inconsistent number of seed distances.");
73 for (size_t i = 0; i < num_seeds; i++) {
74 la_runtime_assert(seed_vertices[i] < num_vertices);
75 Q.push({seed_vertices[i], seed_vertex_dist[i]});
76 }
77
78 while (!Q.empty()) {
79 Entry entry = Q.top();
80 Q.pop();
81
82 Index vi = entry.first;
83 Scalar di = entry.second;
84
85 if (visited[vi]) continue;
86
87 bool done = process(vi, di);
88 if (done) break;
89 visited[vi] = true;
90
91 const auto& adj_vertices = mesh.get_vertices_adjacent_to_vertex(entry.first);
92 for (const auto& vj : adj_vertices) {
93 Scalar dj = di + dist(vi, vj);
94 if (dj < radius) {
95 Q.push({vj, dj});
96 }
97 }
98 }
99}
