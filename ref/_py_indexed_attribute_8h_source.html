34 std::shared_ptr<AttributeBase> ptr()
35 {
36 auto attr = m_attr.lock();
37 if (attr == nullptr) throw Error("Indexed attribute is no longer valid!");
38 return attr;
39 }

41 template <typename CallBack>
42 auto process(CallBack&& cb)
43 {
44 auto attr = m_attr.lock();
45 if (attr == nullptr) throw Error("Indexed attribute is no longer valid!");
46 if (attr->get_element_type() != Indexed)
47 throw Error("Attribute is not an indexed attribute");
48
49#define LA_X_process(Index, ValueType) \
50 { \
51 auto indexed_attr = dynamic_cast<IndexedAttribute<ValueType, Index>*>(attr.get()); \
52 if (indexed_attr != nullptr) { \
53 return cb(*indexed_attr); \
54 } \
55 }
56#define LA_X_process_index(_, Index) LA_ATTRIBUTE_X(process, Index)
57 LA_SURFACE_MESH_INDEX_X(process_index, 0)
58#undef LA_X_process_index
59#undef LA_X_process
60 throw Error("Cannot process indexed attribute with unsupported types!");
61 }
62
63 PyAttribute values()
64 {
65 return process([&](auto& indexed_attr) {
66 auto attr = m_attr.lock();
67 la_debug_assert(attr != nullptr);
68 auto& value_attr = indexed_attr.values();
69 std::shared_ptr<AttributeBase> alias_ptr(attr, &value_attr);
70 return PyAttribute(alias_ptr);
71 });
72 }
73
74 PyAttribute indices()
75 {
76 return process([&](auto& indexed_attr) {
77 auto attr = m_attr.lock();
78 la_debug_assert(attr != nullptr);
79 auto& index_attr = indexed_attr.indices();
80 std::shared_ptr<AttributeBase> alias_ptr(attr, &index_attr);
81 return PyAttribute(alias_ptr);
82 });
83 }
84
85private:
86 std::weak_ptr<AttributeBase> m_attr;
87};
