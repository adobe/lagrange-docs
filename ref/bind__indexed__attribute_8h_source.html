26void bind_indexed_attribute(nanobind::module_& m)
27{
28 namespace nb = nanobind;
29 using namespace nb::literals;
30
31 auto indexed_attr_class = nb::class_<PyIndexedAttribute>(m, "IndexedAttribute");
32
33 indexed_attr_class.def_property_readonly("element_type", [](PyIndexedAttribute& self) {
34 return self->get_element_type();
35 });
36 indexed_attr_class.def_property_readonly("usage", [](PyIndexedAttribute& self) {
37 return self->get_usage();
38 });
39 indexed_attr_class.def_property_readonly("num_channels", [](PyIndexedAttribute& self) {
40 return self->get_num_channels();
41 });
42
43 indexed_attr_class.def_property_readonly("values", &PyIndexedAttribute::values);
44 indexed_attr_class.def_property_readonly("indices", &PyIndexedAttribute::indices);
45}
