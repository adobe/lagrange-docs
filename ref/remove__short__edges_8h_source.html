41template <typename MeshType>
42std::unique_ptr<MeshType> remove_short_edges(
43 const MeshType& in_mesh,
44 typename MeshType::Scalar tol = 0.0)
45{
46 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
47 using Index = typename MeshType::Index;
48
49 lagrange::logger().trace("[remove_short_edges]");
50
51 // Topological degeneracy can affect the index mapping algorithm used here.
52 // So eliminating topological degeneracy first.
53 auto meshPtr = remove_topologically_degenerate_triangles(in_mesh);
54 auto& mesh = *meshPtr;
55
56 const auto num_vertices = mesh.get_num_vertices();
57 const auto num_facets = mesh.get_num_facets();
58 const auto vertex_per_facet = mesh.get_vertex_per_facet();
59
60 DisjointSets<Index> clusters(num_vertices);
61
62 compute_edge_lengths(mesh);
63 const auto num_edges = mesh.get_num_edges();
64 const auto& edge_lengths = mesh.get_edge_attribute("length");
65 for (Index edge_idx = 0; edge_idx < num_edges; ++edge_idx) {
66 const auto& edge = mesh.get_edge_vertices(edge_idx);
67 const auto& length = edge_lengths(edge_idx);
68 if (length <= tol) {
69 clusters.merge(edge[0], edge[1]);
70 }
71 }
72
73 auto facets = mesh.get_facets(); // Intentionally copying the data.
74 std::transform(
75 facets.data(),
76 facets.data() + num_facets * vertex_per_facet,
77 facets.data(),
78 [&](Index i) { return clusters.find(i); });
79 auto mesh2 = create_mesh(mesh.get_vertices(), std::move(facets));
80
81 // Above we create topologically degenerate triangles, but vertex and
82 // facet indexes don't actually change, so we map attributes directly
83 map_attributes(mesh, *mesh2);
84
85 mesh2 = remove_topologically_degenerate_triangles(*mesh2);
86 mesh2 = remove_isolated_vertices(*mesh2);
87 return mesh2;
88}
