42template <typename DerivedE, typename DerivedO>
43bool chain_edges_into_simple_loops(
44 const Eigen::MatrixBase<DerivedE> &edges,
45 std::vector<std::vector<typename DerivedE::Scalar>> &output,
46 Eigen::PlainObjectBase<DerivedO> &remaining_edges)
47{
48 using Index = typename DerivedE::Scalar;
49 using IndexArray = Eigen::Matrix<Index, Eigen::Dynamic, 1>;
50
51 output.clear();
52
53 if (edges.rows() == 0) {
54 // Empty graph
55 return true;
56 }
57
58 Index num_vertices = edges.maxCoeff() + 1;
59 Index num_edges = (Index)edges.rows();
60
61 // Count degree_in and degree_out, and make sure that it matches
62 IndexArray degree_in(num_vertices);
63 IndexArray degree_out(num_vertices);
64 degree_in.setZero();
65 degree_out.setZero();
66 for (Index e = 0; e < num_edges; ++e) {
67 Index v0 = edges(e, 0);
68 Index v1 = edges(e, 1);
69 degree_out[v0]++;
70 degree_in[v1]++;
71 }
72 if (!(degree_in.array() == degree_out.array()).all()) {
73 logger().debug("Input digraph has dangling vertices.");
74 }
75
76 // path -> first edge in the path
77 std::vector<Index> path_to_first_edge;
78
79 // vertex -> single outgoing edge along path (or INVALID if degree_out is > 1)
80 std::vector<Index> vertex_to_outgoing_edge(num_vertices, invalid<Index>());
81
82 // edge -> next edge along path (or INVALID if last edge along path)
83 std::vector<Index> next_edge_along_path(num_edges, invalid<Index>());
84
85 // Chain edges into paths
86 for (Index e = 0; e < num_edges; ++e) {
87 Index v0 = edges(e, 0);
88 if (degree_out[v0] == 1 && degree_in[v0] == 1) {
89 // v0 is a mid-path vertex. There is only one possibility for the next edge along a path
90 // going through v0.
91 vertex_to_outgoing_edge[v0] = e;
92 } else {
93 // v0 is a junction vertex. We start one path for each outgoing edge e starting from
94 // vertex v0.
95 path_to_first_edge.push_back(e);
96 }
97 }
98 // Chain edges together based on the vertex -> outgoing edge mapping we previously computed
99 for (Index e = 0; e < num_edges; ++e) {
100 Index v1 = edges(e, 1);
101 next_edge_along_path[e] = vertex_to_outgoing_edge[v1];
102 }
103
104 // Follow each path until we reach the last edge
105 std::vector<Index> edge_label(num_edges, invalid<Index>());
106 std::vector<Index> path_to_last_edge(path_to_first_edge.size());
107 std::stack<Index> ears;
108 std::vector<std::vector<Index>> paths_in(num_vertices);
109 std::vector<std::vector<Index>> paths_out(num_vertices);
110 std::vector<bool> path_is_pending(path_to_first_edge.size(), false);
111
112 auto first_vertex_in_path = [&](Index a) { return edges(path_to_first_edge[a], 0); };
113
114 auto last_vertex_in_path = [&](Index a) { return edges(path_to_last_edge[a], 1); };
115
116 // For each path that we have started at a junction vertex, follow edges along the path until
117 // each edge has been labeled as belonging to the path. Note that our paths do not contain
118 // junction vertices by construction. So the only case where the path can contain a cycle is
119 // when the path starts and ends at the same vertex.
120 for (Index a = 0; a < (Index)path_to_first_edge.size(); ++a) {
121 // Follow edges along the path and label them as belonging to the path
122 for (Index e = path_to_first_edge[a];
123 e != invalid<Index>() && edge_label[e] == invalid<Index>();
124 e = next_edge_along_path[e]) {
125 edge_label[e] = a;
126 path_to_last_edge[a] = e;
127 }
128
129 const Index v_first = first_vertex_in_path(a);
130 const Index v_last = last_vertex_in_path(a);
131
132 // Compute outgoing and ingoing paths per vertices of degree > 1
133 paths_out[v_first].push_back(a);
134 paths_in[v_last].push_back(a);
135
136 if (v_first == v_last) {
137 // Path is an ear (simple loop), will be popped next
138 la_debug_assert(path_is_pending[a] == false);
139 path_is_pending[a] = true;
140 ears.push(a);
141 }
142 }
143
144 // For paths which are isolated cycles, there is no "starting vertex" (each vertex has total
145 // degree 2). We do an additional pass on each edge and start a new path for each unlabeled
146 // edge.
147 for (Index e = 0; e < num_edges; ++e) {
148 if (edge_label[e] == invalid<Index>()) {
149 // Add a new path starting here
150 const Index a = (Index)path_to_first_edge.size();
151 path_to_first_edge.emplace_back(e);
152 path_to_last_edge.emplace_back(e);
153 path_is_pending.push_back(false);
154
155 // Follow edges along the path and record the last edge on the path
156 edge_label[e] = a;
157 for (Index ei = next_edge_along_path[e];
158 ei != invalid<Index>() && edge_label[ei] == invalid<Index>();
159 ei = next_edge_along_path[ei]) {
160 edge_label[ei] = a;
161 path_to_last_edge[a] = ei;
162 }
163 la_debug_assert(next_edge_along_path[path_to_last_edge[a]] == e);
164 la_debug_assert(first_vertex_in_path(a) == last_vertex_in_path(a));
165 next_edge_along_path[path_to_last_edge[a]] = invalid<Index>();
166
167 // Path is an isolated cycle
168 la_debug_assert(path_is_pending[a] == false);
169 path_is_pending[a] = true;
170 ears.push(a);
171 }
172 }
173
174 // Swap/pop_back trick to remove an element from a std::vector
175 auto remove_from_vector = [](std::vector<Index> &vec, size_t idx) {
176 la_debug_assert(idx < vec.size());
177 std::swap(vec[idx], vec.back());
178 vec.pop_back();
179 };
180
181 Index num_edges_removed = 0;
182 std::vector<bool> edge_is_removed(num_edges, false);
183
184 // Pop ears repeatedly
185 std::vector<bool> path_is_removed(path_to_first_edge.size(), false);
186 while (!ears.empty()) {
187 const Index a = ears.top();
188 ears.pop();
189 la_debug_assert(!path_is_removed[a]);
190 path_is_removed[a] = true;
191
192 // path starts and ends on the same vertex, compute corresponding (simple) loop
193 la_debug_assert(first_vertex_in_path(a) == last_vertex_in_path(a));
194 la_debug_assert(path_to_first_edge[a] != invalid<Index>());
195 std::vector<Index> loop;
196 for (Index e = path_to_first_edge[a]; e != invalid<Index>(); e = next_edge_along_path[e]) {
197 loop.push_back(e);
198 la_debug_assert(edge_is_removed[e] == false);
199 edge_is_removed[e] = true;
200 ++num_edges_removed;
201 }
202 output.emplace_back(std::move(loop));
203
204 // Remove current path from the in/out paths of the endpoint vertex v
205 const Index v = first_vertex_in_path(a);
206 la_debug_assert(v == last_vertex_in_path(a));
207 for (size_t i = 0; i < paths_out[v].size();) {
208 if (paths_out[v][i] == a) {
209 remove_from_vector(paths_out[v], i);
210 } else {
211 la_debug_assert(!path_is_removed[paths_out[v][i]]);
212 ++i;
213 }
214 }
215 for (size_t i = 0; i < paths_in[v].size();) {
216 if (paths_in[v][i] == a) {
217 remove_from_vector(paths_in[v], i);
218 } else {
219 la_debug_assert(!path_is_removed[paths_in[v][i]]);
220 ++i;
221 }
222 }
223
224 // If there are only 1 remaining in/out paths, join them
225 if (paths_in[v].size() == 1 && paths_out[v].size() == 1) {
226 const Index a_in = paths_in[v].front();
227 const Index a_out = paths_out[v].front();
228 la_debug_assert(last_vertex_in_path(a_in) == v);
229 la_debug_assert(first_vertex_in_path(a_out) == v);
230 if (a_in != a_out) {
231 // If the paths are different, then join them
232 la_debug_assert(path_to_first_edge[a_in] != invalid<Index>());
233 la_debug_assert(path_to_last_edge[a_in] != invalid<Index>());
234 la_debug_assert(path_to_first_edge[a_out] != invalid<Index>());
235 la_debug_assert(path_to_last_edge[a_out] != invalid<Index>());
236 la_debug_assert(next_edge_along_path[path_to_last_edge[a_in]] == invalid<Index>());
237 la_debug_assert(
238 edges(path_to_last_edge[a_in], 1) == edges(path_to_first_edge[a_out], 0));
239
240 // Replace a_out by a_in in last_vertex_in_path(a_out)
241 for (auto &ai : paths_in[last_vertex_in_path(a_out)]) {
242 if (ai == a_out) {
243 ai = a_in;
244 }
245 }
246
247 // Update chain to joint a_in --> a_out
248 next_edge_along_path[path_to_last_edge[a_in]] = path_to_first_edge[a_out];
249 path_to_last_edge[a_in] = path_to_last_edge[a_out];
250
251 // Cleaning up
252 path_to_first_edge[a_out] = invalid<Index>();
253 path_to_last_edge[a_out] = invalid<Index>();
254 path_is_removed[a_out] = true;
255 paths_in[v].clear();
256 paths_out[v].clear();
257 }
258 const Index v_first = first_vertex_in_path(a_in);
259 const Index v_last = last_vertex_in_path(a_in);
260 la_debug_assert(!path_is_removed[a_in]);
261 if (v_first == v_last && !path_is_pending[a_in]) {
262 path_is_pending[a_in] = true;
263 ears.push(a_in);
264 }
265 }
266 }
267
268 if (num_edges_removed != num_edges) {
269 logger().warn(
270 "Removing ears didn't result in an empty graph, number of edges remaining: {}",
271 num_edges - num_edges_removed);
272 Index true_num_removed = 0;
273 for (auto b : edge_is_removed) {
274 if (b) {
275 ++true_num_removed;
276 }
277 }
278 la_debug_assert(true_num_removed == num_edges_removed);
279 remaining_edges.resize(num_edges - num_edges_removed, 2);
280 for (Index e = 0, cnt = 0; e < num_edges; ++e) {
281 if (!edge_is_removed[e]) {
282 la_debug_assert(cnt < remaining_edges.rows());
283 remaining_edges.row(cnt++) = edges.row(e);
284 }
285 }
286 return false;
287 }
288
289 return true;
290}
