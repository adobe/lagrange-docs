53template <typename MeshType>
54std::unique_ptr<MeshType> reorder_mesh_vertices(
55 const MeshType& mesh,
56 const typename MeshType::IndexList& forward_mapping)
57{
58 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
59 la_runtime_assert(
60 mesh.get_vertex_per_facet() == 3,
61 std::string("vertex per facet is ") + std::to_string(mesh.get_vertex_per_facet()));
62
63 using Index = typename MeshType::Index;
64 using VertexArray = typename MeshType::VertexArray;
65 using FacetArray = typename MeshType::FacetArray;
66 using IndexList = typename MeshType::IndexList;
67
68 const auto num_old_vertices = mesh.get_num_vertices();
69 const auto& vertices = mesh.get_vertices();
70 const auto& facets = mesh.get_facets();
71 la_runtime_assert(num_old_vertices == safe_cast<Index>(forward_mapping.size()));
72
73 auto forward_mapping_no_invalid = [&forward_mapping](const Index io) {
74 return forward_mapping[io] == invalid<Index>() ? io : forward_mapping[io];
75 };
76
77 // Cound the number of new vertices
78 Index num_new_vertices = 0;
79 for (const auto io : range(num_old_vertices)) {
80 num_new_vertices = std::max(num_new_vertices, forward_mapping_no_invalid(io));
81 }
82 num_new_vertices += 1; // num_of_*** = biggest index + 1
83 la_runtime_assert(num_new_vertices <= num_old_vertices, "Number of vertices should not increase");
84
85
86 // Create the backward mapping and the new vertices
87 IndexList backward_mapping(num_new_vertices, invalid<Index>());
88 VertexArray vertices_new(num_new_vertices, mesh.get_dim());
89 for (const auto io : range(num_old_vertices)) {
90 const auto in = forward_mapping_no_invalid(io);
91 backward_mapping[in] = io;
92 vertices_new.row(in) = vertices.row(io);
93 }
94 // The mapping should be surjective.
95 la_runtime_assert(
96 std::find(backward_mapping.begin(), backward_mapping.end(), invalid<Index>()) ==
97 backward_mapping.end(),
98 "Forward mapping is not surjective");
99
100 // Create new faces
101 FacetArray facets_new(facets.rows(), facets.cols());
102 for (auto i : range(facets.rows())) {
103 facets_new(i, 0) = forward_mapping_no_invalid(facets(i, 0));
104 facets_new(i, 1) = forward_mapping_no_invalid(facets(i, 1));
105 facets_new(i, 2) = forward_mapping_no_invalid(facets(i, 2));
106 }
107
108 // Create new mesh
109 auto mesh2 = create_mesh(std::move(vertices_new), std::move(facets_new));
110
111 // Port attributes
112 map_attributes(mesh, *mesh2, backward_mapping);
113
114 return mesh2;
115}
