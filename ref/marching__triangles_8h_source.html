29template <typename MeshType_>
30struct MarchingTrianglesOutput
31{
32 using MeshType = MeshType_;
33 using Index = typename MeshType::Index;
34 using IndexList = typename MeshType::IndexList;
35 using Scalar = typename MeshType::Scalar;
36 using VertexArray = typename MeshType::VertexArray;
37 using EdgeArray = Eigen::Matrix<Index, Eigen::Dynamic, 2, Eigen::RowMajor>;
38
39 // The extracted vertices of the contour
40 VertexArray vertices;
41 // Note that the direction in the edges bears no particular meaning
42 EdgeArray edges;
43 // The edge parent that gives birth to a marching triangle vertex
44 IndexList vertices_parent_edge;
45 // The param ( 0 =< t =< 1 ) that gives birth to each vertex.
46 // NOTE: t is defined such that position of vertex is (1-t)*v0 + t*v1,
47 // i.e., t=0 means the vertex is on v0 and t=1 means that it is on v1.
48 std::vector<Scalar> vertices_parent_param;
49};

51// Perform marching triangles to extract the isocontours of a field
52// defined by the linear interpolation of a vertex attribute.
53//
54// Adapted from https://www.cs.ubc.ca/~rbridson/download/common_2008_nov_12.tar.gz
55// (code released to the __public domain__ by Robert Bridson)
56//
57// Input:
58// mesh_ref, the mesh to run marching cubes on. Can be 2D or 3D,
59// but must be triangular.
60// isovalue, the isovalue to be extracted.
61// vertex_attribute_name, the name of the vertex attribute
62// attribute_col_index, which column of the vertex attribute should be used?
63//
64template <typename MeshType>
65MarchingTrianglesOutput<MeshType> marching_triangles(
66 MeshType& mesh_ref,
67 const typename MeshType::Scalar isovalue,
68 const std::string vertex_attribute_name,
69 const typename MeshType::Index attribute_col_index = 0)
70{
71 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
72 using Index = typename MeshType::Index;
73 using Scalar = typename MeshType::Scalar;
74 using VertexType = typename MeshType::VertexType;
75 using MarchingTrianglesOutput = ::lagrange::MarchingTrianglesOutput<MeshType>;
76 using Edge = typename MeshType::Edge;
77
78 la_runtime_assert(
79 mesh_ref.has_vertex_attribute(vertex_attribute_name),
80 "attribute does not exist in the mesh");
81 la_runtime_assert(mesh_ref.get_vertex_per_facet() == 3, "only works for triangle meshes");
82 mesh_ref.initialize_edge_data();
83
84 const auto& attribute = mesh_ref.get_vertex_attribute(vertex_attribute_name);
85 la_runtime_assert(attribute_col_index < safe_cast<Index>(attribute.cols()), "col_index is invalid");
86 const auto& facets = mesh_ref.get_facets();
87 const auto& vertices = mesh_ref.get_vertices();
88
89 std::vector<Edge> extracted_edges;
90 std::vector<VertexType> extracted_vertices;
91 std::vector<Index> extracted_vertices_parent_edge;
92 std::vector<Scalar> extracted_vertices_parent_param;
93 std::vector<Index> parent_edge_to_extracted_vertex(
94 mesh_ref.get_num_edges(),
95 invalid<Index>());
96
97 //
98 // Find the point that attains a zero value on an edge
99 // (if it does not exists, creates it)
100 // Returns the index of this vertex
101 //
102 auto find_zero = [&](Index parent_edge_id, Index v0, Index v1, Scalar p0, Scalar p1) -> Index {
103 if (parent_edge_to_extracted_vertex[parent_edge_id] != invalid<Index>()) {
104 return parent_edge_to_extracted_vertex[parent_edge_id];
105 } else {
106 // Get the edge and see if the order of v0 v1 and the edge values are the same.
107 const auto parent_edge = mesh_ref.get_edge_vertices(parent_edge_id);
108 if ((v0 == parent_edge[1]) && (v1 == parent_edge[0])) {
109 std::swap(p0, p1);
110 std::swap(v0, v1);
111 }
112 assert((v1 == parent_edge[1]) && (v0 == parent_edge[0]));
113 // Now construct the vertex
114 const Scalar a = p1 / (p1 - p0);
115 const Scalar b = 1 - a;
116 assert(a >= 0);
117 assert(a <= 1);
118 assert(b >= 0);
119 assert(b <= 1);
120 la_runtime_assert(!std::isnan(a));
121 la_runtime_assert(!std::isnan(b));
122 const Index vertex_index = safe_cast<Index>(extracted_vertices.size());
123 const VertexType pos0 = vertices.row(v0);
124 const VertexType pos1 = vertices.row(v1);
125 // And push it into the data structures
126 extracted_vertices.emplace_back(a * pos0 + b * pos1);
127 extracted_vertices_parent_edge.emplace_back(parent_edge_id);
128 extracted_vertices_parent_param.emplace_back(b);
129 parent_edge_to_extracted_vertex[parent_edge_id] = vertex_index;
130 return vertex_index;
131 }
132 };
133
134 //
135 // Find the contour (if exists) in a triangle
136 //
137 auto contour_triangle = [&](const Index tri_id) {
138 const Index v0 = facets(tri_id, 0);
139 const Index v1 = facets(tri_id, 1);
140 const Index v2 = facets(tri_id, 2);
141 Scalar p0 = attribute(v0, attribute_col_index) - isovalue;
142 Scalar p1 = attribute(v1, attribute_col_index) - isovalue;
143 Scalar p2 = attribute(v2, attribute_col_index) - isovalue;
144
145 const Index e01 = mesh_ref.get_edge(tri_id, 0);
146 const Index e12 = mesh_ref.get_edge(tri_id, 1);
147 const Index e20 = mesh_ref.get_edge(tri_id, 2);
148
149 // guard against topological degeneracies
150 if (p0 == 0) p0 = Scalar(1e-30);
151 if (p1 == 0) p1 = Scalar(1e-30);
152 if (p2 == 0) p2 = Scalar(1e-30);
153
154 if (p0 < 0) {
155 if (p1 < 0) {
156 if (p2 < 0) {
157 return; // no contour here
158 } else { /* p2>0 */
159 extracted_edges.push_back(
160 Edge(find_zero(e12, v1, v2, p1, p2), find_zero(e20, v0, v2, p0, p2)));
161 } // p2
162 } else { // p1>0
163 if (p2 < 0) {
164 extracted_edges.push_back(
165 Edge(find_zero(e01, v0, v1, p0, p1), find_zero(e12, v1, v2, p1, p2)));
166 } else { /* p2>0 */
167 extracted_edges.push_back(
168 Edge(find_zero(e01, v0, v1, p0, p1), find_zero(e20, v0, v2, p0, p2)));
169 } // p2
170 } // p1
171 } else { // p0>0
172 if (p1 < 0) {
173 if (p2 < 0) {
174 extracted_edges.push_back(
175 Edge(find_zero(e20, v0, v2, p0, p2), find_zero(e01, v0, v1, p0, p1)));
176 } else { /* p2>0 */
177 extracted_edges.push_back(
178 Edge(find_zero(e12, v1, v2, p1, p2), find_zero(e01, v0, v1, p0, p1)));
179 } // p2
180 } else { // p1>0
181 if (p2 < 0) {
182 extracted_edges.push_back(
183 Edge(find_zero(e20, v0, v2, p0, p2), find_zero(e12, v1, v2, p1, p2)));
184 } else { /* p2>0 */
185 return; // no contour here
186 } // p2
187 } // p1
188 } // p0
189 };
190
191
192 //
193 // Contour all triangles
194 //
195 for (auto tri : range(mesh_ref.get_num_facets())) {
196 contour_triangle(tri);
197 }
198
199 //
200 // Now convert to output
201 //
202 MarchingTrianglesOutput output;
203 // Output edges
204 output.edges.resize(extracted_edges.size(), 2);
205 for (auto i : range(extracted_edges.size())) {
206 output.edges.row(i) << extracted_edges[i][0], extracted_edges[i][1];
207 }
208 // output vertices
209 // This we can just swap!
210 output.vertices_parent_edge.swap(extracted_vertices_parent_edge);
211 output.vertices_parent_param.swap(extracted_vertices_parent_param);
212 // rest we have to copy
213 output.vertices.resize(extracted_vertices.size(), mesh_ref.get_dim());
214 for (auto i : range(extracted_vertices.size())) {
215 output.vertices.row(i) = extracted_vertices[i];
216 }
217 return output;
218}
