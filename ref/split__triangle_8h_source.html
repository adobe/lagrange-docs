39template <typename VertexArray, typename Index>
40std::vector<Eigen::Matrix<Index, 3, 1>> split_triangle(
41 const VertexArray& vertices,
42 const std::vector<Index>& chain,
43 const Index v0,
44 const Index v1,
45 const Index v2)
46{
47 const Index chain_size = safe_cast<Index>(chain.size());
48 auto next = [&chain_size](Index i) { return (i + 1) % chain_size; };
49 auto prev = [&chain_size](Index i) { return (i + chain_size - 1) % chain_size; };
50 auto sq_length = [&vertices, &chain](Index vi, Index vj) {
51 return (vertices.row(chain[vi]) - vertices.row(chain[vj])).squaredNorm();
52 };
53 auto is_corner = [v0, v1, v2](Index v) { return v == v0 || v == v1 || v == v2; };
54 std::vector<Eigen::Matrix<Index, 3, 1>> facets;
55
56 const double NOT_USED = std::numeric_limits<double>::max();
57 Eigen::Matrix<Index, 3, 6, Eigen::RowMajor> candidates;
58 candidates << v0, next(v0), prev(v0), 0, 0, 0, v1, next(v1), prev(v1), 0, 0, 0, v2, next(v2),
59 prev(v2), 0, 0, 0;
60
61 Eigen::Matrix<double, 3, 2, Eigen::RowMajor> candidate_lengths;
62 candidate_lengths << sq_length(candidates(0, 1), candidates(0, 2)), NOT_USED,
63 sq_length(candidates(1, 1), candidates(1, 2)), NOT_USED,
64 sq_length(candidates(2, 1), candidates(2, 2)), NOT_USED;
65
66 auto index_comp = [&candidate_lengths](Index i, Index j) {
67 // Use greater than operator so the heap is min heap.
68 return candidate_lengths.row(i).minCoeff() > candidate_lengths.row(j).minCoeff();
69 };
70 std::priority_queue<Index, std::vector<Index>, decltype(index_comp)> Q(index_comp);
71 Q.push(0);
72 Q.push(1);
73 Q.push(2);
74
75 Eigen::Matrix<Index, Eigen::Dynamic, 3, Eigen::RowMajor> visited(chain_size, 3);
76 visited.setZero();
77 visited(v0, 0) = 1;
78 visited(v1, 1) = 1;
79 visited(v2, 2) = 1;
80
81 while (!Q.empty()) {
82 Index idx = Q.top();
83 Q.pop();
84 Index selection = 0;
85 if (candidate_lengths(idx, 0) != NOT_USED &&
86 candidate_lengths(idx, 0) <= candidate_lengths(idx, 1)) {
87 selection = 0;
88 } else if (
89 candidate_lengths(idx, 1) != NOT_USED &&
90 candidate_lengths(idx, 1) <= candidate_lengths(idx, 0)) {
91 selection = 1;
92 } else {
93 // Select neither candidate from this corner.
94 continue;
95 }
96 la_runtime_assert(candidate_lengths.row(idx).minCoeff() <= candidate_lengths.minCoeff());
97
98 Index base_v = candidates(idx, selection * 3);
99 Index right_v = candidates(idx, selection * 3 + 1);
100 Index left_v = candidates(idx, selection * 3 + 2);
101 la_runtime_assert(base_v >= 0 && base_v < chain_size);
102 la_runtime_assert(right_v >= 0 && right_v < chain_size);
103 la_runtime_assert(left_v >= 0 && left_v < chain_size);
104 la_runtime_assert(visited(base_v, idx) >= 1);
105
106 // A special case.
107 if (is_corner(base_v) && is_corner(right_v) && is_corner(left_v)) {
108 if (chain_size != 3) {
109 candidate_lengths(idx, selection) = NOT_USED;
110 Q.push(idx);
111 continue;
112 }
113 }
114
115 if (visited.row(base_v).sum() > 1 || visited(right_v, idx) > 1 ||
116 visited(left_v, idx) > 1) {
117 // This canadidate is invalid.
118 candidate_lengths(idx, selection) = NOT_USED;
119 Q.push(idx);
120 continue;
121 }
122
123 visited(base_v, idx) = 2;
124 visited(right_v, idx) = 1;
125 visited(left_v, idx) = 1;
126 facets.push_back({chain[base_v], chain[right_v], chain[left_v]});
127
128 // Update candidate from this corner.
129 if (visited.row(right_v).sum() == 1) {
130 Index right_to_right = next(right_v);
131 candidate_lengths(idx, 0) = sq_length(left_v, right_to_right);
132 candidates(idx, 0) = right_v;
133 candidates(idx, 1) = right_to_right;
134 candidates(idx, 2) = left_v;
135 } else {
136 candidate_lengths(idx, 0) = NOT_USED;
137 }
138
139 if (visited.row(left_v).sum() == 1) {
140 Index left_to_left = prev(left_v);
141 candidate_lengths(idx, 1) = sq_length(right_v, left_to_left);
142 candidates(idx, 3) = left_v;
143 candidates(idx, 4) = right_v;
144 candidates(idx, 5) = left_to_left;
145 } else {
146 candidate_lengths(idx, 1) = NOT_USED;
147 }
148 Q.push(idx);
149 }
150
151 Eigen::Matrix<Eigen::Index, Eigen::Dynamic, 1> visited_sum =
152 (visited.array() > 0).rowwise().count();
153 if ((visited_sum.array() > 1).count() == 3) {
154 Index f[3];
155 Index count = 0;
156 for (Index i = 0; i < chain_size; i++) {
157 if (visited_sum[i] > 1) {
158 la_runtime_assert(count < 3);
159 f[count] = chain[i];
160 count++;
161 }
162 }
163 la_runtime_assert(count == 3);
164 facets.push_back({f[0], f[1], f[2]});
165 }
166 return facets;
167}
