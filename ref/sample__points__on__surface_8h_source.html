23// sample_points_on_surface()
24//
25// Samples points on a mesh as uniformly as possible by splitting edges
26// until they are smaller than a certain value and also not sampling more
27// than one point inside a grid.
28//
29// This is a Lagrangified version of segmentation/PointSample::SampleViaGrid()
30
31// Output values
32template <typename MeshType>
33struct SamplePointsOnSurfaceOutput
34{
35public:
36 using Index = typename MeshType::Index;
37 using IndexList = typename MeshType::IndexList;
38 using VertexArray = typename MeshType::VertexArray;
39 // Same as vertex array, but must have 3 cols rather than whatever vertex array has
40 using BarycentricArray = Eigen::Matrix<
41 typename VertexArray::Scalar,
42 VertexArray::RowsAtCompileTime,
43 3, // VertexArray::ColsAtCompileTime,
44 VertexArray::Options>;
45
46 // Number of samples points
47 Index num_samples = 0;
48 // The facet id of the sampled points
49 IndexList facet_ids = IndexList(0);
50 // Coordinates of the samples points
51 VertexArray positions = VertexArray();
52 // Barycentric coordinates of the samples points
53 BarycentricArray barycentrics = BarycentricArray();
54};

56//
57// A few version follow with different ways of initializing the
58// list of active facts.
59//
60
61// You can choose which facets to sample from by specifying an array of indices.
62// Inputs
63// - The mesh
64// - Approximate number of points to be sampled
65// - Index of the faces to be sampled from, an empty list of indices means all faces.
66template <typename MeshType>
67SamplePointsOnSurfaceOutput<MeshType> sample_points_on_surface(
68 MeshType& mesh,
69 const typename MeshType::Index approx_num_points,
70 const typename MeshType::IndexList& active_facets)
71{
72 // Helper types
73 using Index = typename MeshType::Index;
74 using Scalar = typename MeshType::Scalar;
75 using VertexType = typename MeshType::VertexType;
76 //
77 using Output = SamplePointsOnSurfaceOutput<MeshType>;
78 //
79 using RowVectorS = Eigen::Matrix<Scalar, 1, Eigen::Dynamic>;
80 using RowVector3S = Eigen::Matrix<Scalar, 1, 3>;
81 using RowVectorI = Eigen::Matrix<Index, 1, Eigen::Dynamic>;
82 using BoundingBox = Eigen::AlignedBox<Scalar, Eigen::Dynamic>;
83
84 // Check conditions to be met by input
85 la_runtime_assert(mesh.get_vertex_per_facet() == 3, "only works for triangle meshes");
86 la_runtime_assert(approx_num_points != invalid<Index>());
87 la_runtime_assert(active_facets.size() <= safe_cast<size_t>(mesh.get_num_facets()));
88 for (const auto facet_id : active_facets) {
89 la_runtime_assert(facet_id < mesh.get_num_facets());
90 }
91
92
93 // Information regarding the mesh that we need
94 if (!mesh.has_facet_attribute("area")) {
95 compute_facet_area(mesh);
96 }
97 const auto& facet_area = mesh.get_facet_attribute("area");
98 const auto& facets = mesh.get_facets();
99 const auto& vertices = mesh.get_vertices();
100
101 // Get space dimensions
102 const Index n_dims = mesh.get_dim();
103
104 // Compute the total area of active part of the mesh
105 // Otherwise we could have just used total_mesh_area = facet_area.sum().
106 // In the new version of eigen, we can also use facet_area( active_facets ).sum()
107 Scalar total_mesh_area = 0;
108 for (auto facet_id : range_sparse(mesh.get_num_facets(), active_facets)) {
109 total_mesh_area += facet_area(facet_id);
110 }
111
112 // total area of zero is bad
113 la_runtime_assert(total_mesh_area != 0.0);
114
115 // Estimate the sampling length
116 // If each point is a disk, what should be the radius of disks to cover the surface
117 // this radius would be the sampling length
118 // 1.5 is probably a safety factor
119 const Scalar sampling_length = 1.5 * sqrt((total_mesh_area / (approx_num_points * M_PI)));
120
121 // Find the bounding box of the active facets on the mesh
122 BoundingBox bounding_box(n_dims);
123 for (auto facet_id : range_sparse(mesh.get_num_facets(), active_facets)) {
124 for (auto vertex_offset : range(mesh.get_vertex_per_facet())) {
125 const Index vertex_id = facets(facet_id, vertex_offset);
126 // bbox needs a column vector . Hence the transpose(). Arg...
127 bounding_box.extend(vertices.row(vertex_id).transpose());
128 }
129 }
130
131 // Find the side lengths and dimensions of the grid used for sampling
132 const RowVectorS grid_lens = bounding_box.sizes();
133 const RowVectorI grid_dims =
134 (grid_lens / sampling_length).array().ceil().template cast<Index>();
135
136 // Number of cells in the grid
137 const Index num_grid_cells = grid_dims.prod();
138
139 // Given a 2(3)-D index gives the flattened index of a cell in the grid
140 auto get_flattend_index = [grid_dims, n_dims](const RowVectorI& index_nd) -> Index {
141 // Some debug time assertions
142 assert(index_nd.minCoeff() >= 0);
143 assert((index_nd.array() < grid_dims.array()).all());
144 //
145 Index answer = invalid<Index>();
146 //
147 if (n_dims == 2) {
148 answer = index_nd(1) * grid_dims(0) + index_nd(0);
149 } else if (n_dims == 3) {
150 answer = index_nd(2) * grid_dims(1) * grid_dims(0) + index_nd(1) * grid_dims(0) +
151 index_nd(0);
152 } else {
153 la_runtime_assert(0, "This dimension is not supported");
154 }
155 //
156 assert(answer >= 0 && answer < grid_dims.prod());
157 //
158 return answer;
159 };
160
161 // A data structure to indicate if we have already sampled from a cell of the grid
162 // A vector, this can be too memory consuming
163 // std::vector<bool> is_grid_cell_marked_backend(num_grid_cells, false);
164 // Hash map
165 std::unordered_set<Index> is_grid_cell_marked_backend(num_grid_cells);
166 //
167 // A labmda to see if a grid cell is already marked.
168 auto is_grid_cell_marked = [&is_grid_cell_marked_backend](const Index cell_index) {
169 // for a vector
170 // return is_grid_cell_marked_backend[cell_index];
171 // for the hashmap
172 return is_grid_cell_marked_backend.count(cell_index) != 0;
173 };
174 //
175 // A lambda to mark a grid cell
176 auto mark_grid_cell = [&is_grid_cell_marked_backend](const Index cell_index) {
177 // for a vector
178 // is_grid_cell_marked_backend[cell_index] = true;
179 // for hashmap
180 is_grid_cell_marked_backend.insert(cell_index);
181 };
182
183
184 // Hold the info about the sample points in a list
185 // once the process is finished move them to the output struct
186 std::list<VertexType> sample_positions;
187 std::list<Index> sample_facet_ids;
188 std::list<RowVector3S> sample_barycentrics;
189 Index num_samples = 0;
190
191 //
192 // A queue holding the triangles generated by splitting.
193 // Each triangle is represented by a 3x6 vector.
194 // Each row represents a vertex.
195 // The first three(two) columns represent the positions in the xy(z) space.
196 // The second three columns are the barycentric coordinates of the vertices w.r.t to the
197 // mother triangle (the non splitted one).
198 //
199 const Index num_triangle_cols = 3 + n_dims;
200 using Triangle = Eigen::Matrix<Scalar, 3, Eigen::Dynamic>;
201 auto get_triangle_pos = [&n_dims](const Triangle& tri, const Index vert_offset) -> RowVectorS {
202 return tri.row(vert_offset).head(n_dims);
203 };
204 auto get_triangle_bary = [](const Triangle& tri, const Index vert_offset) -> RowVectorS {
205 return tri.row(vert_offset).tail(3);
206 };
207 auto set_triangle_pos =
208 [&n_dims](Triangle& tri, const Index vert_offset, const RowVectorS& v) -> void {
209 tri.row(vert_offset).head(n_dims) = v;
210 };
211 auto set_triangle_bary = [](Triangle& tri,
212 const Index vert_offset,
213 const RowVectorS& v) -> void { tri.row(vert_offset).tail(3) = v; };
214 std::queue<Triangle> sub_triangles;
215
216 //
217 // Loop over facets
218 // subdivide each facet to the point that you get a triangle smaller
219 // than sampling length. Then get the midpoint of that triangle if
220 // its corresponding grid cell is not marked already.
221 //
222 for (auto facet_id : range_sparse(mesh.get_num_facets(), active_facets)) {
223 //
224 // Splitting code follows.
225 // This is done rather inefficiently using a queue. There is no need for that.
226 // If a samaritan with free time was passing by, they can make it better :).
227 //
228
229 // Empty out the queue
230 sub_triangles = std::queue<Triangle>();
231
232 // Create the mother triangle
233 Triangle mother_facet = Triangle::Zero(3, num_triangle_cols);
234 set_triangle_pos(mother_facet, 0, vertices.row(facets(facet_id, 0)));
235 set_triangle_pos(mother_facet, 1, vertices.row(facets(facet_id, 1)));
236 set_triangle_pos(mother_facet, 2, vertices.row(facets(facet_id, 2)));
237 set_triangle_bary(mother_facet, 0, RowVector3S(1, 0, 0));
238 set_triangle_bary(mother_facet, 1, RowVector3S(0, 1, 0));
239 set_triangle_bary(mother_facet, 2, RowVector3S(0, 0, 1));
240
241 // Push it to the queue
242 sub_triangles.push(mother_facet);
243
244 // Now process the triagles in the queue
245 while (!sub_triangles.empty()) {
246 const Triangle current_facet = sub_triangles.front();
247 sub_triangles.pop();
248
249 // Find the longest edge in the triangle
250 Scalar longest_edge_length = -std::numeric_limits<Scalar>::max();
251 Index longest_edge_offset = invalid<Index>();
252 for (auto edge_offset : range(3)) {
253 const auto edge_offset_p1 = (edge_offset + 1) % 3;
254 const Scalar edge_length = (get_triangle_pos(current_facet, edge_offset) //
255 - get_triangle_pos(current_facet, edge_offset_p1))
256 .norm();
257 if (edge_length > longest_edge_length) {
258 longest_edge_length = edge_length;
259 longest_edge_offset = edge_offset;
260 }
261 }
262
263 // If too big, subdivide
264 if (longest_edge_length > 2 * sampling_length) {
265 Triangle f1 = Triangle::Zero(3, num_triangle_cols);
266 Triangle f2 = Triangle::Zero(3, num_triangle_cols);
267 //
268 f1.row(0) = (current_facet.row(longest_edge_offset) +
269 current_facet.row((longest_edge_offset + 1) % 3)) /
270 2;
271 f1.row(1) = current_facet.row(longest_edge_offset);
272 f1.row(2) = current_facet.row((longest_edge_offset + 2) % 3);
273 //
274 f2.row(0) = f1.row(0);
275 f2.row(1) = current_facet.row((longest_edge_offset + 1) % 3);
276 f2.row(2) = current_facet.row((longest_edge_offset + 2) % 3);
277 //
278 sub_triangles.push(f1);
279 sub_triangles.push(f2);
280 } else { // Otherwise, sample the centroid
281 const RowVector3S point_barycentric =
282 (get_triangle_bary(current_facet, 0) + get_triangle_bary(current_facet, 1) +
283 get_triangle_bary(current_facet, 2)) /
284 3.;
285 const VertexType point_position =
286 (get_triangle_pos(current_facet, 0) + get_triangle_pos(current_facet, 1) +
287 get_triangle_pos(current_facet, 2)) /
288 3.;
289
290
291 // Get the corresponding cell in the grid
292 // min() returns column vector, hence the tranpose(), arg...
293 const RowVectorI grid_cell =
294 ((point_position - bounding_box.min().transpose()) / sampling_length)
295 .array()
296 .floor()
297 .template cast<Index>();
298 const Index grid_cell_id = get_flattend_index(grid_cell);
299
300 if (!is_grid_cell_marked(grid_cell_id)) {
301 mark_grid_cell(grid_cell_id);
302 // record sample facet
303 sample_facet_ids.push_back(facet_id);
304 // record barycetric (in mother facet)
305 sample_barycentrics.push_back(point_barycentric);
306 // record the position
307 sample_positions.push_back(point_position);
308 //
309 ++num_samples;
310 } // end of grid check
311 } // end of size check
312 } // end of loop over queue
313 } // end of loop over faces
314
315
316 // Now copy the result into the output struct
317 Output output;
318 {
319 output.num_samples = num_samples;
320 //
321 output.barycentrics.resize(num_samples, 3);
322 output.facet_ids.resize(num_samples);
323 output.positions.resize(num_samples, n_dims);
324 //
325 auto it_barycentric = sample_barycentrics.begin();
326 auto it_facet_id = sample_facet_ids.begin();
327 auto it_position = sample_positions.begin();
328 //
329 for (Index i : range(num_samples)) {
330 output.barycentrics.row(i) = *it_barycentric;
331 output.facet_ids[i] = *it_facet_id;
332 output.positions.row(i) = *it_position;
333 ++it_barycentric;
334 ++it_facet_id;
335 ++it_position;
336 }
337 }
338
339 return output;
340}

342// Active facets are specified by an array of bools
343// Inputs
344// - The mesh
345// - Approximate number of points to be sampled
346// - Array of bools saying whether a facet should be sampled from or not.
347template <typename MeshType>
348SamplePointsOnSurfaceOutput<MeshType> sample_points_on_surface(
349 MeshType& mesh,
350 const typename MeshType::Index approx_num_points,
351 const std::vector<bool>& is_facet_active)
352{
353 typename MeshType::IndexList active_facets;
354 for (auto i : range(safe_cast<typename MeshType::Index>(is_facet_active.size()))) {
355 if (is_facet_active[i]) {
356 active_facets.push_back(i);
357 }
358 }
359 return sample_points_on_surface(mesh, approx_num_points, active_facets);
360}

362// All the facets will be sampled
363// Inputs
364// - The mesh
365// - Approximate number of points to be sampled
366template <typename MeshType>
367SamplePointsOnSurfaceOutput<MeshType> sample_points_on_surface(
368 MeshType& mesh,
369 const typename MeshType::Index approx_num_points)
370{
371 return sample_points_on_surface(mesh, approx_num_points, typename MeshType::IndexList());
372}
