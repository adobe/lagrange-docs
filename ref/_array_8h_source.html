39public:
40 template <typename TargetType>
41 Eigen::Map<TargetType> view()
42 {

56

58 void set(Eigen::MatrixBase<Derived>&& data);

60 ScalarEnum get_scalar_type() const { return m_scalar_type; }
61
62 template <typename TargetType>
63 const auto& get() const;
64
65 template <typename TargetType>
66 auto& get();
67
68 template <typename Scalar>
69 Scalar* data()
70 {
71 la_runtime_assert(ScalarToEnum_v<Scalar> == m_scalar_type);
72 return static_cast<Scalar*>(data());
73 }

84 virtual bool is_row_major() const = 0;

96 {
97 return row_slice(safe_cast<Index>(row_indices.size()), [&](Index i) {

158template <typename _EigenType>
159class EigenArray : public ArrayBase
160{
161public:
162 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
163 using EigenType = _EigenType;
164 using Index = ArrayBase::Index;
165 using ArrayBase::WeightedIndexFunction;
166 static_assert(
167 std::is_base_of<typename Eigen::EigenBase<EigenType>, EigenType>::value,
168 "Template parameter `_EigenType` is not an Eigen type!");
169 static_assert(!std::is_const<EigenType>::value, "EigenType should not be const type.");
170 static_assert(!std::is_reference<EigenType>::value, "EigenType should not be reference type.");
171
172public:
173 EigenArray()
174 : ArrayBase(ScalarToEnum_v<typename EigenType::Scalar>)
175 {}
176
177 template <typename T>
178 explicit EigenArray(const Eigen::MatrixBase<T>& data)
179 : ArrayBase(ScalarToEnum_v<typename EigenType::Scalar>)
180 , m_data(data)
181 {}
182
183 template <typename T>
184 explicit EigenArray(Eigen::MatrixBase<T>&& data)
185 : ArrayBase(ScalarToEnum_v<typename EigenType::Scalar>)
186 , m_data(std::move(data.derived()))
187 {}
188
189public:
190 EigenType& get_ref() { return m_data; }
191 const EigenType& get_ref() const { return m_data; }
192
193 template <typename Derived>
194 void set(Derived&& data)
195 {
196 m_data = std::forward<Derived>(data);
197 }
198
199 Index rows() const override { return m_data.rows(); }
200 Index cols() const override { return m_data.cols(); }
201 bool is_row_major() const override { return EigenType::IsRowMajor != 0; }
202 void* data() override { return static_cast<void*>(m_data.derived().data()); }
203 const void* data() const override { return static_cast<const void*>(m_data.derived().data()); }
204 void resize(Index r, Index c) override { m_data.resize(r, c); }
205
206 std::unique_ptr<ArrayBase> clone() const override
207 {
208 return std::make_unique<EigenArray<EigenType>>(m_data);
209 }
210
211 std::unique_ptr<ArrayBase> row_slice(Index num_rows, const IndexFunction& mapping_fn)
212 const override
213 {
214 return row_slice_impl(m_data, num_rows, mapping_fn);
215 }
216 std::unique_ptr<ArrayBase> row_slice(Index num_rows, const WeightedIndexFunction& mapping_fn)
217 const override
218 {
219 return row_slice_impl(m_data, num_rows, mapping_fn);
220 }
221
222 std::string type_name() const override
223 {
224 std::string scalar_name(ScalarToEnum<typename EigenType::Scalar>::name);
225 return string_format(
226 "EigenArray<Eigen::Matrix<{}, {}, {}, {}>>",
227 scalar_name,
228 EigenType::RowsAtCompileTime,
229 EigenType::ColsAtCompileTime,
230 EigenType::Options);
231 }
232
233private:
234 EigenType m_data;
235};
