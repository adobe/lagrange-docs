34template <typename DerivedF>
35std::pair<std::unique_ptr<index_t[]>, std::unique_ptr<index_t[]>> convert_index_buffer(
36 const Eigen::MatrixBase<DerivedF> &facets)
37{
38 const auto num_elems = safe_cast<index_t>(facets.rows());
39 const auto elem_size = safe_cast<index_t>(facets.cols());
40
41 auto e_ptr = std::unique_ptr<index_t[]>(new index_t[num_elems + 1]);
42 auto e_ind = std::unique_ptr<index_t[]>(new index_t[num_elems * elem_size]);
43 for (index_t f = 0; f < num_elems; ++f) {
44 e_ptr[f] = f * elem_size;
45 for (index_t lv = 0; lv < elem_size; ++lv) {
46 e_ind[f * elem_size + lv] = facets(f, lv);
47 }
48 }
49 e_ptr[num_elems] = num_elems * elem_size;
50 return std::make_pair(std::move(e_ptr), std::move(e_ind));
51}
