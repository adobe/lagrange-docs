39template <typename MeshType>
40std::unique_ptr<MeshType> remove_duplicate_facets(const MeshType& mesh)
41{
42 static_assert(MeshTrait<MeshType>::is_mesh(), "Input type is not Mesh");
43 using Index = typename MeshType::FacetArray::Scalar;
44 using Facet = std::array<Index, 3>;
45
46 const Index num_facets = mesh.get_num_facets();
47 const auto& facets = mesh.get_facets();
48
49 auto facet_hash = [](const Facet& f) { return f[0] + f[1] + f[2]; };
50 auto facet_eq = [](const Facet& f1, const Facet& f2) {
51 return f1[0] == f2[0] && f1[1] == f2[1] && f1[2] == f2[2];
52 };
53
54 std::unordered_map<Facet, std::list<Index>, decltype(facet_hash), decltype(facet_eq)> facet_map(
55 num_facets,
56 facet_hash,
57 facet_eq);
58 for (Index i = 0; i < num_facets; i++) {
59 Facet f{facets(i, 0), facets(i, 1), facets(i, 2)};
60 std::sort(f.begin(), f.end());
61 auto itr = facet_map.find(f);
62 if (itr == facet_map.end()) {
63 facet_map.insert({f, {i}});
64 } else {
65 itr->second.push_back(i);
66 }
67 }
68
69 const auto num_unique_facets = facet_map.size();
70 typename MeshType::FacetArray unique_facets(num_unique_facets, 3);
71 std::vector<Index> ori_facet_indices(num_unique_facets);
72 Index count = 0;
73 for (const auto& entry : facet_map) {
74 const auto& fs = entry.second;
75 assert(fs.size() > 0);
76 auto fid = fs.front();
77 if (fs.size() > 1) {
78 // Orientation votes. Use the orientation with most votes.
79 int orientation = 0;
80 Index inverted_fid = fid;
81 for (const auto fj : fs) {
82 if ((facets(fj, 0) == facets(fid, 0) && facets(fj, 1) == facets(fid, 1) &&
83 facets(fj, 2) == facets(fid, 2)) ||
84 (facets(fj, 0) == facets(fid, 1) && facets(fj, 1) == facets(fid, 2) &&
85 facets(fj, 2) == facets(fid, 0)) ||
86 (facets(fj, 0) == facets(fid, 2) && facets(fj, 1) == facets(fid, 0) &&
87 facets(fj, 2) == facets(fid, 1))) {
88 orientation += 1;
89 } else {
90 orientation -= 1;
91 inverted_fid = fid;
92 }
93 }
94 if (orientation < 0) {
95 fid = inverted_fid;
96 }
97 }
98 unique_facets.row(count) = facets.row(fid);
99 ori_facet_indices[count] = fid;
100 count++;
101 }
102
103 const auto vertices = mesh.get_vertices();
104 auto out_mesh = create_mesh(std::move(vertices), std::move(unique_facets));
105
106 map_attributes(mesh, *out_mesh, {}, ori_facet_indices);
107
108 return out_mesh;
109}
