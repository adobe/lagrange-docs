22template <typename T, std::size_t MaxSize = 8, typename NonReboundT = T>
23struct SmallBufferAllocator
24{
25 alignas(alignof(T)) char m_small_buffer[MaxSize * sizeof(T)];
26 std::allocator<T> m_alloc{};
27 bool m_small_buffer_used = false;
28
29 using value_type = T;
30 // we have to set these three values, as they are responsible for the correct handling of the
31 // move assignment operator
32 using propagate_on_container_move_assignment = std::false_type;
33 using propagate_on_container_swap = std::false_type;
34 using is_always_equal = std::false_type;
35
36 template <class U>
37 struct rebind
38 {
39 typedef SmallBufferAllocator<U, MaxSize, NonReboundT> other;
40 };
41
42 constexpr SmallBufferAllocator() noexcept = default;
43 template <class U>
44 constexpr SmallBufferAllocator(const SmallBufferAllocator<U, MaxSize, NonReboundT>&) noexcept
45 {}
46 // don't copy the small buffer for the copy/move constructors, as the copying is done through
47 // the vector
48 constexpr SmallBufferAllocator(const SmallBufferAllocator& other) noexcept
49 : m_small_buffer_used(other.m_small_buffer_used)
50 {}
51 constexpr SmallBufferAllocator& operator=(const SmallBufferAllocator& other) noexcept
52 {
53 m_small_buffer_used = other.m_small_buffer_used;
54 return *this;
55 }
56 constexpr SmallBufferAllocator(SmallBufferAllocator&&) noexcept {}
57 constexpr SmallBufferAllocator& operator=(const SmallBufferAllocator&&) noexcept
58 {
59 return *this;
60 }
61
62 constexpr T* allocate(const std::size_t n)
63 {
64 // when the allocator was rebound we don't want to use the small buffer
65 if constexpr (std::is_same<T, NonReboundT>::value) {
66 if (n <= MaxSize) {
67 m_small_buffer_used = true;
68 // as long as we use less memory than the small buffer, we return a pointer to it
69 return reinterpret_cast<T*>(&m_small_buffer);
70 }
71 }
72 m_small_buffer_used = false;
73 // otherwise use the default allocator
74 return m_alloc.allocate(n);
75 }
76 constexpr void deallocate(void* p, const std::size_t n)
77 {
78 // we don't deallocate anything if the memory was allocated in small buffer
79 if (&m_small_buffer != p) m_alloc.deallocate(static_cast<T*>(p), n);
80 m_small_buffer_used = false;
81 }
82 // according to the C++ standard when propagate_on_container_move_assignment is set to false,
83 // the comparison operators are used to check if two allocators are equal. When they are not, an
84 // element wise move is done instead of just taking over the memory. For our implementation this
85 // means the comparision has to return false, when the small buffer is active
86 friend constexpr bool operator==(
87 const SmallBufferAllocator& lhs,
88 const SmallBufferAllocator& rhs)
89 {
90 return !lhs.m_small_buffer_used && !rhs.m_small_buffer_used;
91 }
92 friend constexpr bool operator!=(
93 const SmallBufferAllocator& lhs,
94 const SmallBufferAllocator& rhs)
95 {
96 return !(lhs == rhs);
97 }
98};

109template <typename T, std::size_t N = 8>
110class SmallVector : public std::vector<T, SmallBufferAllocator<T, N>>
111{
112private:
113 using vectorT = std::vector<T, SmallBufferAllocator<T, N>>;
114
115public:
116 // default initialize with the small buffer size
117 constexpr SmallVector() noexcept { vectorT::reserve(N); }
118 SmallVector(const SmallVector&) = default;
119 SmallVector& operator=(const SmallVector&) = default;
120 SmallVector(SmallVector&& other) noexcept(std::is_nothrow_move_constructible<T>::value)
121 {
122 if (other.size() <= N) vectorT::reserve(N);
123 vectorT::operator=(std::move(other));
124 }
125 SmallVector& operator=(SmallVector&& other) noexcept(
126 std::is_nothrow_move_constructible<T>::value)
127 {
128 if (other.size() <= N) vectorT::reserve(N);
129 vectorT::operator=(std::move(other));
130 return *this;
131 }
132 // use the default constructor first to reserve then construct the values
133 explicit SmallVector(std::size_t count)
134 : SmallVector()
135 {
136 vectorT::resize(count);
137 }
138 SmallVector(std::size_t count, const T& value)
139 : SmallVector()
140 {
141 vectorT::assign(count, value);
142 }
143 template <class InputIt>
144 SmallVector(InputIt first, InputIt last)
145 : SmallVector()
146 {
147 vectorT::insert(vectorT::begin(), first, last);
148 }
149 SmallVector(std::initializer_list<T> init)
150 : SmallVector()
151 {
152 vectorT::insert(vectorT::begin(), init);
153 }
154 friend void swap(SmallVector& a, SmallVector& b) noexcept
155 {
156 using std::swap;
157 swap(static_cast<vectorT&>(a), static_cast<vectorT&>(b));
158 }
159};
