92 EmbreeRayCaster(
93 RTCSceneFlags scene_flags = RTC_SCENE_FLAG_DYNAMIC,
94 RTCBuildQuality build_quality = RTC_BUILD_QUALITY_LOW)
95 {
96// Embree strongly recommend to have the Flush to Zero and
97// Denormals are Zero mode of the MXCSR control and status
98// register enabled for each thread before calling the
99// rtcIntersect and rtcOccluded functions.
100#ifdef _MM_SET_FLUSH_ZERO_MODE
101 _MM_SET_FLUSH_ZERO_MODE(_MM_FLUSH_ZERO_ON);
102#endif
103#ifdef _MM_SET_DENORMALS_ZERO_MODE
104 _MM_SET_DENORMALS_ZERO_MODE(_MM_DENORMALS_ZERO_ON);
105#endif
106
107 m_scene_flags = scene_flags;
108 m_build_quality = build_quality;
109 m_device = rtcNewDevice(NULL);
110 ensure_no_errors_internal();
111 m_embree_world_scene = rtcNewScene(m_device);
112 ensure_no_errors_internal();
113 m_instance_index_ranges.push_back(safe_cast<Index>(0));
114
115 m_need_rebuild = true;
116 m_need_commit = false;
117 }

186 template <typename MeshType>
187 Index add_mesh(
188 std::shared_ptr<MeshType> mesh,
189 const Transform& trans = Transform::Identity(),
190 RTCBuildQuality build_quality = RTC_BUILD_QUALITY_MEDIUM)
191 {
192 m_meshes.push_back(std::move(std::make_unique<RaycasterMeshDerived<MeshType>>(mesh)));
193 m_transforms.push_back(trans);
194 m_mesh_build_qualities.push_back(build_quality);
195 m_visibility.push_back(true);
196 for (auto& f : m_filters) { // per-mesh, not per-instance
197 f.push_back(nullptr);
198 }
199 Index mesh_index = safe_cast<Index>(m_meshes.size() - 1);
200 la_runtime_assert(m_instance_index_ranges.size() > 0);
201 Index instance_index = m_instance_index_ranges.back();
202 la_runtime_assert(instance_index == safe_cast<Index>(m_instance_to_user_mesh.size()));
203 m_instance_index_ranges.push_back(instance_index + 1);
204 m_instance_to_user_mesh.resize(instance_index + 1, mesh_index);
205 m_need_rebuild = true;
206 return mesh_index;
207 }

215 template <typename MeshType>
216 Index add_meshes(
217 std::shared_ptr<MeshType> mesh,
218 const TransformVector& trans_vector,
219 RTCBuildQuality build_quality = RTC_BUILD_QUALITY_MEDIUM)
220 {
221 m_meshes.push_back(std::move(std::make_unique<RaycasterMeshDerived<MeshType>>(mesh)));
222 m_transforms.insert(m_transforms.end(), trans_vector.begin(), trans_vector.end());
223 m_mesh_build_qualities.push_back(build_quality);
224 m_visibility.resize(m_visibility.size() + trans_vector.size(), true);
225 for (auto& f : m_filters) { // per-mesh, not per-instance
226 f.push_back(nullptr);
227 }
228 Index mesh_index = safe_cast<Index>(m_meshes.size() - 1);
229 la_runtime_assert(m_instance_index_ranges.size() > 0);
230 Index instance_index = m_instance_index_ranges.back();
231 la_runtime_assert(instance_index == safe_cast<Index>(m_instance_to_user_mesh.size()));
232 Index new_instance_size = instance_index + trans_vector.size();
233 m_instance_index_ranges.push_back(new_instance_size);
234 m_instance_to_user_mesh.resize(new_instance_size, mesh_index);
235 m_need_rebuild = true;
236 return mesh_index;
237 }

245 template <typename MeshType>
246 void update_mesh(
247 Index index,
248 std::shared_ptr<MeshType> mesh,
249 RTCBuildQuality build_quality = RTC_BUILD_QUALITY_MEDIUM)
250 {
251 la_runtime_assert(mesh->get_dim() == 3);
252 la_runtime_assert(mesh->get_vertex_per_facet() == 3);
253 la_runtime_assert(index < safe_cast<Index>(m_meshes.size()));
254 m_meshes[index] = std::move(std::make_unique<RaycasterMeshDerived<MeshType>>(mesh));
255 m_mesh_build_qualities[index] = build_quality;
256 m_need_rebuild = true; // TODO: Make this more fine-grained so only the affected part of
257 // the Embree scene is updated
258 }
259
265 void update_mesh_vertices(Index index)
266 {
267 la_runtime_assert(index < safe_cast<Index>(m_meshes.size()));
268 if (m_need_rebuild) return;
269
270 la_runtime_assert(index < safe_cast<Index>(m_embree_mesh_scenes.size()));
271 auto geom = rtcGetGeometry(m_embree_mesh_scenes[index], 0);
272
273 // Update the vertex buffer in Embree
274 auto const& mesh = m_meshes[index];
275 la_runtime_assert(
276 safe_cast<Index>(mesh->get_num_vertices()) == m_mesh_vertex_counts[index]);
277
278 auto vbuf =
279 reinterpret_cast<float*>(rtcGetGeometryBufferData(geom, RTC_BUFFER_TYPE_VERTEX, 0));
280 la_runtime_assert(vbuf);
281 mesh->vertices_to_float(vbuf);
282 rtcUpdateGeometryBuffer(geom, RTC_BUFFER_TYPE_VERTEX, 0);
283
284 // Re-commit the mesh geometry and scene
285 rtcCommitGeometry(geom);
286 rtcCommitScene(m_embree_mesh_scenes[index]);
287
288 // Re-commit every instance of the mesh
289 for (Index instance_index = m_instance_index_ranges[index];
290 instance_index < m_instance_index_ranges[index + 1];
291 ++instance_index) {
292 Index rtc_inst_id = m_instance_index_ranges[index] + instance_index;
293 auto geom_inst =
294 rtcGetGeometry(m_embree_world_scene, static_cast<unsigned>(rtc_inst_id));
295 rtcCommitGeometry(geom_inst);
296 }
297
298 // Mark the world scene as needing a re-commit (will be called lazily)
299 m_need_commit = true;
300 }

313 void update_transformation(Index mesh_index, Index instance_index, const Transform& trans)
314 {
315 la_runtime_assert(mesh_index + 1 < safe_cast<Index>(m_instance_index_ranges.size()));
316 Index index = m_instance_index_ranges[mesh_index] + instance_index;
317 la_runtime_assert(index < m_instance_index_ranges[mesh_index + 1]);
318 la_runtime_assert(index < safe_cast<Index>(m_transforms.size()));
319 m_transforms[index] = trans;
320 if (!m_need_rebuild) {
321 auto geom = rtcGetGeometry(m_embree_world_scene, static_cast<unsigned>(index));
322 Eigen::Matrix<float, 4, 4> T = trans.template cast<float>();
323 rtcSetGeometryTransform(geom, 0, RTC_FORMAT_FLOAT4X4_COLUMN_MAJOR, T.eval().data());
324 rtcCommitGeometry(geom);
325 m_need_commit = true;
326 }

354 m_need_commit = true;

453 uint32_t cast4(
454 uint32_t batch_size,
455 const Point4& origin,
456 const Direction4& direction,
457 const Mask4& mask,
458 Index4& mesh_index,
459 Index4& instance_index,
460 Index4& facet_index,
461 Scalar4& ray_depth,
462 Point4& barycentric_coord,
463 Point4& normal,
464 const Scalar4& tmin = Scalar4::Zero(),
465 const Scalar4& tmax = Scalar4::Constant(std::numeric_limits<Scalar>::infinity()))
466 {
467 la_debug_assert(batch_size <= 4);
468
469 update_internal();
470
471 RTCRayHit4 embree_raypacket;
472 for (int i = 0; i < static_cast<int>(batch_size); ++i) {
473 // Set ray origins
474 embree_raypacket.ray.org_x[i] = static_cast<float>(origin(i, 0));
475 embree_raypacket.ray.org_y[i] = static_cast<float>(origin(i, 1));
476 embree_raypacket.ray.org_z[i] = static_cast<float>(origin(i, 2));
477
478 // Set ray directions
479 embree_raypacket.ray.dir_x[i] = static_cast<float>(direction(i, 0));
480 embree_raypacket.ray.dir_y[i] = static_cast<float>(direction(i, 1));
481 embree_raypacket.ray.dir_z[i] = static_cast<float>(direction(i, 2));
482
483 // Misc
484 embree_raypacket.ray.tnear[i] = static_cast<float>(tmin[i]);
485 embree_raypacket.ray.tfar[i] = std::isinf(tmax[i]) ? std::numeric_limits<float>::max()
486 : static_cast<float>(tmax[i]);
487 embree_raypacket.ray.mask[i] = 0xFFFFFFFF;
488 embree_raypacket.ray.id[i] = static_cast<unsigned>(i);
489 embree_raypacket.ray.flags[i] = 0;
490
491 // Required initialization of the hit substructure
492 embree_raypacket.hit.geomID[i] = RTC_INVALID_GEOMETRY_ID;
493 embree_raypacket.hit.primID[i] = RTC_INVALID_GEOMETRY_ID;
494 embree_raypacket.hit.instID[0][i] = RTC_INVALID_GEOMETRY_ID;
495 }
496
497 // Modify the mask to make 100% sure extra rays in the packet will be ignored
498 auto packet_mask = mask;
499 for (int i = static_cast<int>(batch_size); i < 4; ++i) packet_mask[i] = 0;
500
501 ensure_no_errors_internal();
502 {
503 RTCIntersectContext context;
504 rtcInitIntersectContext(&context);
505 rtcIntersect4(packet_mask.data(), m_embree_world_scene, &context, &embree_raypacket);
506 }
507 ensure_no_errors_internal();
508
509 uint32_t is_hits = 0;
510 for (int i = 0; i < static_cast<int>(batch_size); ++i) {
511 if (embree_raypacket.hit.geomID[i] != RTC_INVALID_GEOMETRY_ID) {
512 Index rtc_inst_id = embree_raypacket.hit.instID[0][i];
513 Index rtc_mesh_id = (rtc_inst_id == RTC_INVALID_GEOMETRY_ID)
514 ? embree_raypacket.hit.geomID[i]
515 : rtc_inst_id;
516 assert(rtc_mesh_id < m_instance_to_user_mesh.size());
517 assert(m_visibility[rtc_mesh_id]);
518 mesh_index[i] = m_instance_to_user_mesh[rtc_mesh_id];
519 assert(mesh_index[i] + 1 < m_instance_index_ranges.size());
520 assert(mesh_index[i] < safe_cast<Index>(m_meshes.size()));
521 instance_index[i] = rtc_mesh_id - m_instance_index_ranges[mesh_index[i]];
522 facet_index[i] = embree_raypacket.hit.primID[i];
523 ray_depth[i] = embree_raypacket.ray.tfar[i];
524 barycentric_coord(i, 0) =
525 1.0f - embree_raypacket.hit.u[i] - embree_raypacket.hit.v[i];
526 barycentric_coord(i, 1) = embree_raypacket.hit.u[i];
527 barycentric_coord(i, 2) = embree_raypacket.hit.v[i];
528 normal(i, 0) = embree_raypacket.hit.Ng_x[i];
529 normal(i, 1) = embree_raypacket.hit.Ng_y[i];
530 normal(i, 2) = embree_raypacket.hit.Ng_z[i];
531 is_hits = is_hits | (1 << i);
532 }
533 }
534
535 return is_hits;
536 }

574 uint32_t cast4(
575 uint32_t batch_size,
576 const Point4& origin,
577 const Direction4& direction,
578 const Mask4& mask,
579 const Scalar4& tmin = Scalar4::Zero(),
580 const Scalar4& tmax = Scalar4::Constant(std::numeric_limits<Scalar>::infinity()))
581 {
582 la_debug_assert(batch_size <= 4);
583
584 update_internal();
585
586 RTCRay4 embree_raypacket;
587 for (int i = 0; i < static_cast<int>(batch_size); ++i) {
588 // Set ray origins
589 embree_raypacket.org_x[i] = static_cast<float>(origin(i, 0));
590 embree_raypacket.org_y[i] = static_cast<float>(origin(i, 1));
591 embree_raypacket.org_z[i] = static_cast<float>(origin(i, 2));
592
593 // Set ray directions
594 embree_raypacket.dir_x[i] = static_cast<float>(direction(i, 0));
595 embree_raypacket.dir_y[i] = static_cast<float>(direction(i, 1));
596 embree_raypacket.dir_z[i] = static_cast<float>(direction(i, 2));
597
598 // Misc
599 embree_raypacket.tnear[i] = static_cast<float>(tmin[i]);
600 embree_raypacket.tfar[i] = std::isinf(tmax[i]) ? std::numeric_limits<float>::max()
601 : static_cast<float>(tmax[i]);
602 embree_raypacket.mask[i] = 0xFFFFFFFF;
603 embree_raypacket.id[i] = static_cast<unsigned>(i);
604 embree_raypacket.flags[i] = 0;
605 }
606
607 // Modify the mask to make 100% sure extra rays in the packet will be ignored
608 auto packet_mask = mask;
609 for (int i = static_cast<int>(batch_size); i < 4; ++i) packet_mask[i] = 0;
610
611 ensure_no_errors_internal();
612 {
613 RTCIntersectContext context;
614 rtcInitIntersectContext(&context);
615 rtcOccluded4(packet_mask.data(), m_embree_world_scene, &context, &embree_raypacket);
616 }
617 ensure_no_errors_internal();
618
619 // If hit, the tfar field will be set to -inf.
620 uint32_t is_hits = 0;
621 for (uint32_t i = 0; i < batch_size; ++i)
622 if (!std::isfinite(embree_raypacket.tfar[i])) is_hits = is_hits | (1 << i);
623
624 return is_hits;
625 }

631 bool cast(
632 const Point& origin,
633 const Direction& direction,
634 Index& mesh_index,
635 Index& instance_index,
636 Index& facet_index,
637 Scalar& ray_depth,
638 Point& barycentric_coord,
639 Point& normal,
640 Scalar tmin = 0,
641 Scalar tmax = std::numeric_limits<Scalar>::infinity())
642 {
643 // Overloaded when specializing tnear and tfar
644
645 update_internal();
646
647 RTCRayHit embree_rayhit;
648 embree_rayhit.ray.org_x = static_cast<float>(origin.x());
649 embree_rayhit.ray.org_y = static_cast<float>(origin.y());
650 embree_rayhit.ray.org_z = static_cast<float>(origin.z());
651 embree_rayhit.ray.dir_x = static_cast<float>(direction.x());
652 embree_rayhit.ray.dir_y = static_cast<float>(direction.y());
653 embree_rayhit.ray.dir_z = static_cast<float>(direction.z());
654 embree_rayhit.ray.tnear = static_cast<float>(tmin);
655 embree_rayhit.ray.tfar =
656 std::isinf(tmax) ? std::numeric_limits<float>::max() : static_cast<float>(tmax);
657 embree_rayhit.hit.geomID = RTC_INVALID_GEOMETRY_ID;
658 embree_rayhit.hit.primID = RTC_INVALID_GEOMETRY_ID;
659 embree_rayhit.hit.instID[0] = RTC_INVALID_GEOMETRY_ID;
660 embree_rayhit.ray.mask = 0xFFFFFFFF;
661 embree_rayhit.ray.id = 0;
662 embree_rayhit.ray.flags = 0;
663 ensure_no_errors_internal();
664 {
665 RTCIntersectContext context;
666 rtcInitIntersectContext(&context);
667 rtcIntersect1(m_embree_world_scene, &context, &embree_rayhit);
668 }
669 ensure_no_errors_internal();
670
671 if (embree_rayhit.hit.geomID != RTC_INVALID_GEOMETRY_ID) {
672 Index rtc_inst_id = embree_rayhit.hit.instID[0];
673 Index rtc_mesh_id =
674 (rtc_inst_id == RTC_INVALID_GEOMETRY_ID) ? embree_rayhit.hit.geomID : rtc_inst_id;
675 assert(rtc_mesh_id < m_instance_to_user_mesh.size());
676 assert(m_visibility[rtc_mesh_id]);
677 mesh_index = m_instance_to_user_mesh[rtc_mesh_id];
678 assert(mesh_index + 1 < m_instance_index_ranges.size());
679 assert(mesh_index < safe_cast<Index>(m_meshes.size()));
680 instance_index = rtc_mesh_id - m_instance_index_ranges[mesh_index];
681 facet_index = embree_rayhit.hit.primID;
682 ray_depth = embree_rayhit.ray.tfar;
683 barycentric_coord[0] = 1.0f - embree_rayhit.hit.u - embree_rayhit.hit.v;
684 barycentric_coord[1] = embree_rayhit.hit.u;
685 barycentric_coord[2] = embree_rayhit.hit.v;
686 normal[0] = embree_rayhit.hit.Ng_x;
687 normal[1] = embree_rayhit.hit.Ng_y;
688 normal[2] = embree_rayhit.hit.Ng_z;
689 return true;
690 } else {
691 // Ray missed.
692 mesh_index = invalid<Index>();
693 instance_index = invalid<Index>();
694 facet_index = invalid<Index>();
695 return false;
696 }
697 }

739 embree_ray.org_y = static_cast<float>(origin.y());

741 embree_ray.dir_x = static_cast<float>(direction.x());
742 embree_ray.dir_y = static_cast<float>(direction.y());

747 embree_ray.mask = 0xFFFFFFFF;

749 embree_ray.flags = 0;

755 rtcOccluded1(m_embree_world_scene, &context, &embree_ray);

760 return !std::isfinite(embree_ray.tfar);

794 void generate_scene()
795 {
796 if (!m_need_rebuild) return;
797
798 // Scene needs to be updated
799 rtcReleaseScene(m_embree_world_scene);
800 m_embree_world_scene = rtcNewScene(m_device);
801 auto scene_flags = get_scene_flags(); // FIXME: or just m_scene_flags?
802 auto scene_build_quality = get_scene_build_quality(); // FIXME: or just m_build_quality?
803 rtcSetSceneFlags(
804 m_embree_world_scene,
805 scene_flags); // TODO: maybe also set RTC_SCENE_FLAG_CONTEXT_FILTER_FUNCTION
806 rtcSetSceneBuildQuality(m_embree_world_scene, scene_build_quality);
807 m_float_data.clear();
808 m_int_data.clear();
809 const auto num_meshes = m_meshes.size();
810 la_runtime_assert(num_meshes + 1 == m_instance_index_ranges.size());
811 m_embree_mesh_scenes.resize(num_meshes);
812 m_mesh_vertex_counts.resize(num_meshes);
813 ensure_no_errors_internal();
814
815 bool is_mask_supported =
816 rtcGetDeviceProperty(m_device, RTC_DEVICE_PROPERTY_RAY_MASK_SUPPORTED);
817 for (size_t i = 0; i < num_meshes; i++) {
818 // Initialize RTC meshes
819 const auto& mesh = m_meshes[i];
820 // const auto& vertices = mesh->get_vertices();
821 // const auto& facets = mesh->get_facets();
822 const Index num_vertices = m_mesh_vertex_counts[i] =
823 safe_cast<Index>(mesh->get_num_vertices());
824 const Index num_facets = safe_cast<Index>(mesh->get_num_facets());
825
826 auto& embree_mesh_scene = m_embree_mesh_scenes[i];
827 embree_mesh_scene = rtcNewScene(m_device);
828
829 rtcSetSceneFlags(embree_mesh_scene, scene_flags);
830 rtcSetSceneBuildQuality(embree_mesh_scene, scene_build_quality);
831 ensure_no_errors_internal();
832
833 RTCGeometry geom = rtcNewGeometry(
834 m_device,
835 RTC_GEOMETRY_TYPE_TRIANGLE); // EMBREE_FIXME: check if geometry gets properly
836 // committed
837 rtcSetGeometryBuildQuality(geom, m_mesh_build_qualities[i]);
838 // rtcSetGeometryTimeStepCount(geom, 1);
839
840 const float* vertex_data = extract_float_data(*mesh);
841 const unsigned* facet_data = extract_int_data(*mesh);
842
843 rtcSetSharedGeometryBuffer(
844 geom,
845 RTC_BUFFER_TYPE_VERTEX,
846 0,
847 RTC_FORMAT_FLOAT3,
848 vertex_data,
849 0,
850 sizeof(float) * 3,
851 num_vertices);
852 rtcSetSharedGeometryBuffer(
853 geom,
854 RTC_BUFFER_TYPE_INDEX,
855 0,
856 RTC_FORMAT_UINT3,
857 facet_data,
858 0,
859 sizeof(int) * 3,
860 num_facets);
861
862 set_intersection_filter(geom, m_filters[FILTER_INTERSECT][i], is_mask_supported);
863 set_occlusion_filter(geom, m_filters[FILTER_OCCLUDED][i], is_mask_supported);
864
865 rtcCommitGeometry(geom);
866 rtcAttachGeometry(embree_mesh_scene, geom);
867 rtcReleaseGeometry(geom);
868 ensure_no_errors_internal();
869
870 // Initialize RTC instances
871 for (Index instance_index = m_instance_index_ranges[i];
872 instance_index < m_instance_index_ranges[i + 1];
873 ++instance_index) {
874 const auto& trans = m_transforms[instance_index];
875
876 RTCGeometry geom_inst = rtcNewGeometry(
877 m_device,
878 RTC_GEOMETRY_TYPE_INSTANCE); // EMBREE_FIXME: check if geometry gets properly
879 // committed
880 rtcSetGeometryInstancedScene(geom_inst, embree_mesh_scene);
881 rtcSetGeometryTimeStepCount(geom_inst, 1);
882
883 Eigen::Matrix<float, 4, 4> T = trans.template cast<float>();
884 rtcSetGeometryTransform(geom_inst, 0, RTC_FORMAT_FLOAT4X4_COLUMN_MAJOR, T.data());
885 ensure_no_errors_internal();
886
887 if (is_mask_supported) {
888 rtcSetGeometryMask(
889 geom_inst,
890 m_visibility[instance_index] ? 0xFFFFFFFF : 0x00000000);
891 }
892 ensure_no_errors_internal();
893
894 rtcCommitGeometry(geom_inst);
895 unsigned rtc_instance_id = rtcAttachGeometry(m_embree_world_scene, geom_inst);
896 rtcReleaseGeometry(geom_inst);
897 la_runtime_assert(safe_cast<Index>(rtc_instance_id) == instance_index);
898 ensure_no_errors_internal();
899 }
900
901 rtcCommitScene(embree_mesh_scene);
902 ensure_no_errors_internal();
903 }
904 rtcCommitScene(m_embree_world_scene);
905 ensure_no_errors_internal();
906
907 m_need_rebuild = m_need_commit = false;
908 }

977 template <int IntersectionOrOcclusion> // 0: intersection, 1: occlusion
978 static void wrap_filter(const RTCFilterFunctionNArguments* args)
979 {
980 // Embree never actually calls a filter callback with different geometry or instance IDs
981 // So we can assume they are the same for all the hits in this batch. Also, every single
982 // mesh in this class is instanced (never used raw), so we can ignore geomID.
983 const auto* obj = reinterpret_cast<EmbreeRayCaster*>(args->geometryUserPtr);
984 auto rtc_inst_id = RTCHitN_instID(args->hit, args->N, 0, 0);
985 assert(rtc_inst_id < obj->m_instance_to_user_mesh.size());
986
987 auto filter = obj->m_filters[IntersectionOrOcclusion][rtc_inst_id];
988 if (!filter) {
989 return;
990 }
991
992 Index mesh_index = obj->m_instance_to_user_mesh[rtc_inst_id];
993 assert(mesh_index + 1 < obj->m_instance_index_ranges.size());
994 assert(mesh_index < safe_cast<Index>(obj->m_meshes.size()));
995 Index instance_index = rtc_inst_id - obj->m_instance_index_ranges[mesh_index];
996
997 // In case Embree's implementation changes in the future, the callback should be written
998 // generally, without assuming the single geometry/instance condition above.
999 Index4 mesh_index4;
1000 mesh_index4.fill(mesh_index);
1001 Index4 instance_index4;
1002 instance_index4.fill(instance_index);
1003
1004 // Call the wrapped filter with the indices specific to this object
1005 filter(obj, mesh_index4.data(), instance_index4.data(), args);
1006 }
1007
1013 template <int IntersectionOrOcclusion> // 0: intersection, 1: occlusion
1014 static void wrap_filter_and_mask(const RTCFilterFunctionNArguments* args)
1015 {
1016 // Embree never actually calls a filter callback with different geometry or instance IDs
1017 // So we can assume they are the same for all the hits in this batch. Also, every single
1018 // mesh in this class is instanced (never used raw), so we can ignore geomID.
1019 const auto* obj = reinterpret_cast<EmbreeRayCaster*>(args->geometryUserPtr);
1020 auto rtc_inst_id = RTCHitN_instID(args->hit, args->N, 0, 0);
1021 if (!obj->m_visibility[rtc_inst_id]) {
1022 // Object is invisible. Make the hits of all the rays with this object invalid.
1023 std::fill(args->valid, args->valid + args->N, 0);
1024 return;
1025 }
1026
1027 // Delegate to the regular filtering after having checked visibility
1028 wrap_filter<IntersectionOrOcclusion>(args);
1029 }

1084template <typename Scalar>
1085auto EmbreeRayCaster<Scalar>::query_closest_point(const Point& p) const -> ClosestPoint
1086{
1087 RTCPointQuery query;
1088 query.x = (float)(p.x());
1089 query.y = (float)(p.y());
1090 query.z = (float)(p.z());
1091 query.radius = std::numeric_limits<float>::max();
1092 query.time = 0.f;
1093 ensure_no_errors_internal();
1094
1095 ClosestPoint result;
1096
1097 // Callback to retrieve triangle corner positions
1098 result.populate_triangle =
1099 [&](unsigned mesh_index, unsigned facet_index, Point& v0, Point& v1, Point& v2) {
1100 // TODO: There's no way to call this->get_mesh<> since we need to template the function
1101 // by the (derived) type, which we don't know here... This means our only choice is so
1102 // use the float data instead of the (maybe) double point coordinate if available. Note
1103 // that we could also call rtcSetGeometryPointQueryFunction() when we register our mesh,
1104 // since we know the derived type at this point.
1105 const unsigned* face = m_int_data[mesh_index].data() + 3 * facet_index;
1106 const float* vertices = m_float_data[mesh_index].data();
1107 v0 = Point(vertices[3 * face[0]], vertices[3 * face[0] + 1], vertices[3 * face[0] + 2]);
1108 v1 = Point(vertices[3 * face[1]], vertices[3 * face[1] + 1], vertices[3 * face[1] + 2]);
1109 v2 = Point(vertices[3 * face[2]], vertices[3 * face[2] + 1], vertices[3 * face[2] + 2]);
1110 };
1111
1112 {
1113 RTCPointQueryContext context;
1114 rtcInitPointQueryContext(&context);
1115 rtcPointQuery(
1116 m_embree_world_scene,
1117 &query,
1118 &context,
1119 &embree_closest_point<Scalar>,
1120 reinterpret_cast<void*>(&result));
1121 assert(
1122 result.mesh_index != RTC_INVALID_GEOMETRY_ID ||
1123 result.facet_index != RTC_INVALID_GEOMETRY_ID);
1124 assert(
1125 result.mesh_index != invalid<unsigned>() || result.facet_index != invalid<unsigned>());
1126 }
1127 ensure_no_errors_internal();
1128
1129 return result;
1130}
